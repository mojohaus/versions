42      @Test
43      public void test4DigitVersions() throws Exception {
44          ArtifactVersion[] versions = versions( "1.0.0.1", "1.0.0.2", "2.121.2.1", "2.100.0.1", "3.1.0.1", "1.1.1");
45          final DefaultArtifact artifact =
46                  new DefaultArtifact( "group", "artifact", VersionRange.createFromVersionSpec( "[1.0,3.0]" ), "foo", "bar",
47                                       "jar", new DefaultArtifactHandler() );
48          // TODO This should also work for the MavenVersionComparator when using maven 3.x libraries
49          ArtifactVersions instance =
50                  new ArtifactVersions(artifact, Arrays.asList( versions ), new MercuryVersionComparator() );
51          assertEquals( "artifact", instance.getArtifactId() );
52          assertEquals( "group", instance.getGroupId() );
53          assertThat(instance.getVersions(),
54                     Matchers.arrayContaining(versions( "1.0.0.1", "1.0.0.2", "1.1.1", "2.100.0.1", "2.121.2.1",
55                                                        "3.1.0.1" )));
56          assertThat(instance.getVersions( new DefaultArtifactVersion( "1.1" ), null ),
57                     Matchers.arrayContaining(versions("1.1.1", "2.100.0.1", "2.121.2.1", "3.1.0.1")));
58  
59          assertThat(instance.getVersions( new DefaultArtifactVersion( "1.0.0.2" ), null ),
60                            //Matchers.arrayContaining(versions("1.1.1", "2.121.2.1", "2.100.0.1", "3.1.0.1")));
61                     Matchers.arrayContaining(versions("1.1.1", "2.100.0.1", "2.121.2.1", "3.1.0.1")));
62  
63          assertEquals( new DefaultArtifactVersion( "2.121.2.1" ),
64                        instance.getNewestVersion( new DefaultArtifactVersion( "1.0" ),
65                                                   new DefaultArtifactVersion( "3.0" ) ) );
66          assertNull(
67                  instance.getNewestVersion( new DefaultArtifactVersion( "1.1.1" ),
68                                             new DefaultArtifactVersion( "2.0" ) ) );
69      }

71      @Test
72      public void testSmokes()
73          throws Exception
74      {
75          ArtifactVersion[] versions = versions( "1.0", "3.0", "1.1", "1.0", "1.0.1" );
76          final DefaultArtifact artifact =
77              new DefaultArtifact( "group", "artifact", VersionRange.createFromVersionSpec( "[1.0,3.0]" ), "foo", "bar",
78                                   "jar", new DefaultArtifactHandler() );
79          ArtifactVersions instance =
80              new ArtifactVersions( artifact, Arrays.asList( versions ), new MavenVersionComparator() );
81          assertEquals( "artifact", instance.getArtifactId() );
82          assertEquals( "group", instance.getGroupId() );
83          assertArrayEquals(
84              versions( "1.0", "1.0.1", "1.1", "3.0" ),
85              instance.getVersions() );
86          assertArrayEquals( versions( "3.0" ),
87                             instance.getVersions( new DefaultArtifactVersion( "1.1" ), null ) );
88          assertArrayEquals(
89              versions( "1.1", "3.0" ),
90              instance.getVersions( new DefaultArtifactVersion( "1.0.1" ), null ) );
91          assertEquals( new DefaultArtifactVersion( "1.1" ),
92                        instance.getNewestVersion( new DefaultArtifactVersion( "1.0" ),
93                                                   new DefaultArtifactVersion( "3.0" ) ) );
94          assertNull(
95              instance.getNewestVersion( new DefaultArtifactVersion( "1.1" ), new DefaultArtifactVersion( "3.0" ) ) );
96      }
97  
98      private ArtifactVersion[] versions(String... versions) {
99          ArtifactVersion[] artifactVersions = new ArtifactVersion[versions.length];
100         for ( int i = 0; i < versions.length; i++ ) {
101             artifactVersions[i] = new DefaultArtifactVersion( versions[i] );
102         }
103         return artifactVersions;
104     }
