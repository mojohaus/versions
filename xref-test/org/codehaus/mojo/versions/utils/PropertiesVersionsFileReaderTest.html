17      private static final String TEST_PROPERTIES_FILE =
18          "src/test/resources/org/codehaus/mojo/versions/utils/testPropertiesVersionsFile.properties";
19  
20      @Test
21      public void testRead() throws IOException
22      {
23          PropertiesVersionsFileReader reader = new PropertiesVersionsFileReader( TEST_PROPERTIES_FILE );
24          reader.read();
25  
26          int numberOfPropertiesConfig = 3;
27          assertTrue( equalsCvsUnordered( "booking-api.version,booking-lib.version,be-air-impl.version",
28                                          reader.getProperties() ) );
29          assertEquals( numberOfPropertiesConfig, reader.getPropertiesConfig().length );
30      }
31  
32      private boolean equalsCvsUnordered( String csvExpected, String csvActual )
33      {
34          if ( StringUtils.isEmpty( csvExpected ) )
35          {
36              return false;
37          }
38          Set<String> listExpected = new HashSet<>( Arrays.asList( csvExpected.split( "," ) ) );
39          Set<String> listActual = new HashSet<>( Arrays.asList( csvActual.split( "," ) ) );
40          return listExpected.equals( listActual );
41      }
