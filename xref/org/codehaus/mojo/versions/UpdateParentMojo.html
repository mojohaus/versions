22  import javax.xml.stream.XMLStreamException;
23  
24  import org.apache.maven.artifact.Artifact;
25  import org.apache.maven.artifact.metadata.ArtifactMetadataRetrievalException;
26  import org.apache.maven.artifact.versioning.ArtifactVersion;
27  import org.apache.maven.artifact.versioning.InvalidVersionSpecificationException;
28  import org.apache.maven.artifact.versioning.VersionRange;
29  import org.apache.maven.model.Dependency;
30  import org.apache.maven.plugin.MojoExecutionException;
31  import org.apache.maven.plugin.MojoFailureException;
32  import org.apache.maven.plugins.annotations.Mojo;
33  import org.apache.maven.plugins.annotations.Parameter;
34  import org.codehaus.mojo.versions.api.PomHelper;
35  import org.codehaus.mojo.versions.rewriting.ModifiedPomXMLEventReader;
36  
37  /**
38   * Sets the parent version to the latest parent version.
39   *
40   * @author Stephen Connolly
41   * @since 1.0-alpha-1
42   */
43  @Mojo( name = "update-parent",
44         requiresProject = true,
45         requiresDirectInvocation = true,
46         threadSafe = true )
47  public class UpdateParentMojo extends AbstractVersionsUpdaterMojo
48  {
49  
50      // ------------------------------ FIELDS ------------------------------
51  
52      /**
53       * Version specification to control artifact resolution.
54       *
55       * @since 1.0-alpha-1
56       */
57      @Parameter( property = "parentVersion" )
58      protected String parentVersion = null;
59  
60      /**
61       * to update parent version by force when it is RELEASE or LATEST
62       *
63       * @since 2.9
64       */
65      @Parameter( property = "forceUpdate", defaultValue = "false" )
66      protected boolean forceUpdate = false;
67  
68      /**
69       * <p>Whether to downgrade a snapshot dependency if <code>allowSnapshots</code> is <code>false</code>
70       * and there exists a version within the range fulfilling the criteria.</p>
71       * <p>Default <code>false</code></p>
72       *
73       * @since 2.12.0
74       */
75      @Parameter( property = "allowDowngrade",
76                  defaultValue = "false" )
77      protected boolean allowDowngrade;
78  
79      // -------------------------- OTHER METHODS --------------------------
80  
81      /**
82       * @param pom the pom to update.
83       * @throws MojoExecutionException when things go wrong
84       * @throws MojoFailureException   when things go wrong in a very bad way
85       * @throws XMLStreamException     when things go wrong with XML streaming
86       * @see AbstractVersionsUpdaterMojo#update(ModifiedPomXMLEventReader)
87       * @since 1.0-alpha-1
88       */
89      protected void update( ModifiedPomXMLEventReader pom )
90          throws MojoExecutionException, MojoFailureException, XMLStreamException
91      {
92          if ( getProject().getParent() == null )
93          {
94              getLog().info( "Project does not have a parent" );
95              return;
96          }
97  
98          if ( reactorProjects.contains( getProject().getParent() ) )
99          {
100             getLog().info( "Project's parent is part of the reactor" );
101             return;
102         }
103 
104         String currentVersion = getProject().getParent().getVersion();
105         String version = currentVersion;
106 
107         if ( parentVersion != null )
108         {
109             version = parentVersion;
110         }
111 
112         Dependency dependency = new Dependency();
113         dependency.setGroupId( getProject().getParent().getGroupId() );
114         dependency.setArtifactId( getProject().getParent().getArtifactId() );
115         dependency.setVersion( version );
116         dependency.setType( "pom" );
117         Artifact artifact = getHelper().createDependencyArtifact( dependency );
118 
119         VersionRange versionRange;
120         try
121         {
122             versionRange = VersionRange.createFromVersionSpec( version );
123             if ( versionRange.getRecommendedVersion() != null )
124             {
125                 versionRange = versionRange.restrict(
126                         VersionRange.createFromVersionSpec( "[" + versionRange.getRecommendedVersion() + ",)" ) );
127             }
128         }
129         catch ( InvalidVersionSpecificationException e )
130         {
131             throw new MojoExecutionException( "Invalid version range specification: " + version, e );
132         }
133 
134         ArtifactVersion artifactVersion;
135         try
136         {
137             artifactVersion = findLatestVersion( artifact, versionRange, false, true,
138                     allowDowngrade );
139         }
140         catch ( ArtifactMetadataRetrievalException e )
141         {
142             throw new MojoExecutionException( e.getMessage(), e );
143         }
144 
145 
146         if ( !shouldApplyUpdate( artifact, currentVersion, artifactVersion, forceUpdate ) )
147         {
148             return;
149         }
150 
151         getLog().info( "Updating parent from " + currentVersion + " to " + artifactVersion.toString() );
152 
153         if ( PomHelper.setProjectParentVersion( pom, artifactVersion.toString() ) )
154         {
155             getLog().debug( "Made an update from " + currentVersion + " to " + artifactVersion );
156 
157             this.getChangeRecorder().recordUpdate( "updateParent", artifact.getGroupId(), artifact.getArtifactId(),
158                                                    currentVersion, artifactVersion.toString() );
159         }
160     }
161 
162 }
