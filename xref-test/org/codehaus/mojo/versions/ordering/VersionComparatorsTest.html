22  import org.apache.maven.artifact.versioning.ArtifactVersion;
23  import org.apache.maven.artifact.versioning.DefaultArtifactVersion;
24  import org.junit.Test;
25  
26  import static org.junit.Assert.assertTrue;
27  
28  public class VersionComparatorsTest
29  {
30      private final String[] versionDataset = {
31          "1",
32          "1.0",
33          "1.0.0",
34          "1.0.0-1",
35          "1.0.0-sp1",
36          "foobar",
37          "1-alpha-1",
38      };
39  
40      @Test
41      public void testMavenVersionComparator() throws InvalidSegmentException
42      {
43          assertVersions( new MavenVersionComparator() );
44      }
45  
46      @Test
47      public void testMercuryVersionComparator() throws InvalidSegmentException
48      {
49          assertVersions( new MercuryVersionComparator() );
50      }
51  
52      @Test
53      public void testNumericVersionComparator() throws InvalidSegmentException
54      {
55          assertVersions( new NumericVersionComparator() );
56      }
57  
58      public void assertVersions( VersionComparator instance ) throws InvalidSegmentException
59      {
60          for ( int i = 0; i < versionDataset.length; i++ )
61          {
62              assertLater( versionDataset[i], instance );
63              assertLater( versionDataset[i] + "-SNAPSHOT", instance );
64          }
65      }
66  
67      public void assertLater( String version, VersionComparator instance ) throws InvalidSegmentException
68      {
69          ArtifactVersion v1 = new DefaultArtifactVersion( version );
70          int count = instance.getSegmentCount( v1 );
71          for ( int i = 0; i < count; i++ )
72          {
73              ArtifactVersion v2 = instance.incrementSegment( v1, i );
74              assertTrue( v1 + " < " + v2, instance.compare( v1, v2 ) < 0 );
75          }
76      }
77  }
