171     /**
172      * Try to find the dependency artifact that matches the given dependency.
173      *
174      * @param dependency
175      * @return
176      * @since 1.0-alpha-3
177      */
178     protected Artifact findArtifact( Dependency dependency )
179     {
180         if ( getProject().getDependencyArtifacts() == null )
181         {
182             return null;
183         }
184         Iterator iter = getProject().getDependencyArtifacts().iterator();
185         while ( iter.hasNext() )
186         {
187             Artifact artifact = (Artifact) iter.next();
188             if ( compare( artifact, dependency ) )
189             {
190                 return artifact;
191             }
192         }
193         return null;
194     }
195 
196     /**
197      * Try to find the dependency artifact that matches the given dependency.
198      *
199      * @param dependency
200      * @return
201      * @since 1.0-alpha-3
202      */
203     protected Artifact toArtifact( Dependency dependency )
204         throws MojoExecutionException
205     {
206         Artifact artifact = findArtifact( dependency );
207         if ( artifact == null )
208         {
209             try
210             {
211                 return getHelper().createDependencyArtifact( dependency );
212             }
213             catch ( InvalidVersionSpecificationException e )
214             {
215                 throw new MojoExecutionException( e.getMessage(), e );
216             }
217         }
218         return artifact;
219     }

221     protected String toString( Dependency d )
222     {
223         StringBuilder buf = new StringBuilder();
224         buf.append( d.getGroupId() );
225         buf.append( ':' );
226         buf.append( d.getArtifactId() );
227         if ( d.getType() != null && d.getType().length() > 0 )
228         {
229             buf.append( ':' );
230             buf.append( d.getType() );
231         }
232         else
233         {
234             buf.append( ":jar" );
235         }
236         if ( d.getClassifier() != null && d.getClassifier().length() > 0 )
237         {
238             buf.append( ':' );
239             buf.append( d.getClassifier() );
240         }
241         if ( d.getVersion() != null && d.getVersion().length() > 0 )
242         {
243             buf.append( ":" );
244             buf.append( d.getVersion() );
245         }
246         return buf.toString();
247     }

257     {
258         Iterator iter = reactorProjects.iterator();
259         while ( iter.hasNext() )
260         {
261             MavenProject project = (MavenProject) iter.next();
262             if ( compare( project, dependency ) )
263             {
264                 return true;
265             }
266         }
267         return false;

271     /**
272      * Compare a project to a dependency. Returns true only if the groupId and artifactId are all
273      * equal.
274      *
275      * @param project the project
276      * @param dep     the dependency
277      * @return true if project and dep refer to the same artifact
278      */
279     private boolean compare( MavenProject project, Dependency dep )
280     {
281         if ( !StringUtils.equals( project.getGroupId(), dep.getGroupId() ) )
282         {
283             return false;
284         }
285         if ( !StringUtils.equals( project.getArtifactId(), dep.getArtifactId() ) )
286         {
287             return false;
288         }
289         return true;
290     }
291 
292     /**
293      * Compare and artifact to a dependency. Returns true only if the groupId, artifactId, type, and classifier are all
294      * equal.
295      *
296      * @param artifact
297      * @param dep
298      * @return true if artifact and dep refer to the same artifact
299      */
300     private boolean compare( Artifact artifact, Dependency dep )
301     {
302         if ( !StringUtils.equals( artifact.getGroupId(), dep.getGroupId() ) )
303         {
304             return false;
305         }
306         if ( !StringUtils.equals( artifact.getArtifactId(), dep.getArtifactId() ) )
307         {
308             return false;
309         }
310         if ( !StringUtils.equals( artifact.getType(), dep.getType() ) )
311         {
312             return false;
313         }
314         if ( !StringUtils.equals( artifact.getClassifier(), dep.getClassifier() ) )
315         {
316             return false;
317         }
318         return true;
319     }

321     /**
322      * Determine if the artifact is included in the list of artifacts to be processed.
323      *
324      * @param artifact The artifact we want to check.
325      * @return true if the artifact should be processed, false otherwise.
326      */
327     protected boolean isIncluded( Artifact artifact )
328     {
329         boolean result = true;
330 
331         ArtifactFilter includesFilter = this.getIncludesArtifactFilter();
332 
333         if ( includesFilter != null )
334         {
335             result = includesFilter.include( artifact );
336         }
337 
338         ArtifactFilter excludesFilter = this.getExcludesArtifactFilter();
339 
340         if ( excludesFilter != null )
341         {
342             result = result && excludesFilter.include( artifact );
343         }
344 
345         return result;
346 
347     }
348 
349     private ArtifactFilter getIncludesArtifactFilter()
350     {
351         if ( includesFilter == null && ( includes != null || includesList != null ) )
352         {
353             List patterns = new ArrayList();
354             if ( this.includesList != null )
355             {
356                 patterns.addAll( separatePatterns( includesList ) );
357             }
358             else if ( includes != null )
359             {
360                 patterns.addAll( Arrays.asList( includes ) );
361             }
362             includesFilter = new PatternIncludesArtifactFilter( patterns );
363         }
364         return includesFilter;
365     }
366 
367     private ArtifactFilter getExcludesArtifactFilter()
368     {
369         if ( excludesFilter == null && ( excludes != null || excludesList != null ) )
370         {
371             List patterns = new ArrayList();
372             if ( excludesList != null )
373             {
374                 patterns.addAll( separatePatterns( excludesList ) );
375             }
376             else if ( excludes != null )
377             {
378                 patterns.addAll( Arrays.asList( excludes ) );
379             }
380             excludesFilter = new PatternExcludesArtifactFilter( patterns );
381         }
382         return excludesFilter;
383     }

385     /**
386      * To handle multiple includes with version range like "group:artifact:jar:[1.0.0,2.2)",
387      * we have to use a parsing a little bit more complex than split().
388      *
389      * @param includeString the string to parse
390      * @return list of patterns
391      */
392     protected List separatePatterns( String includeString )
393     {
394         if ( includeString == null )
395         {
396             return Collections.EMPTY_LIST;
397         }
398 
399         List patterns = new ArrayList();
400         int indexOf = nextCommaIndex( includeString );
401         while ( indexOf >= 0 )
402         {
403             patterns.add( includeString.substring( 0, indexOf ) );
404             includeString = includeString.substring( indexOf + 1 );
405             indexOf = nextCommaIndex( includeString );
406         }
407         patterns.add( includeString );
408 
409         return patterns;
410     }
411 
412     private int nextCommaIndex( final String includeString )
413     {
414 
415         int indexOfComma = includeString.indexOf( ',' );
416         int nextRangeStartDelimiterIndex = findFirstChar( includeString, START_RANGE_CHARS );
417         if ( nextRangeStartDelimiterIndex >= 0 )
418         {
419             if ( !( indexOfComma >= 0 && indexOfComma < nextRangeStartDelimiterIndex ) )
420             {
421                 int nextStopDelimiterIndex = findFirstChar( includeString, END_RANGE_CHARS );
422 
423                 // recursive call
424                 int tmp = nextCommaIndex( includeString.substring( nextStopDelimiterIndex + 1 ) );
425                 indexOfComma = ( tmp >= 0 ) ? nextStopDelimiterIndex + 1 + tmp : -1;
426             }
427         }
428         return indexOfComma;
429 
430     }
431 
432     private int findFirstChar( final String includeString, final String chars )
433     {
434         int nextRangeStartDelimiterIndex = -1;
435 
436         char[] delimiters = chars.toCharArray();
437         for ( int i = 0; i < delimiters.length; i++ )
438         {
439             int index = includeString.indexOf( delimiters[i] );
440             if ( index >= 0 && nextRangeStartDelimiterIndex >= 0 )
441             {
442                 nextRangeStartDelimiterIndex = Math.min( index, nextRangeStartDelimiterIndex );
443             }
444             else
445             {
446                 if ( index >= 0 )
447                 {
448                     nextRangeStartDelimiterIndex = index;
449                 }
450             }
451         }
452         return nextRangeStartDelimiterIndex;
453     }
