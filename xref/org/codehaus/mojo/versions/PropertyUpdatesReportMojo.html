44  /**
45   * Generates a report of available updates for properties of a project which are linked to the dependencies and/or
46   * plugins of a project.
47   *
48   * @author Stephen Connolly
49   * @since 1.0-beta-1
50   */
51  @Mojo( name = "property-updates-report", requiresDependencyResolution = ResolutionScope.RUNTIME,
52         threadSafe = true )
53  public class PropertyUpdatesReportMojo extends AbstractVersionsReport<PropertyUpdatesModel>
54  {
55      /**
56       * Any restrictions that apply to specific properties.
57       *
58       * @since 1.0-beta-1
59       */
60      @Parameter
61      private Property[] properties;
62  
63      /**
64       * A comma separated list of properties to include in the report.
65       *
66       * @since 1.0-beta-1
67       */
68      @Parameter( property = "includeProperties" )
69      private String includeProperties = null;
70  
71      /**
72       * A comma separated list of properties to not include in the report.
73       *
74       * @since 1.0-beta-1
75       */
76      @Parameter( property = "excludeProperties" )
77      private String excludeProperties = null;
78  
79      /**
80       * Whether properties linking versions should be auto-detected or not.
81       *
82       * @since 1.0-beta-1
83       */
84      @Parameter( property = "autoLinkItems", defaultValue = "true" )
85      private boolean autoLinkItems;
86  
87      @Inject
88      protected PropertyUpdatesReportMojo( I18N i18n, RepositorySystem repositorySystem,
89                                           ArtifactResolver artifactResolver,
90                                           ArtifactMetadataSource artifactMetadataSource, WagonManager wagonManager,
91                                           ReportRendererFactory rendererFactory )
92      {
93          super( i18n, repositorySystem, artifactResolver, artifactMetadataSource, wagonManager, rendererFactory );
94      }
95  
96      /**
97       * {@inheritDoc}
98       */
99      public boolean isExternalReport()
100     {
101         return false;
102     }
103 
104     /**
105      * {@inheritDoc}
106      */
107     public boolean canGenerateReport()
108     {
109         return haveBuildProperties();
110     }
111 
112     private boolean haveBuildProperties()
113     {
114         return getProject().getProperties() != null && !getProject().getProperties().isEmpty();
115     }
116 
117     protected void doGenerateReport( Locale locale, Sink sink )
118         throws MavenReportException
119     {
120         final Map<Property, PropertyVersions> updateSet = new TreeMap<>( PropertyComparator.INSTANCE );
121         try
122         {
123             updateSet.putAll( getHelper().getVersionPropertiesMap( getProject(), properties, includeProperties,
124                                                                    excludeProperties, autoLinkItems ) );
125         }
126         catch ( MojoExecutionException e )
127         {
128             throw new MavenReportException( e.getMessage(), e );
129         }
130         rendererFactory.createReportRenderer( getOutputName(), getSink(), locale,
131                 new PropertyUpdatesModel( updateSet ) ).render();
132     }
133 
134     /**
135      * {@inheritDoc}
136      */
137     public String getOutputName()
138     {
139         return "property-updates-report";
140     }
141 }
