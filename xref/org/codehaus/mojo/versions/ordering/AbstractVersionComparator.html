22  import org.apache.maven.artifact.versioning.ArtifactVersion;
23  
24  /**
25   * Base class for version comparators.
26   *
27   * @since 1.0-beta-1
28   */
29  public abstract class AbstractVersionComparator
30      implements VersionComparator
31  {
32      /**
33       * {@inheritDoc}
34       */
35      public abstract int compare( ArtifactVersion o1, ArtifactVersion o2 );
36  
37      /**
38       * {@inheritDoc}
39       */
40      public final int getSegmentCount( ArtifactVersion v )
41      {
42          if ( v == null )
43          {
44              return 0;
45          }
46          if ( VersionComparators.isSnapshot( v ) )
47          {
48              return innerGetSegmentCount( VersionComparators.stripSnapshot( v ) );
49          }
50          return innerGetSegmentCount( v );
51  
52      }
53  
54      protected abstract int innerGetSegmentCount( ArtifactVersion v );
55  
56      /**
57       * Returns a hash code value for the comparator class.
58       *
59       * @return the hash code.
60       */
61      public int hashCode()
62      {
63          return getClass().hashCode();
64      }
65  
66      /**
67       * Returns true if this object is the same type of comparator as the parameter.
68       *
69       * @param obj the reference object with which to compare.
70       * @return <code>true</code> if this object is the same as the obj argument; <code>false</code> otherwise.
71       * @see #hashCode()
72       * @see java.util.Hashtable
73       */
74      public boolean equals( Object obj )
75      {
76          return obj == this || ( obj != null && getClass().equals( obj.getClass() ) );
77      }
78  
79  }
