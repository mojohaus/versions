24  	public void testSeparatePatternsWithNull() {

29  	public void testSeparatePatternsWithSinglePattern() {
30  		List patterns = mojo.separatePatterns("group:artifact:type:version");
31  		assertEquals(1, patterns.size());
32  		assertEquals("group:artifact:type:version", patterns.get(0));
33  	}

35  	public void testSeparatePatternWithSingleRange() {
36  		List patterns = mojo.separatePatterns("group:artifact:type:[1.0.2,2.0.0]");
37  		assertEquals(1, patterns.size());
38  		assertEquals("group:artifact:type:[1.0.2,2.0.0]", patterns.get(0));
39  
40  		patterns = mojo.separatePatterns("group:artifact:type:(1.0.2,2.0.0]");
41  		assertEquals(1, patterns.size());
42  		assertEquals("group:artifact:type:(1.0.2,2.0.0]", patterns.get(0));
43  	}

45  	public void testSeparatePatternWithSeveralPatternsAndRanges() {
46  		List patterns = mojo.separatePatterns("group:artifact:type:[1.0.2,2.0.0),group2:artifact:type:(1.0.2,2.0.0]");
47  		assertEquals(2, patterns.size());
48  		assertEquals("group:artifact:type:[1.0.2,2.0.0)", patterns.get(0));
49  		assertEquals("group2:artifact:type:(1.0.2,2.0.0]", patterns.get(1));
50  	}
