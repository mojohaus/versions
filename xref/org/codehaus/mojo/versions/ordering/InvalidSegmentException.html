19  /**
20   * Represents an invalid segment being identified within a version.
21   */
22  public class InvalidSegmentException
23      extends RuntimeException
24  {
25      /**
26       * Constructs a new exception.
27       *
28       * @param segment      the invalid segment index.
29       * @param segmentCount the number of segments.
30       * @param version      the version string.
31       */
32      public InvalidSegmentException( int segment, int segmentCount, String version )
33      {
34          super(
35              String.format( "Invalid segment, %d, for the %d segment version: '%s'", segment, segmentCount, version ) );
36      }
37  }
