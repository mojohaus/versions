22  import java.util.Locale;
23  
24  import org.apache.maven.doxia.sink.Sink;
25  import org.codehaus.mojo.versions.api.ReportRenderer;
26  
27  /**
28   * Factory for report renderers
29   */
30  public interface ReportRendererFactory
31  {
32      /**
33       * Creates a new {@linkplain ReportRenderer} instance for the given report renderer name.
34       *
35       * @param <T> concrete class of the report renderer
36       * @param reportName name of the report to generate
37       * @param sink sink to use for rendering
38       * @param locale locale to use for rendering
39       * @param model data to render
40       * @return new report renderer
41       * @throws IllegalArgumentException thrown if the report with the given name could not be found
42       */
43      <T extends ReportRenderer, U> T createReportRenderer( String reportName, Sink sink, Locale locale, U model )
44              throws IllegalArgumentException;
45  }
