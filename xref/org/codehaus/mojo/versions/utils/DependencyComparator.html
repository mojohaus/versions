22  import java.util.Comparator;
23  
24  import org.apache.maven.model.Dependency;
25  
26  /**
27   * A comparator used to sort dependencies by group id, artifact id and finally version.
28   *
29   * @since 1.0-alpha-1
30   */
31  public class DependencyComparator
32      implements Comparator<Dependency>
33  {
34  
35      /**
36       * @param d1 the first dependency
37       * @param d2 the second dependency.
38       * @return the comparison result
39       * @see java.util.Comparator#compare(Object, Object)
40       * @since 1.0-alpha-1
41       */
42      public int compare( Dependency d1, Dependency d2 )
43      {
44          int r = d1.getGroupId().compareTo( d2.getGroupId() );
45          if ( r == 0 )
46          {
47              r = d1.getArtifactId().compareTo( d2.getArtifactId() );
48          }
49          if ( r == 0 )
50          {
51              String v1 = d1.getVersion();
52              String v2 = d2.getVersion();
53              if ( v1 == null )
54              {
55                  // hope I got the +1/-1 the right way around
56                  return v2 == null ? 0 : -1;
57              }
58              if ( v2 == null )
59              {
60                  return 1;
61              }
62              r = v1.compareTo( v2 );
63          }
64          return r;
65      }
66  
67  }
