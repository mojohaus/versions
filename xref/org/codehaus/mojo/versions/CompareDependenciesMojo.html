48  /**
49   * Compare dependency versions of the current project to dependencies or dependency management of a remote repository
50   * project. Can optionally update locally the project instead of reporting the comparison
51   *
52   * @author Paul Gier
53   * @since 1.3
54   */
55  @Mojo( name = "compare-dependencies", requiresProject = true, requiresDirectInvocation = true )
56  public class CompareDependenciesMojo
57      extends AbstractVersionsDependencyUpdaterMojo
58  {

60      /**
61       * The width to pad info messages.
62       *
63       * @since 1.0-alpha-1
64       */
65      private static final int INFO_PAD_SIZE = 68;

67      /**
68       * The groupId, artifactId, and version of the remote project (POM) to which we are comparing. This should be in the
69       * form "groupId:artifactId:version"
70       */
71      @Parameter( property = "remotePom", required = true )
72      protected String remotePom;
73  
74      /**
75       * Ignore the list of remote dependencies and only compare the remote dependencyManagement
76       */
77      @Parameter( property = "ignoreRemoteDependencies", defaultValue = "false" )
78      protected boolean ignoreRemoteDependencies;
79  
80      /**
81       * Ignore the remote dependency management and only check against the actual dependencies of the remote project
82       */
83      @Parameter( property = "ignoreRemoteDependencyManagement", defaultValue = "false" )
84      protected boolean ignoreRemoteDependencyManagement;
85  
86      /**
87       * Update dependency versions in the current POM.
88       */
89      @Parameter( property = "updateDependencies", defaultValue = "false" )
90      protected boolean updateDependencies;
91  
92      /**
93       * Update dependency versions stored in properties
94       *
95       * @parameter property="updatePropertyVersions" default-value="false"
96       */
97      @Parameter( property = "updatePropertyVersions", defaultValue = "false" )
98      protected boolean updatePropertyVersions;

101      * Display the dependency version differences on the command line, but do not update the versions in the current
102      * pom. If updateDependencies is set to "true" this will automatically be set to false.

104     @Parameter( property = "reportMode", defaultValue = "true" )
105     protected boolean reportMode;
106 
107     /**
108      * If the output file is set, the diff report will be written to this file.

111     protected File reportOutputFile;

121     /**
122      * @param pom the pom to update.
123      * @throws org.apache.maven.plugin.MojoExecutionException Something wrong with the plugin itself
124      * @throws org.apache.maven.plugin.MojoFailureException The plugin detected an error in the build
125      * @throws javax.xml.stream.XMLStreamException when things go wrong with XML streaming
126      * @see AbstractVersionsUpdaterMojo#update(org.codehaus.mojo.versions.rewriting.ModifiedPomXMLEventReader)
127      */
128     protected void update( ModifiedPomXMLEventReader pom )
129         throws MojoExecutionException, MojoFailureException, XMLStreamException
130     {
131         if ( this.ignoreRemoteDependencies && this.ignoreRemoteDependencyManagement )
132         {
133             throw new MojoFailureException( " ignoreRemoteDependencies and ignoreRemoteDependencyManagement"
134                 + "are both set to true.  At least one of these needs to be false " );
135         }
136 
137         if ( updateDependencies )
138         {
139             reportMode = false;
140         }
141 
142         String[] remotePomParts = this.remotePom.split( ":" );
143         if ( remotePomParts.length != 3 )
144         {
145             throw new MojoFailureException( " Invalid format for remotePom: " + remotePom );
146         }
147         String rGroupId = remotePomParts[0];
148         String rArtifactId = remotePomParts[1];
149         String rVersion = remotePomParts[2];
150 
151         Dependency remoteDependency = new Dependency();
152         remoteDependency.setGroupId( rGroupId );
153         remoteDependency.setArtifactId( rArtifactId );
154         remoteDependency.setVersion( rVersion );
155 
156         Artifact remoteArtifact = this.toArtifact( remoteDependency );
157         MavenProject remoteMavenProject = null;
158         try
159         {
160             remoteMavenProject =
161                 mavenProjectBuilder.buildFromRepository( remoteArtifact, remoteArtifactRepositories, localRepository );
162         }
163         catch ( ProjectBuildingException e )
164         {
165             throw new MojoExecutionException( "Unable to build remote project " + remoteArtifact, e );
166         }
167 
168         Map<String, Dependency> remoteDepsMap = new HashMap<String, Dependency>();
169         if ( !ignoreRemoteDependencyManagement )
170         {
171             List<Dependency> remoteProjectDepMgmtDeps = ( remoteMavenProject.getDependencyManagement() == null ) ? null
172                             : remoteMavenProject.getDependencyManagement().getDependencies();
173             mapDependencies( remoteDepsMap, remoteProjectDepMgmtDeps );
174         }
175         if ( !ignoreRemoteDependencies )
176         {
177             List<Dependency> remoteProjectDeps = remoteMavenProject.getDependencies();
178             mapDependencies( remoteDepsMap, remoteProjectDeps );
179         }
180 
181         List<String> totalDiffs = new ArrayList<String>();
182         List<String> propertyDiffs = new ArrayList<String>();
183         if ( getProject().getDependencyManagement() != null && isProcessingDependencyManagement() )
184         {
185             List<String> depManDiffs =
186                 compareVersions( pom, getProject().getDependencyManagement().getDependencies(), remoteDepsMap );
187             totalDiffs.addAll( depManDiffs );
188         }
189         if ( getProject().getDependencies() != null && isProcessingDependencies() )
190         {
191             List<String> depDiffs = compareVersions( pom, getProject().getDependencies(), remoteDepsMap );
192             totalDiffs.addAll( depDiffs );
193         }
194         if ( updatePropertyVersions )
195         {
196             Map<Property, PropertyVersions> versionProperties =
197                 this.getHelper().getVersionPropertiesMap( getProject(), null, null, null, true );
198             List<String> diff = updatePropertyVersions( pom, versionProperties, remoteDepsMap );
199             propertyDiffs.addAll( diff );
200         }
201 
202         if ( reportMode )
203         {
204             getLog().info( "The following differences were found:" );
205             if ( totalDiffs.size() == 0 )
206             {
207                 getLog().info( "  none" );
208             }
209             else
210             {
211                 for ( String totalDiff : totalDiffs )
212                 {
213                     getLog().info( "  " + totalDiff );
214                 }
215             }
216             getLog().info( "The following property differences were found:" );
217             if ( propertyDiffs.size() == 0 )
218             {
219                 getLog().info( "  none" );
220             }
221             else
222             {
223                 for ( String propertyDiff : propertyDiffs )
224                 {
225                     getLog().info( "  " + propertyDiff );
226                 }
227             }
228         }
229 
230         if ( reportOutputFile != null )
231         {
232             writeReportFile( totalDiffs, propertyDiffs );
233         }
234 
235     }

237     /**
238      * Compare the dependency versions of the current project with the dependency versions of a remote project
239      *
240      * @throws XMLStreamException
241      */
242     private List<String> compareVersions( ModifiedPomXMLEventReader pom, List<Dependency> dependencies,
243                                           Map<String, Dependency> remoteDependencies )
244         throws MojoExecutionException, XMLStreamException
245     {
246         List<String> updates = new ArrayList<>();
247         for ( Dependency dep : dependencies )
248         {
249             Artifact artifact = this.toArtifact( dep );
250             if ( !isIncluded( artifact ) )
251             {
252                 continue;
253             }
254 
255             Dependency remoteDep = remoteDependencies.get( dep.getManagementKey() );
256             if ( remoteDep != null )
257             {
258                 String remoteVersion = remoteDep.getVersion();
259 
260                 if ( !dep.getVersion().equals( remoteVersion ) )
261                 {
262                     StringBuilder buf = writeDependencyDiffMessage( dep, remoteVersion );
263                     updates.add( buf.toString() );
264                     if ( !reportMode )
265                     {
266                         if ( PomHelper.setDependencyVersion( pom, dep.getGroupId(), dep.getArtifactId(),
267                                                              dep.getVersion(), remoteVersion,
268                                                              getProject().getModel() ) )
269                         {
270                             getLog().info( "Updated " + toString( dep ) + " to version " + remoteVersion );
271                         }
272                     }
273 
274                 }
275 
276             }
277         }
278 
279         return updates;
280 
281     }

283     /**
284      * Updates the properties holding a version if necessary.
285      */
286     private List<String> updatePropertyVersions( ModifiedPomXMLEventReader pom,
287                                                  Map<Property, PropertyVersions> versionProperties,
288                                                  Map<String, Dependency> remoteDependencies )
289         throws XMLStreamException
290     {
291         List<String> result = new ArrayList<>();
292         for ( Map.Entry<Property, PropertyVersions> entry : versionProperties.entrySet() )
293         {
294             Property property = entry.getKey();
295             PropertyVersions version = entry.getValue();
296 
297             String candidateVersion = computeCandidateVersion( remoteDependencies, property, version );
298             if ( candidateVersion != null )
299             {
300                 String originalVersion = version.getAssociations()[0].getArtifact().getVersion(); // Yekes
301                 if ( !candidateVersion.equals( originalVersion ) ) // Update needed
302                 {
303                     result.add( writeDiffMessage( property.getName(), originalVersion, candidateVersion ).toString() );
304                     if ( !reportMode
305                         && PomHelper.setPropertyVersion( pom, null, property.getName(), candidateVersion ) )
306                     {
307                         getLog().info( "Updated ${" + property.getName() + "} from " + originalVersion + " to "
308                             + candidateVersion );
309                     }
310                 }
311             }
312         }
313         return result;
314     }

316     /**
317      * Returns the candidate version to use for the specified property.
318      * <p/>
319      * The dependencies currently linked to the property must all be defined by the remote POM and they should refer to
320      * the same version. If that's the case, that same version is returned. Otherwise, <tt>null</tt> is returned
321      * indicating that there is no candidate.
322      *
323      * @param remoteDependencies the remote dependencies
324      * @param property the property to update
325      * @param propertyVersions the association
326      * @return the candidate version or <tt>null</tt> if there isn't any
327      */
328     private String computeCandidateVersion( Map<String, Dependency> remoteDependencies, Property property,
329                                             PropertyVersions propertyVersions )
330     {
331         String candidateVersion = null;
332         for ( ArtifactAssociation artifactAssociation : propertyVersions.getAssociations() )
333         {
334             String id = generateId( artifactAssociation.getArtifact() );
335             Dependency dependency = remoteDependencies.get( id );
336             if ( dependency == null )
337             {
338                 getLog().info( "Not updating ${" + property.getName() + "}: no info for " + id );
339                 return null;
340             }
341             else
342             {
343                 if ( candidateVersion == null )
344                 {
345                     candidateVersion = dependency.getVersion();
346                 }
347                 else if ( !candidateVersion.equals( dependency.getVersion() ) )
348                 {
349                     getLog().warn( "Could not update ${" + property.getName() + "}: version mismatch" );
350                     return null;
351                 }
352             }
353         }
354         return candidateVersion;
355     }

357     private void writeReportFile( List<String> dependenciesUpdate, List<String> propertiesUpdate )
358         throws MojoExecutionException
359     {
360         if ( !reportOutputFile.getParentFile().exists() )
361         {
362             reportOutputFile.getParentFile().mkdirs();
363         }
364 
365         try( FileWriter fw = new FileWriter( reportOutputFile ); //
366             PrintWriter pw = new PrintWriter( fw ) )
367         {
368             pw.println( "The following differences were found:" );
369             pw.println();
370             if ( dependenciesUpdate.size() == 0 )
371             {
372                 pw.println( "  none" );
373             }
374             else
375             {
376                 for ( String dependencyUpdate : dependenciesUpdate )
377                 {
378                     pw.println( "  " + dependencyUpdate );
379                 }
380             }
381             pw.println();
382             pw.println( "The following property differences were found:" );
383             pw.println();
384             if ( propertiesUpdate.size() == 0 )
385             {
386                 pw.println( "  none" );
387             }
388             else
389             {
390                 for ( String propertyUpdate : propertiesUpdate )
391                 {
392                     pw.println( "  " + propertyUpdate );
393                 }
394             }
395         }
396         catch ( IOException e )
397         {
398             throw new MojoExecutionException( "Unable to write report file. ", e );
399         }
400     }

402     /**
403      * Create a simple message describing the version diff
404      *
405      * @param dep
406      * @param remoteVersion
407      * @return The message
408      */
409     private StringBuilder writeDependencyDiffMessage( Dependency dep, String remoteVersion )
410     {
411         String id = dep.getGroupId() + ":" + dep.getArtifactId();
412         return writeDiffMessage( id, dep.getVersion(), remoteVersion );
413     }
414 
415     private StringBuilder writeDiffMessage( String id, String originalVersion, String targetVersion )
416     {
417         StringBuilder buf = new StringBuilder();
418         buf.append( id );
419         buf.append( ' ' );
420         int padding = INFO_PAD_SIZE - originalVersion.length() - targetVersion.length() - 4;
421         while ( buf.length() < padding )
422         {
423             buf.append( '.' );
424         }
425         buf.append( ' ' );
426         buf.append( originalVersion );
427         buf.append( " -> " );
428         buf.append( targetVersion );
429         return buf;
430     }

432     /**
433      * Add a list of dependencies to a Map for easy access
434      *
435      * @param map
436      * @param deps
437      */
438     private void mapDependencies( Map<String, Dependency> map, List<Dependency> deps )
439     {
440         if ( deps != null )
441         {
442             for ( Dependency nextDep : deps )
443             {
444                 map.put( nextDep.getManagementKey(), nextDep );
445             }
446         }
447     }

449     /**
450      * Creates a key that is similar to what {@link Dependency#getManagementKey()} generates for a dependency.
451      */
452     private static String generateId( Artifact artifact )
453     {
454         StringBuilder sb = new StringBuilder();
455         sb.append( artifact.getGroupId() ).append( ":" ).append( artifact.getArtifactId() ).append( ":" ).append( artifact.getType() );
456         if ( artifact.getClassifier() != null )
457         {
458             sb.append( ":" ).append( artifact.getClassifier() );
459         }
460         return sb.toString();
461     }
