22  import java.nio.file.Files;
23  import java.nio.file.Path;
24  import java.nio.file.Paths;
25  
26  import org.apache.maven.plugin.testing.AbstractMojoTestCase;
27  import org.apache.maven.plugin.testing.MojoRule;
28  import org.junit.Rule;
29  import org.junit.Test;
30  
31  import static org.hamcrest.MatcherAssert.assertThat;
32  import static org.hamcrest.Matchers.allOf;
33  import static org.hamcrest.Matchers.matchesPattern;
34  
35  /**
36   * Basic tests for {@linkplain SetPropertyMojoTest}.
37   *
38   * @author Andrzej Jarmoniuk
39   */
40  public class SetScmTagMojoTest extends AbstractMojoTestCase
41  {
42      @Rule
43      public MojoRule mojoRule = new MojoRule( this );
44  
45      @Test
46      public void testNewScmValues() throws Exception
47      {
48          Path pomFile = Paths.get( "target/test-classes/org/codehaus/mojo/set-scm-tag/pom.xml" );
49          mojoRule.lookupConfiguredMojo( pomFile.toFile().getParentFile(), "set-scm-tag" )
50                  .execute();
51          String output = String.join( "", Files.readAllLines( pomFile ) )
52                  .replaceAll( "\\s*", "" );
53          assertThat( output, allOf(
54                          matchesPattern( ".*<scm>.*<tag>\\s*newTag\\s*</tag>.*</scm>.*" ),
55                          matchesPattern( ".*<scm>.*<url>\\s*url\\s*</url>.*</scm>.*" ),
56                          matchesPattern( ".*<scm>.*<connection>\\s*connection\\s*</connection>.*</scm>.*" ),
57                          matchesPattern( ".*<scm>.*<developerConnection>\\s*"
58                                  + "developerConnection\\s*</developerConnection>.*</scm>.*" )
59                  )
60          );
61      }
62  }
