34  /**
35   * Builds {@link org.codehaus.mojo.versions.api.PropertyVersions} instances.
36   *
37   * @author Stephen Connolly
38   * @since 1.0-beta-1
39   */
40  class PropertyVersionsBuilder
41  {
42      private final String name;
43  
44      private final String profileId;
45  
46      private final Set<ArtifactAssociation> associations;
47  
48      private final VersionsHelper helper;
49  
50      private final Map<String, Boolean> upperBounds = new LinkedHashMap<String, Boolean>();
51  
52      private final Map<String, Boolean> lowerBounds = new LinkedHashMap<String, Boolean>();
53  
54      /**
55       * Constructs a new {@link org.codehaus.mojo.versions.api.PropertyVersions}.
56       *
57       * @param profileId The profileId.
58       * @param name The property name.
59       * @param helper The {@link org.codehaus.mojo.versions.api.DefaultVersionsHelper}.
60       */
61      public PropertyVersionsBuilder( String profileId, String name, VersionsHelper helper )
62      {
63          this.profileId = profileId;
64          this.name = name;
65          this.associations = new TreeSet<ArtifactAssociation>();
66          this.helper = helper;
67      }
68  
69      public void addAssociation( Artifact artifact, boolean usePluginRepositories )
70      {
71          associations.add( new DefaultArtifactAssociation( artifact, usePluginRepositories ) );
72      }
73  
74      public void removeAssociation( Artifact artifact, boolean usePluginRepositories )
75      {
76          associations.remove( new DefaultArtifactAssociation( artifact, usePluginRepositories ) );
77      }
78  
79      public void clearAssociations()
80      {
81          associations.clear();
82      }
83  
84      public boolean isAssociated()
85      {
86          return !associations.isEmpty();
87      }
88  
89      public ArtifactAssociation[] getAssociations()
90      {
91          return associations.toArray( new ArtifactAssociation[associations.size()] );
92      }
93  
94      public PropertyVersions newPropertyVersions()
95          throws ArtifactMetadataRetrievalException
96      {
97          return new PropertyVersions( profileId, name, helper, associations );
98      }
99  
100     public String getName()
101     {
102         return name;
103     }
104 
105     public String getVersionRange()
106     {
107         Comparator<ArtifactVersion> comparator = new PropertyVersionComparator();
108         if ( lowerBounds.isEmpty() && upperBounds.isEmpty() )
109         {
110             return null;
111         }
112         ArtifactVersion lowerBound = null;
113         boolean includeLower = true;
114         for ( Map.Entry<String, Boolean> entry : lowerBounds.entrySet() )
115         {
116             ArtifactVersion candidate = helper.createArtifactVersion( entry.getKey() );
117             if ( lowerBound == null )
118             {
119                 lowerBound = candidate;
120                 includeLower = entry.getValue();
121             }
122             else
123             {
124                 final int result = comparator.compare( lowerBound, candidate );
125                 if ( result > 0 )
126                 {
127                     lowerBound = candidate;
128                     includeLower = entry.getValue();
129                 }
130                 else if ( result == 0 )
131                 {
132                     includeLower = includeLower && entry.getValue();
133                 }
134             }
135         }
136         ArtifactVersion upperBound = null;
137         boolean includeUpper = true;
138         for ( Map.Entry<String, Boolean> entry : upperBounds.entrySet() )
139         {
140             ArtifactVersion candidate = helper.createArtifactVersion( entry.getKey() );
141             if ( upperBound == null )
142             {
143                 upperBound = candidate;
144                 includeUpper = entry.getValue();
145             }
146             else
147             {
148                 final int result = comparator.compare( upperBound, candidate );
149                 if ( result < 0 )
150                 {
151                     upperBound = candidate;
152                     includeUpper = entry.getValue();
153                 }
154                 else if ( result == 0 )
155                 {
156                     includeUpper = includeUpper && entry.getValue();
157                 }
158             }
159         }
160         StringBuilder buf = new StringBuilder();
161         if ( includeLower )
162         {
163             buf.append( '[' );
164         }
165         else
166         {
167             buf.append( '(' );
168         }
169         if ( lowerBound != null )
170         {
171             buf.append( lowerBound );
172         }
173         buf.append( ',' );
174         if ( upperBound != null )
175         {
176             buf.append( upperBound );
177         }
178         if ( includeUpper )
179         {
180             buf.append( ']' );
181         }
182         else
183         {
184             buf.append( ')' );
185         }
186         return buf.toString();
187     }
188 
189     public void addLowerBound( String lowerBound, boolean includeLower )
190     {
191         Boolean value = lowerBounds.get( lowerBound );
192         if ( value == null )
193         {
194             value = includeLower;
195         }
196         else
197         {
198             value = includeLower && value;
199         }
200         lowerBounds.put( lowerBound, value );
201     }
202 
203     public void addUpperBound( String upperBound, boolean includeUpper )
204     {
205         Boolean value = upperBounds.get( upperBound );
206         if ( value == null )
207         {
208             value = includeUpper;
209         }
210         else
211         {
212             value = includeUpper && value;
213         }
214         upperBounds.put( upperBound, value );
215     }

217     private VersionComparator[] lookupComparators()
218     {
219         Set<VersionComparator> result = new HashSet<VersionComparator>();
220         for ( ArtifactAssociation association : associations )
221         {
222             result.add( helper.getVersionComparator( association.getArtifact() ) );
223         }
224         return result.toArray( new VersionComparator[result.size()] );
225     }
226 
227     private final class PropertyVersionComparator
228         implements Comparator<ArtifactVersion>
229     {
230         public int compare( ArtifactVersion v1, ArtifactVersion v2 )
231         {
232             return innerCompare( v1, v2 );
233         }
234 
235         private int innerCompare( ArtifactVersion v1, ArtifactVersion v2 )
236         {
237             if ( !isAssociated() )
238             {
239                 throw new IllegalStateException( "Cannot compare versions for a property with no associations" );
240             }
241             VersionComparator[] comparators = lookupComparators();
242             assert comparators.length >= 1 : "we have at least one association => at least one comparator";
243             int result = comparators[0].compare( v1, v2 );
244             for ( int i = 1; i < comparators.length; i++ )
245             {
246                 int alt = comparators[i].compare( v1, v2 );
247                 if ( result != alt && ( result >= 0 && alt < 0 ) || ( result <= 0 && alt > 0 ) )
248                 {
249                     throw new IllegalStateException( "Property " + name + " is associated with multiple artifacts"
250                         + " and these artifacts use different version sorting rules and these rules are effectively"
251                         + " incompatible for the two of versions being compared.\nFirst rule says compare(\"" + v1
252                         + "\", \"" + v2 + "\") = " + result + "\nSecond rule says compare(\"" + v1 + "\", \"" + v2
253                         + "\") = " + alt );
254                 }
255             }
256             return result;
257         }
258 
259     }
