252     private static RuleSet readRulesFromStream(InputStream stream)
253         throws IOException {
254         RuleXpp3Reader reader = new RuleXpp3Reader();
255         BufferedInputStream bis = new BufferedInputStream( stream );
256 
257         try
258         {
259             return reader.read( bis );
260         }
261         catch ( XmlPullParserException e )
262         {
263             final IOException ioe = new IOException();
264             ioe.initCause( e );
265             throw ioe;
266         }
267         finally
268         {
269             try
270             {
271                 bis.close();
272             }
273             catch ( IOException e )
274             {
275                 // ignore
276             }
277         }
278     }

292     private static RuleSet loadRuleSet( String serverId, Settings settings, WagonManager wagonManager, String rulesUri,
293                                         Log logger )
294         throws MojoExecutionException {
295         RuleSet ruleSet = new RuleSet();
296         boolean rulesUriGiven = isRulesUriNotBlank(rulesUri);
297 
298         if (rulesUriGiven) {
299             RuleSet loadedRules;
300 
301             if (isClasspathUri(rulesUri)) {
302                 loadedRules = getRulesFromClasspath(rulesUri, logger);
303             } else {
304                 loadedRules = getRulesViaWagon(rulesUri, logger, serverId, serverId, wagonManager,
305                                                        settings);
306             }
307 
308             ruleSet.setIgnoreVersions(loadedRules.getIgnoreVersions());
309             ruleSet.setRules(loadedRules.getRules());
310         }
311 
312         return ruleSet;
313     }

315     private static RuleSet getRulesFromClasspath(String uri, Log logger)
316         throws MojoExecutionException {
317         logger.debug("Going to load rules from \"" + uri + "\"");
318 
319         String choppedUrl = uri.substring(CLASSPATH_PROTOCOL.length() + 3);
320 
321         URL url = DefaultVersionsHelper.class.getResource(choppedUrl);
322 
323         if (null == url) {
324             String message = "Resource \"" + uri + "\" not found in classpath.";
325 
326             throw new MojoExecutionException(message);
327         }
328 
329         try {
330             RuleSet rules = readRulesFromStream(url.openStream());
331             logger.debug("Loaded rules from \"" + uri + "\" successfully");
332             return rules;
333         }
334         catch (IOException e) {
335             throw new MojoExecutionException("Could not load specified rules from " + uri, e);
336         }
337     }

339     private static boolean isRulesUriNotBlank(String rulesUri) {
340         return rulesUri != null && rulesUri.trim().length() != 0;
341     }

343     private static RuleSet getRulesViaWagon(String rulesUri, Log logger, String serverId, String id,
344                                             WagonManager wagonManager, Settings settings)
345         throws MojoExecutionException {
346         RuleSet loadedRules = new RuleSet();
347 
348         int split = rulesUri.lastIndexOf('/');
349         String baseUri = rulesUri;
350         String fileUri = "";
351 
352         if (split != -1) {
353             baseUri = rulesUri.substring(0, split) + '/';
354             fileUri = split + 1 < rulesUri.length() ? rulesUri.substring(split + 1) : "";
355         }
356 
357         try {
358             Wagon wagon = WagonUtils.createWagon(serverId, baseUri, wagonManager, settings, logger);
359             try {
360                 logger.debug("Trying to load ruleset from file \"" + fileUri + "\" in " + baseUri);
361                 loadedRules = getRuleSet(wagon, fileUri);
362             }
363             finally {
364                 logger.debug("Rule set loaded");
365 
366                 if (wagon != null) {
367                     try {
368                         wagon.disconnect();
369                     }
370                     catch (ConnectionException e) {
371                         logger.warn("Could not disconnect wagon!", e);
372                     }
373                 }
374             }
375         }
376         catch (TransferFailedException e) {
377             throw new MojoExecutionException("Could not transfer rules from " + rulesUri, e);
378         }
379         catch (AuthorizationException e) {
380             throw new MojoExecutionException("Authorization failure trying to load rules from " + rulesUri, e);
381         }
382         catch (ResourceDoesNotExistException e) {
383             throw new MojoExecutionException("Could not load specified rules from " + rulesUri, e);
384         }
385         catch (AuthenticationException e) {
386             throw new MojoExecutionException("Authentication failure trying to load rules from " + rulesUri, e);
387         }
388         catch (UnsupportedProtocolException e) {
389             throw new MojoExecutionException("Unsupported protocol for " + rulesUri, e);
390         }
391         catch (ConnectionException e) {
392             throw new MojoExecutionException("Could not establish connection to " + rulesUri, e);
393         }
394         catch (IOException e) {
395             throw new MojoExecutionException("Could not load specified rules from " + rulesUri, e);
396         }
397 
398         return loadedRules;
399     }

401     static boolean isClasspathUri(String uri) {
402         boolean startsWithProtocol = null != uri && uri.startsWith(CLASSPATH_PROTOCOL);
403         boolean hasColonNext = null != uri && uri.charAt(CLASSPATH_PROTOCOL.length()) == ':';
404 
405         return startsWithProtocol && hasColonNext;
406     }

424     /**
425      * {@inheritDoc}
426      */
427     public ArtifactVersions lookupArtifactVersions( Artifact artifact, boolean usePluginRepositories )
428         throws ArtifactMetadataRetrievalException
429     {
430         List remoteRepositories = usePluginRepositories ? remotePluginRepositories : remoteArtifactRepositories;
431         final List<ArtifactVersion> versions =
432             artifactMetadataSource.retrieveAvailableVersions( artifact, localRepository, remoteRepositories );
433         final List<IgnoreVersion> ignoredVersions = getIgnoredVersions( artifact );
434         if ( !ignoredVersions.isEmpty() )
435         {
436             if ( getLog().isDebugEnabled() )
437             {
438                 getLog().debug( "Found ignored versions: " + showIgnoredVersions( ignoredVersions ) );
439             }
440 
441             final Iterator<ArtifactVersion> i = versions.iterator();
442             while ( i.hasNext() )
443             {
444                 final String version = i.next().toString();
445                 for ( final IgnoreVersion ignoreVersion : ignoredVersions )
446                 {
447                     if ( TYPE_REGEX.equals( ignoreVersion.getType() ) )
448                     {
449                         Pattern p = Pattern.compile( ignoreVersion.getVersion() );
450                         if ( p.matcher( version ).matches() )
451                         {
452                             if ( getLog().isDebugEnabled() )
453                             {
454                                 getLog().debug( "Version " + version + " for artifact "
455                                     + ArtifactUtils.versionlessKey( artifact ) + " found on ignore list: "
456                                     + ignoreVersion );
457                             }
458                             i.remove();
459                             break;
460                         }
461                     }
462                     else if ( TYPE_EXACT.equals( ignoreVersion.getType() ) )
463                     {
464                         if ( version.equals( ignoreVersion.getVersion() ) )
465                         {
466                             if ( getLog().isDebugEnabled() )
467                             {
468                                 getLog().debug( "Version " + version + " for artifact "
469                                     + ArtifactUtils.versionlessKey( artifact ) + " found on ignore list: "
470                                     + ignoreVersion );
471                             }
472                             i.remove();
473                             break;
474                         }
475                     }
476                 }
477             }
478         }
479         return new ArtifactVersions( artifact, versions, getVersionComparator( artifact ) );
480     }

482     /**
483      * Returns a list of versions which should not be considered when looking for updates.
484      *
485      * @param artifact The artifact
486      * @return List of ignored version
487      */
488     private List<IgnoreVersion> getIgnoredVersions( Artifact artifact )
489     {
490         final List<IgnoreVersion> ret = new ArrayList<IgnoreVersion>();
491 
492         for ( final IgnoreVersion ignoreVersion : ruleSet.getIgnoreVersions() )
493         {
494             if ( !TYPE_EXACT.equals( ignoreVersion.getType() ) && !TYPE_REGEX.equals( ignoreVersion.getType() ) )
495             {
496                 getLog().warn( "The type attribute '" + ignoreVersion.getType() + "' for global ignoreVersion["
497                     + ignoreVersion + "] is not valid." + " Please use either '" + TYPE_EXACT + "' or '" + TYPE_REGEX
498                     + "'." );
499             }
500             else
501             {
502                 ret.add( ignoreVersion );
503             }
504         }
505 
506         final Rule rule = getBestFitRule( artifact.getGroupId(), artifact.getArtifactId() );
507 
508         if ( rule != null )
509         {
510             for ( IgnoreVersion ignoreVersion : rule.getIgnoreVersions() )
511             {
512                 if ( !TYPE_EXACT.equals( ignoreVersion.getType() ) && !TYPE_REGEX.equals( ignoreVersion.getType() ) )
513                 {
514                     getLog().warn( "The type attribute '" + ignoreVersion.getType() + "' for " + rule + " is not valid."
515                         + " Please use either '" + TYPE_EXACT + "' or '" + TYPE_REGEX + "'." );
516                 }
517                 else
518                 {
519                     ret.add( ignoreVersion );
520                 }
521             }
522         }
523 
524         return ret;
525     }

528      * Pretty print a list of ignored versions.
529      *
530      * @param ignoredVersions A list of ignored versions
531      * @return A String representation of the list

533     private String showIgnoredVersions( List<IgnoreVersion> ignoredVersions )
534     {
535         StringBuilder buf = new StringBuilder();
536         Iterator<IgnoreVersion> iterator = ignoredVersions.iterator();
537         while ( iterator.hasNext() )
538         {
539             IgnoreVersion ignoreVersion = iterator.next();
540             buf.append( ignoreVersion );
541             if ( iterator.hasNext() )
542             {
543                 buf.append( ", " );
544             }
545         }
546         return buf.toString();
547     }

574     /**
575      * Find the rule, if any, which best fits the artifact details given.
576      *
577      * @param groupId Group id of the artifact
578      * @param artifactId Artifact id of the artifact
579      * @return Rule which best describes the given artifact
580      */
581     protected Rule getBestFitRule( String groupId, String artifactId )
582     {
583         Rule bestFit = null;
584         final List<Rule> rules = ruleSet.getRules();
585         int bestGroupIdScore = Integer.MAX_VALUE;
586         int bestArtifactIdScore = Integer.MAX_VALUE;
587         boolean exactGroupId = false;
588         boolean exactArtifactId = false;
589         for ( Rule rule : rules )
590         {
591             int groupIdScore = RegexUtils.getWildcardScore( rule.getGroupId() );
592             if ( groupIdScore > bestGroupIdScore )
593             {
594                 continue;
595             }
596             boolean exactMatch = exactMatch( rule.getGroupId(), groupId );
597             boolean match = exactMatch || match( rule.getGroupId(), groupId );
598             if ( !match || ( exactGroupId && !exactMatch ) )
599             {
600                 continue;
601             }
602             if ( bestGroupIdScore > groupIdScore )
603             {
604                 bestArtifactIdScore = Integer.MAX_VALUE;
605                 exactArtifactId = false;
606             }
607             bestGroupIdScore = groupIdScore;
608             if ( exactMatch && !exactGroupId )
609             {
610                 exactGroupId = true;
611                 bestArtifactIdScore = Integer.MAX_VALUE;
612                 exactArtifactId = false;
613             }
614             int artifactIdScore = RegexUtils.getWildcardScore( rule.getArtifactId() );
615             if ( artifactIdScore > bestArtifactIdScore )
616             {
617                 continue;
618             }
619             exactMatch = exactMatch( rule.getArtifactId(), artifactId );
620             match = exactMatch || match( rule.getArtifactId(), artifactId );
621             if ( !match || ( exactArtifactId && !exactMatch ) )
622             {
623                 continue;
624             }
625             bestArtifactIdScore = artifactIdScore;
626             if ( exactMatch && !exactArtifactId )
627             {
628                 exactArtifactId = true;
629             }
630             bestFit = rule;
631         }
632         return bestFit;
633     }

711     /**
712      * {@inheritDoc}
713      */
714     public Map<Dependency, ArtifactVersions> lookupDependenciesUpdates( Set<Dependency> dependencies,
715                                                                         boolean usePluginRepositories )
716         throws ArtifactMetadataRetrievalException, InvalidVersionSpecificationException
717     {
718         // Create the request for details collection for parallel lookup...
719         final List<Callable<DependencyArtifactVersions>> requestsForDetails =
720             new ArrayList<Callable<DependencyArtifactVersions>>( dependencies.size() );
721         for ( final Dependency dependency : dependencies )
722         {
723             requestsForDetails.add( new DependencyLookup( dependency, usePluginRepositories ) );
724         }
725 
726         final Map<Dependency, ArtifactVersions> dependencyUpdates =
727             new TreeMap<Dependency, ArtifactVersions>( new DependencyComparator() );
728 
729         // Lookup details in parallel...
730         final ExecutorService executor = Executors.newFixedThreadPool( LOOKUP_PARALLEL_THREADS );
731         try
732         {
733             final List<Future<DependencyArtifactVersions>> responseForDetails =
734                 executor.invokeAll( requestsForDetails );
735 
736             // Construct the final results...
737             for ( final Future<DependencyArtifactVersions> details : responseForDetails )
738             {
739                 final DependencyArtifactVersions dav = details.get();
740                 dependencyUpdates.put( dav.getDependency(), dav.getArtifactVersions() );
741             }
742         }
743         catch ( final ExecutionException ee )
744         {
745             throw new ArtifactMetadataRetrievalException( "Unable to acquire metadata for dependencies " + dependencies
746                 + ": " + ee.getMessage(), ee );
747         }
748         catch ( final InterruptedException ie )
749         {
750             throw new ArtifactMetadataRetrievalException( "Unable to acquire metadata for dependencies " + dependencies
751                 + ": " + ie.getMessage(), ie );
752         }
753         finally
754         {
755             executor.shutdownNow();
756         }
757         return dependencyUpdates;
758     }

826      * {@inheritDoc}
827      */
828     public PluginUpdatesDetails lookupPluginUpdates( Plugin plugin, boolean allowSnapshots )
829         throws ArtifactMetadataRetrievalException, InvalidVersionSpecificationException
830     {
831         String version = plugin.getVersion();
832         version = version == null ? "LATEST" : version;
833         getLog().debug( "Checking " + ArtifactUtils.versionlessKey( plugin.getGroupId(), plugin.getArtifactId() )
834             + " for updates newer than " + version );
835 
836         VersionRange versionRange = VersionRange.createFromVersion( version );
837 
838         final boolean includeSnapshots = allowSnapshots;
839 
840         final ArtifactVersions pluginArtifactVersions =
841             lookupArtifactVersions( createPluginArtifact( plugin.getGroupId(), plugin.getArtifactId(), versionRange ),
842                                     true );
843 
844         Set<Dependency> pluginDependencies = new TreeSet<Dependency>( new DependencyComparator() );
845         if ( plugin.getDependencies() != null )
846         {
847             pluginDependencies.addAll( plugin.getDependencies() );
848         }
849         Map<Dependency, ArtifactVersions> pluginDependencyDetails =
850             lookupDependenciesUpdates( pluginDependencies, false );
851 
852         return new PluginUpdatesDetails( pluginArtifactVersions, pluginDependencyDetails, includeSnapshots );
853     }

863     /**
864      * {@inheritDoc}
865      */
866     public Map<Property, PropertyVersions> getVersionPropertiesMap( MavenProject project,
867                                                                     Property[] propertyDefinitions,
868                                                                     String includeProperties, String excludeProperties,
869                                                                     boolean autoLinkItems )
870         throws MojoExecutionException
871     {
872         Map<String, Property> properties = new HashMap<String, Property>();
873         if ( propertyDefinitions != null )
874         {
875             for ( Property propertyDefinition : propertyDefinitions )
876             {
877                 properties.put( propertyDefinition.getName(), propertyDefinition );
878             }
879         }
880         Map<String, PropertyVersionsBuilder> builders = new HashMap<String, PropertyVersionsBuilder>();
881         if ( autoLinkItems )
882         {
883             final PropertyVersionsBuilder[] propertyVersionsBuilders;
884             try
885             {
886                 propertyVersionsBuilders = PomHelper.getPropertyVersionsBuilders( this, project );
887             }
888             catch ( ExpressionEvaluationException e )
889             {
890                 throw new MojoExecutionException( e.getMessage(), e );
891             }
892             catch ( IOException e )
893             {
894                 throw new MojoExecutionException( e.getMessage(), e );
895             }
896 
897             for ( PropertyVersionsBuilder propertyVersionsBuilder : propertyVersionsBuilders )
898             {
899                 final String name = propertyVersionsBuilder.getName();
900                 builders.put( name, propertyVersionsBuilder );
901                 if ( !properties.containsKey( name ) )
902                 {
903                     final Property value = new Property( name );
904                     getLog().debug( "Property ${" + name + "}: Adding inferred version range of "
905                         + propertyVersionsBuilder.getVersionRange() );
906                     value.setVersion( propertyVersionsBuilder.getVersionRange() );
907                     properties.put( name, value );
908                 }
909             }
910         }
911 
912         List<String> includePropertiesList = getSplittedProperties( includeProperties );
913         List<String> excludePropertiesList = getSplittedProperties( excludeProperties );
914 
915         getLog().debug( "Searching for properties associated with builders" );
916         Iterator<Property> i = properties.values().iterator();
917         while ( i.hasNext() )
918         {
919             Property property = i.next();
920 
921             getLog().debug( "includePropertiesList:" + includePropertiesList + " property: " + property.getName() );
922             getLog().debug( "excludePropertiesList:" + excludePropertiesList + " property: " + property.getName() );
923             if ( !includePropertiesList.isEmpty() && !includePropertiesList.contains( property.getName() ) )
924             {
925                 getLog().debug( "Skipping property ${" + property.getName() + "}" );
926                 i.remove();
927             }
928             else if ( !excludePropertiesList.isEmpty() && excludePropertiesList.contains( property.getName() ) )
929             {
930                 getLog().debug( "Ignoring property ${" + property.getName() + "}" );
931                 i.remove();
932             }
933         }
934         i = properties.values().iterator();
935         Map<Property, PropertyVersions> propertyVersions =
936             new LinkedHashMap<Property, PropertyVersions>( properties.size() );
937         while ( i.hasNext() )
938         {
939             Property property = i.next();
940             getLog().debug( "Property ${" + property.getName() + "}" );
941             PropertyVersionsBuilder builder = builders.get( property.getName() );
942             if ( builder == null || !builder.isAssociated() )
943             {
944                 getLog().debug( "Property ${" + property.getName() + "}: Looks like this property is not "
945                     + "associated with any dependency..." );
946                 builder = new PropertyVersionsBuilder( null, property.getName(), this );
947             }
948             if ( !property.isAutoLinkDependencies() )
949             {
950                 getLog().debug( "Property ${" + property.getName() + "}: Removing any autoLinkDependencies" );
951                 builder.clearAssociations();
952             }
953             Dependency[] dependencies = property.getDependencies();
954             if ( dependencies != null )
955             {
956                 for ( Dependency dependency : dependencies )
957                 {
958                     try
959                     {
960                         getLog().debug( "Property ${" + property.getName() + "}: Adding association to " + dependency );
961                         builder.addAssociation( this.createDependencyArtifact( dependency ), false );
962                     }
963                     catch ( InvalidVersionSpecificationException e )
964                     {
965                         throw new MojoExecutionException( e.getMessage(), e );
966                     }
967                 }
968             }
969             try
970             {
971                 final PropertyVersions versions = builder.newPropertyVersions();
972                 if ( property.isAutoLinkDependencies() && StringUtils.isEmpty( property.getVersion() )
973                     && !StringUtils.isEmpty( builder.getVersionRange() ) )
974                 {
975                     getLog().debug( "Property ${" + property.getName() + "}: Adding inferred version range of "
976                         + builder.getVersionRange() );
977                     property.setVersion( builder.getVersionRange() );
978                 }
979                 versions.setCurrentVersion( project.getProperties().getProperty( property.getName() ) );
980                 propertyVersions.put( property, versions );
981             }
982             catch ( ArtifactMetadataRetrievalException e )
983             {
984                 throw new MojoExecutionException( e.getMessage(), e );
985             }
986         }
987         return propertyVersions;
988     }

990     private List<String> getSplittedProperties( String commaSeparatedProperties )
991     {
992         List<String> propertiesList = Collections.emptyList();
993         if ( StringUtils.isNotEmpty( commaSeparatedProperties ) )
994         {
995             String[] splittedProps = StringUtils.split( commaSeparatedProperties, "," );
996             propertiesList = Arrays.asList( StringUtils.stripAll( splittedProps ) );
997         }
998         return propertiesList;
999     }
