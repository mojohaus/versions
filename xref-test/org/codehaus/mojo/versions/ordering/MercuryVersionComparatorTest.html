22  import org.apache.maven.artifact.versioning.DefaultArtifactVersion;
23  import org.junit.Test;
24  
25  import static org.junit.Assert.assertEquals;
26  
27  public class MercuryVersionComparatorTest extends VersionComparatorTestBase
28  {
29      public MercuryVersionComparatorTest()
30      {
31          super( new MercuryVersionComparator() );
32      }
33  
34      @Test
35      public void testSegmentCounting()
36      {
37          assertEquals( 1, instance.getSegmentCount( new DefaultArtifactVersion( "5" ) ) );
38          assertEquals( 2, instance.getSegmentCount( new DefaultArtifactVersion( "5.0" ) ) );
39          assertEquals( 2, instance.getSegmentCount( new DefaultArtifactVersion( "5-0" ) ) );
40          assertEquals( 3, instance.getSegmentCount( new DefaultArtifactVersion( "5.3.a" ) ) );
41          assertEquals( 6, instance.getSegmentCount( new DefaultArtifactVersion( "5.0.a.1.4.5" ) ) );
42          assertEquals( 0, instance.getSegmentCount( new DefaultArtifactVersion( "" ) ) );
43      }
