1   package org.codehaus.mojo.versions.ordering;
2   
3   import org.apache.maven.artifact.versioning.ArtifactVersion;
4   
5   /**
6    * Base class for version comparators.
7    *
8    * @since 1.0-beta-1
9    */
10  public abstract class AbstractVersionComparator
11      implements VersionComparator
12  {
13      /**
14       * {@inheritDoc}
15       */
16      public abstract int compare( ArtifactVersion o1, ArtifactVersion o2 );
17  
18      /**
19       * {@inheritDoc}
20       */
21      public final int getSegmentCount( ArtifactVersion v )
22      {
23          if ( v == null )
24          {
25              return 0;
26          }
27          if ( VersionComparators.isSnapshot( v ) )
28          {
29              return innerGetSegmentCount( VersionComparators.stripSnapshot( v ) );
30          }
31          return innerGetSegmentCount( v );
32  
33      }
34  
35      protected abstract int innerGetSegmentCount( ArtifactVersion v );
36  
37      /**
38       * {@inheritDoc}
39       */
40      public final ArtifactVersion incrementSegment( ArtifactVersion v, int segment )
41      {
42          if ( VersionComparators.isSnapshot( v ) )
43          {
44              return VersionComparators.copySnapshot( v, innerIncrementSegment( VersionComparators.stripSnapshot( v ),
45                                                                                segment ) );
46          }
47          return innerIncrementSegment( v, segment );
48      }
49  
50      protected abstract ArtifactVersion innerIncrementSegment( ArtifactVersion v, int segment );
51  
52      /**
53       * Returns a hash code value for the comparator class.
54       *
55       * @return the hash code.
56       */
57      public int hashCode()
58      {
59          return getClass().hashCode();
60      }
61  
62      /**
63       * Returns true if this object is the same type of comparator as the parameter.
64       *
65       * @param obj the reference object with which to compare.
66       * @return <code>true</code> if this object is the same as the obj
67       *         argument; <code>false</code> otherwise.
68       * @see #hashCode()
69       * @see java.util.Hashtable
70       */
71      public boolean equals( Object obj )
72      {
73          return obj == this || ( obj != null && getClass().equals( obj.getClass() ) );
74      }
75  
76  }
