23  import java.util.Optional;
24  
25  import org.apache.maven.plugin.logging.Log;
26  import org.codehaus.mojo.versions.api.Segment;
27  
28  import static java.util.Optional.empty;
29  import static java.util.Optional.of;
30  import static org.codehaus.mojo.versions.api.Segment.INCREMENTAL;
31  import static org.codehaus.mojo.versions.api.Segment.MAJOR;
32  import static org.codehaus.mojo.versions.api.Segment.MINOR;
33  
34  /**
35   * Utility class for manipulating with {@link Segment} objects
36   */
37  public class SegmentUtils
38  {
39      /**
40       * <p>Based on the passed flags, determines which segment which is not to be changed.</p>
41       *
42       * <p>Also, logs the enriched values of the {@code allowMajorUpdates}, {@code allowMinorUpdates},
43       * and {@code allowIncrementalUpdates} options so that {@code allowMinorUpdates} equal to {@code false}
44       * implies that {@code allowMajorUpdates} is also {@code false}.</p>
45       * <p>Also, {@code allowIncrementalUpdates} equal to {@code false}
46       * implies that both {@code allowMajorUpdates} and {@code allowMinorUpdates} are also {@code false}.</p>
47       *
48       * <table>
49       * <caption>Effective values for update options</caption>
50       * <thead>
51       * <tr><th>allowMajorUpdates</th><th>allowMinorUpdates</th><th>allowIncrementalUpdates</th></tr>
52       * </thead>
53       * <tbody>
54       * <tr><td>true</td><td>true</td><td>true</td></tr>
55       * <tr><td></td><td>true</td><td>true</td></tr>
56       * <tr><td></td><td></td><td>true</td></tr>
57       * <tr><td></td><td></td><td></td></tr>
58       * </tbody>
59       * </table>
60       *
61       * @param allowMajorUpdates whether all updates should be allowed
62       * @param allowMinorUpdates if major updates are disallowed, minor, incremental updates should be allowed
63       * @param allowIncrementalUpdates if major and minor updates are disallowed, incremental updates are allowed
64       * @param log                     If not null, the {@linkplain Log} object to log the selected scope
65       * @return Returns the segment (0-based) that is unchangeable. If any segment can change, returns -1.
66       */
67      public static Optional<Segment> determineUnchangedSegment( boolean allowMajorUpdates, boolean allowMinorUpdates,
68                                                                 boolean allowIncrementalUpdates, Log log )
69      {
70          if ( log != null && !allowIncrementalUpdates )
71          {
72              log.info( "Assuming allowMinorUpdates false because allowIncrementalUpdates is false." );
73          }
74  
75          if ( log != null && !allowMinorUpdates )
76          {
77              log.info( "Assuming allowMajorUpdates false because allowMinorUpdates is false." );
78          }
79  
80          Optional<Segment> unchangedSegment = allowMajorUpdates && allowMinorUpdates && allowIncrementalUpdates
81                  ? empty()
82                  : allowMinorUpdates && allowIncrementalUpdates
83                      ? of( MAJOR )
84                      : allowIncrementalUpdates
85                          ? of( MINOR )
86                          : of( INCREMENTAL );
87          if ( log != null && log.isDebugEnabled() )
88          {
89              log.debug(
90                      unchangedSegment.map( s -> Segment.of( s.value() + 1 ).toString() )
91                              .orElse( "ALL" ) + " version changes allowed" );
92          }
93          return unchangedSegment;
94      }
95  }
