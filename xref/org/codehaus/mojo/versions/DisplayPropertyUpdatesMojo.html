38  /**
39   * Displays properties that are linked to artifact versions and have updates available.
40   *
41   * @author Stephen Connolly
42   * @since 1.0-beta-1
43   */
44  @Mojo( name = "display-property-updates", requiresProject = true, requiresDirectInvocation = true, threadSafe = true )
45  public class DisplayPropertyUpdatesMojo
46      extends AbstractVersionsDisplayMojo
47  {
48  
49      /**
50       * The width to pad info messages.
51       *
52       * @since 1.0-alpha-1
53       */
54      private static final int INFO_PAD_SIZE = 68;
55  
56      // ------------------------------ FIELDS ------------------------------
57  
58      /**
59       * Any restrictions that apply to specific properties.
60       *
61       * @since 1.0-alpha-3
62       */
63      @Parameter
64      private Property[] properties;
65  
66      /**
67       * A comma separated list of properties to update.
68       *
69       * @since 1.0-alpha-1
70       */
71      @Parameter( property = "includeProperties" )
72      private String includeProperties = null;
73  
74      /**
75       * A comma separated list of properties to not update.
76       *
77       * @parameter property="excludeProperties"
78       * @since 1.0-alpha-1
79       */
80      @Parameter( property = "excludeProperties" )
81      private String excludeProperties = null;
82  
83      /**
84       * Whether properties linking versions should be auto-detected or not.
85       *
86       * @since 1.0-alpha-2
87       */
88      @Parameter( property = "autoLinkItems", defaultValue = "true" )
89      private boolean autoLinkItems;
90  
91      /**
92       * Whether to allow the major version number to be changed.
93       *
94       * @since 2.5
95       */
96      @Parameter( property = "allowMajorUpdates", defaultValue = "true" )
97      private boolean allowMajorUpdates;
98  
99      /**
100      * Whether to allow the minor version number to be changed.
101      *
102      * @since 2.5
103      */
104     @Parameter( property = "allowMinorUpdates", defaultValue = "true" )
105     private boolean allowMinorUpdates;
106 
107     /**
108      * Whether to allow the incremental version number to be changed.
109      *
110      * @since 2.5
111      */
112     @Parameter( property = "allowIncrementalUpdates", defaultValue = "true" )
113     private boolean allowIncrementalUpdates;
114 
115     // -------------------------- STATIC METHODS --------------------------
116 
117     // -------------------------- OTHER METHODS --------------------------
118 
119     public void execute()
120         throws MojoExecutionException, MojoFailureException
121     {
122         logInit();
123         List<String> current = new ArrayList<>();
124         List<String> updates = new ArrayList<>();
125 
126         Map<Property, PropertyVersions> propertyVersions =
127             this.getHelper().getVersionPropertiesMap( getProject(), properties, includeProperties, excludeProperties,
128                                                       autoLinkItems );
129         for ( Map.Entry<Property, PropertyVersions> entry : propertyVersions.entrySet() )
130         {
131             Property property = entry.getKey();
132             PropertyVersions version = entry.getValue();
133 
134             final String currentVersion = getProject().getProperties().getProperty( property.getName() );
135             if ( currentVersion == null )
136             {
137                 continue;
138             }
139 
140             int segment = determineUnchangedSegment( allowMajorUpdates, allowMinorUpdates, allowIncrementalUpdates );
141             ArtifactVersion winner = version.getNewestVersion( currentVersion, property, this.allowSnapshots,
142                                                                this.reactorProjects, this.getHelper(), false, segment );
143 
144             if ( winner != null && !currentVersion.equals( winner.toString() ) )
145             {
146                 StringBuilder buf = new StringBuilder();
147                 buf.append( "${" );
148                 buf.append( property.getName() );
149                 buf.append( "} " );
150                 final String newVersion = winner.toString();
151                 int padding = INFO_PAD_SIZE - currentVersion.length() - newVersion.length() - 4;
152                 while ( buf.length() < padding )
153                 {
154                     buf.append( '.' );
155                 }
156                 buf.append( ' ' );
157                 buf.append( currentVersion );
158                 buf.append( " -> " );
159                 buf.append( newVersion );
160                 updates.add( buf.toString() );
161             }
162             else
163             {
164                 StringBuilder buf = new StringBuilder();
165                 buf.append( "${" );
166                 buf.append( property.getName() );
167                 buf.append( "} " );
168                 int padding = INFO_PAD_SIZE - currentVersion.length();
169                 while ( buf.length() < padding )
170                 {
171                     buf.append( '.' );
172                 }
173                 buf.append( ' ' );
174                 buf.append( currentVersion );
175                 current.add( buf.toString() );
176             }
177 
178         }
179 
180         logLine( false, "" );
181         if ( !current.isEmpty() )
182         {
183             logLine( false, "The following version properties are referencing the newest available version:" );
184             for ( String s : new TreeSet<>(current) )
185             {
186                 logLine( false, "  " + s );
187             }
188         }
189         if ( updates.isEmpty() && current.isEmpty() )
190         {
191             logLine( false, "This project does not have any properties associated with versions." );
192         }
193         else if ( updates.isEmpty() )
194         {
195             logLine( false, "All version properties are referencing the newest version available." );
196         }
197 
198         if ( !updates.isEmpty() )
199         {
200             logLine( false, "The following version property updates are available:" );
201             for ( String update : new TreeSet<>(updates) )
202             {
203                 logLine( false, "  " + update );
204             }
205         }
206         logLine( false, "" );
207     }

209     protected void update( ModifiedPomXMLEventReader pom )
210         throws MojoExecutionException, MojoFailureException, XMLStreamException, ArtifactMetadataRetrievalException
211     {
212     }
