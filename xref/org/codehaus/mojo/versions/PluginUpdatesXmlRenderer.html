38  /**
39   * XML renderer for PluginUpdatesReport creates an xml file in target directory and writes report about available
40   * plugin/plugin management updates.
41   * 
42   * @author Illia Dubinin
43   * @since 2.4
44   */
45  public class PluginUpdatesXmlRenderer
46  {
47  
48      private static final String GROUP_ID = "groupId";
49  
50      private static final String ARTIFACT_ID = "artifactId";
51  
52      private static final String OPEN_TAG = "<";
53  
54      private static final String CLOSE_TAG = ">";
55  
56      private static final String OPEN_CLOSING_TAG = "</";
57  
58      private static final String NL = "\n";
59  
60      private static final String TAB = "\t";
61  
62      private Map<Plugin, PluginUpdatesDetails> pluginUpdates;
63  
64      private Map<Plugin, PluginUpdatesDetails> pluginManagementUpdates;
65  
66      private String outputFileName;
67  
68      public PluginUpdatesXmlRenderer( Map<Plugin, PluginUpdatesDetails> pluginUpdates,
69                                       Map<Plugin, PluginUpdatesDetails> pluginManagementUpdates, String outputFileName )
70      {
71          this.pluginUpdates = pluginUpdates;
72          this.pluginManagementUpdates = pluginManagementUpdates;
73          this.outputFileName = outputFileName;
74      }
75  
76      /**
77       * Makes report file with given name in target directory.
78       * 
79       * @throws MavenReportException if something went wrong
80       */
81      public void render()
82          throws MavenReportException
83      {
84          StringBuilder sb = new StringBuilder();
85          sb.append( "<PluginUpdatesReport>" ).append( NL );
86          Map<Plugin, PluginUpdatesDetails> allUpdates =
87              new TreeMap<Plugin, PluginUpdatesDetails>( new PluginComparator() );
88          allUpdates.putAll( pluginManagementUpdates );
89          allUpdates.putAll( pluginUpdates );
90          sb.append( getSummaryBlock( allUpdates ) );
91          sb.append( getPluginsInfoBlock( pluginManagementUpdates, "pluginManagements", "pluginManagement" ) );
92          sb.append( getPluginsInfoBlock( pluginUpdates, "plugins", "plugin" ) );
93          sb.append( "</PluginUpdatesReport>" ).append( NL );
94          PrintWriter pw;
95          try
96          {
97              pw = new PrintWriter( outputFileName, "UTF8" );
98              pw.print( sb.toString() );
99              pw.close();
100         }
101         catch ( IOException e )
102         {
103             throw new MavenReportException( "Cannot create xml report.", e );
104         }
105     }
106 
107     private static String getSummaryBlock( Map<Plugin, PluginUpdatesDetails> allUpdates )
108     {
109         Collection<ArtifactVersions> allVersions = new ArrayList<ArtifactVersions>();
110         for ( PluginUpdatesDetails details : allUpdates.values() )
111         {
112             allVersions.add( details.getArtifactVersions() );
113         }
114         return DependencyUpdatesXmlRenderer.getSummaryBlock( allVersions );
115     }
116 
117     private static String getPluginsInfoBlock( Map<Plugin, PluginUpdatesDetails> pluginUpdates, String blockName,
118                                                String subblockName )
119     {
120         StringBuilder sBuilder = new StringBuilder();
121         sBuilder.append( TAB ).append( OPEN_TAG ).append( blockName ).append( CLOSE_TAG ).append( NL );
122         for ( Entry<Plugin, PluginUpdatesDetails> entry : pluginUpdates.entrySet() )
123         {
124             sBuilder.append( TAB ).append( TAB ).append( OPEN_TAG ).append( subblockName ).append( CLOSE_TAG ).append( NL );
125 
126             Plugin plugin = entry.getKey();
127             sBuilder.append( TAB ).append( TAB ).append( TAB ).append( wrapElement( plugin.getGroupId(),
128                                                                                     GROUP_ID ) ).append( NL );
129             sBuilder.append( TAB ).append( TAB ).append( TAB ).append( wrapElement( plugin.getArtifactId(),
130                                                                                     ARTIFACT_ID ) ).append( NL );
131 
132             sBuilder.append( getVersionsBlocks( entry.getValue().getArtifactVersions() ) );
133 
134             sBuilder.append( TAB ).append( TAB ).append( OPEN_CLOSING_TAG ).append( subblockName ).append( CLOSE_TAG ).append( NL );
135         }
136         sBuilder.append( TAB ).append( OPEN_CLOSING_TAG ).append( blockName ).append( CLOSE_TAG ).append( NL );
137         return sBuilder.toString();
138     }
