47      @Test
48      public void test4DigitVersions() throws Exception
49      {
50          ArtifactVersion[] versions = versions( "1.0.0.1", "1.0.0.2", "2.121.2.1", "2.100.0.1", "3.1.0.1", "1.1.1" );
51          final DefaultArtifact artifact =
52              new DefaultArtifact( "group", "artifact", VersionRange.createFromVersionSpec( "[1.0,3.0]" ), "foo", "bar",
53                                   "jar", new DefaultArtifactHandler() );
54          // TODO This should also work for the MavenVersionComparator when using maven 3.x libraries
55          ArtifactVersions instance =
56              new ArtifactVersions( artifact, Arrays.asList( versions ), new MercuryVersionComparator() );
57          assertEquals( "artifact", instance.getArtifactId() );
58          assertEquals( "group", instance.getGroupId() );
59          assertThat( instance.getVersions(),
60                      arrayContaining( versions( "1.0.0.1", "1.0.0.2", "1.1.1", "2.100.0.1", "2.121.2.1",
61                                                          "3.1.0.1" ) ) );
62          assertThat( instance.getVersions( new DefaultArtifactVersion( "1.1" ), null ),
63                      arrayContaining( versions( "1.1.1", "2.100.0.1", "2.121.2.1", "3.1.0.1" ) ) );
64  
65          assertThat( instance.getVersions( new DefaultArtifactVersion( "1.0.0.2" ), null ),
66                      //Matchers.arrayContaining(versions("1.1.1", "2.121.2.1", "2.100.0.1", "3.1.0.1")));
67                      arrayContaining( versions( "1.1.1", "2.100.0.1", "2.121.2.1", "3.1.0.1" ) ) );
68  
69          assertEquals( new DefaultArtifactVersion( "2.121.2.1" ),
70                        instance.getNewestVersion( new DefaultArtifactVersion( "1.0" ),
71                                                   new DefaultArtifactVersion( "3.0" ) ) );
72          assertNull(
73              instance.getNewestVersion( new DefaultArtifactVersion( "1.1.1" ),
74                                         new DefaultArtifactVersion( "2.0" ) ) );
75      }

77      @Test
78      public void testSmokes()
79          throws Exception
80      {
81          ArtifactVersion[] versions = versions( "1.0", "3.0", "1.1", "1.0", "1.0.1" );
82          final DefaultArtifact artifact =
83              new DefaultArtifact( "group", "artifact", VersionRange.createFromVersionSpec( "[1.0,3.0]" ), "foo", "bar",
84                                   "jar", new DefaultArtifactHandler() );
85          ArtifactVersions instance =
86              new ArtifactVersions( artifact, Arrays.asList( versions ), new MavenVersionComparator() );
87          assertEquals( "artifact", instance.getArtifactId() );
88          assertEquals( "group", instance.getGroupId() );
89          assertArrayEquals(
90              versions( "1.0", "1.0.1", "1.1", "3.0" ),
91              instance.getVersions() );
92          assertArrayEquals( versions( "3.0" ),
93                             instance.getVersions( new DefaultArtifactVersion( "1.1" ), null ) );
94          assertArrayEquals(
95              versions( "1.1", "3.0" ),
96              instance.getVersions( new DefaultArtifactVersion( "1.0.1" ), null ) );
97          assertEquals( new DefaultArtifactVersion( "1.1" ),
98                        instance.getNewestVersion( new DefaultArtifactVersion( "1.0" ),
99                                                   new DefaultArtifactVersion( "3.0" ) ) );
100         assertNull(
101             instance.getNewestVersion( new DefaultArtifactVersion( "1.1" ), new DefaultArtifactVersion( "3.0" ) ) );
102     }

104     private ArtifactVersion[] versions( String... versions )
105     {
106         ArtifactVersion[] artifactVersions = new ArtifactVersion[versions.length];
107         for ( int i = 0; i < versions.length; i++ )
108         {
109             artifactVersions[i] = new DefaultArtifactVersion( versions[i] );
110         }
111         return artifactVersions;
112     }

114     @Test
115     public void testReportLabels()
116     {
117         ArtifactVersion[] versions = versions( "1.0.1", "1.0", "1.1.0-2", "1.1.1", "1.1.1-2", "1.1.2",
118                 "1.1.2-SNAPSHOT", "1.1.3", "1.1", "1.1-SNAPSHOT", "1.2.1", "1.2.2", "1.2", "1.3",
119                 "1.9.1-SNAPSHOT", "2.0", "2.1.1-SNAPSHOT", "2.1", "3.0", "3.1.1-SNAPSHOT",
120                 "3.1.5-SNAPSHOT", "3.4.0-SNAPSHOT" );
121         ArtifactVersions instance =
122                 new ArtifactVersions( new DefaultArtifact( "default-group", "dummy-api",
123                         "1.1", "foo", "bar",
124                         "jar", null ),
125                         Arrays.asList( versions ), new MavenVersionComparator() );
126 
127         assertThat( instance.getNewestUpdate( of( SUBINCREMENTAL ) ).toString(), is( "1.1.0-2" ) );
128         assertThat( instance.getNewestUpdate( of( INCREMENTAL ) ).toString(), is( "1.1.3" ) );
129     }

131     @Test
132     public void testGetNewerVersionsWithSnapshot() throws InvalidSegmentException
133     {
134         ArtifactVersion[] versions = versions( "1.0.0-SNAPSHOT", "1.0.0" );
135         ArtifactVersions instance =
136                 new ArtifactVersions( new DefaultArtifact( "default-group", "dummy-api",
137                         "1.0.0-SNAPSHOT", "foo", "bar",
138                         "jar", null ),
139                         Arrays.asList( versions ), new MavenVersionComparator() );
140 
141         assertThat( instance.getNewerVersions( "1.0.0-SNAPSHOT", of( SUBINCREMENTAL ), false, false ),
142                 arrayContaining( new DefaultArtifactVersion( "1.0.0" ) ) );
143     }
