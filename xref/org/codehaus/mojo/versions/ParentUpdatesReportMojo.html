44  /**
45   * Generates a report on available updates for parent artifacts
46   *
47   * @author Andrzej Jarmoniuk
48   * @since 2.13.0
49   */
50  @Mojo( name = "parent-updates-report",
51         requiresDependencyResolution = ResolutionScope.RUNTIME, threadSafe = true )
52  public class ParentUpdatesReportMojo extends AbstractVersionsReport<ParentUpdatesModel>
53  {
54      @Parameter( defaultValue = "${reactorProjects}", required = true, readonly = true )
55      protected List<MavenProject> reactorProjects;
56  
57      @Inject
58      protected ParentUpdatesReportMojo( I18N i18n, RepositorySystem repositorySystem, ArtifactResolver artifactResolver,
59                                         ArtifactMetadataSource artifactMetadataSource, WagonManager wagonManager,
60                                         ReportRendererFactory rendererFactory )
61      {
62          super( i18n, repositorySystem, artifactResolver, artifactMetadataSource, wagonManager, rendererFactory );
63      }
64  
65      /**
66       * {@inheritDoc}
67       */
68      public boolean isExternalReport()
69      {
70          return false;
71      }
72  
73      /**
74       * {@inheritDoc}
75       */
76      public boolean canGenerateReport()
77      {
78          if ( getProject().getParent() == null )
79          {
80              getLog().warn( "Project does not have a parent." );
81              return false;
82          }
83  
84          if ( reactorProjects.contains( getProject().getParent() ) )
85          {
86              getLog().warn( "Parent project is part of the reactor." );
87              return false;
88          }
89  
90          return true;
91      }
92  
93      /**
94       * generates an empty report in case there are no sources to generate a report with
95       *
96       * @param locale the locale to generate the report for.
97       * @param sink   the report formatting tool
98       */
99      @SuppressWarnings( "deprecation" )
100     protected void doGenerateReport( Locale locale, Sink sink ) throws MavenReportException
101     {
102         try
103         {
104             ArtifactVersions artifactVersions = getHelper().lookupArtifactVersions( project.getParentArtifact(),
105                     false );
106             artifactVersions.setIncludeSnapshots( allowSnapshots );
107             rendererFactory.createReportRenderer( getOutputName(), sink, locale,
108                             new ParentUpdatesModel( DependencyBuilder.newBuilder()
109                                     .withGroupId( artifactVersions.getGroupId() )
110                                     .withArtifactId( artifactVersions.getArtifactId() )
111                                     .withVersion( artifactVersions.getArtifact().getVersion() )
112                                     .withScope( artifactVersions.getArtifact().getScope() )
113                                     .withType( artifactVersions.getArtifact().getType() )
114                                     .withClassifier( artifactVersions.getArtifact().getClassifier() )
115                                     .build(), artifactVersions ) )
116                     .render();
117         }
118         catch ( ArtifactMetadataRetrievalException e )
119         {
120             throw new MavenReportException( e.getMessage(), e );
121         }
122     }
123 
124     /**
125      * {@inheritDoc}
126      */
127     public String getOutputName()
128     {
129         return "parent-updates-report";
130     }
131 }
