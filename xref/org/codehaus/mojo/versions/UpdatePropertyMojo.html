22  import org.apache.maven.artifact.versioning.ArtifactVersion;
23  import org.apache.maven.plugin.MojoExecutionException;
24  import org.apache.maven.plugin.MojoFailureException;
25  import org.apache.maven.plugins.annotations.Mojo;
26  import org.apache.maven.plugins.annotations.Parameter;
27  import org.codehaus.mojo.versions.api.ArtifactAssociation;
28  import org.codehaus.mojo.versions.api.PropertyVersions;
29  import org.codehaus.mojo.versions.rewriting.ModifiedPomXMLEventReader;
30  
31  import javax.xml.stream.XMLStreamException;
32  import java.util.Map;
33  
34  /**
35   * Sets a property to the latest version in a given range of associated artifacts.
36   *
37   * @author Eric Pabst
38   * @since 1.3
39   */
40  @Mojo( name = "update-property", requiresProject = true, requiresDirectInvocation = true, threadSafe = true )
41  public class UpdatePropertyMojo
42      extends AbstractVersionsUpdaterMojo
43  {
44  
45      // ------------------------------ FIELDS ------------------------------
46  
47      /**
48       * A property to update.
49       * 
50       * @since 1.3
51       */
52      @Parameter( property = "property" )
53      private String property = null;
54  
55      /**
56       * The new version to set the property to (can be a version range to find a version within).
57       * <ul>
58       * <li><code>1.0</code>x >= 1.0. The default Maven meaning for 1.0 is everything (,) but with 1.0 recommended.</li>
59       * <li><code>[1.0,2.0)</code> Versions 1.0 (included) to 2.0 (not included)</li>
60       * <li><code>[1.0,2.0]</code> Versions 1.0 to 2.0 (both included)</li>
61       * <li><code>[1.5,)</code> Versions 1.5 and higher</li>
62       * <li><code>(,1.0],[1.2,)</code> Versions up to 1.0 (included) and 1.2 or higher</li>
63       * </ul>
64       * If you like to define the version to be used exactly you have to use it like this:
65       * <code>-DnewVersion=[19.0]</code> otherwise a newer existing version will be used. If you need to downgrade a
66       * version you have to define <code>-DallowDowngrade=true</code> as well otherwise
67       * the version will be kept.
68       * 
69       * @since 1.3
70       */
71      @Parameter( property = "newVersion" )
72      private String newVersion = null;
73  
74      /**
75       * Whether properties linking versions should be auto-detected or not.
76       *
77       * @since 1.0-alpha-2
78       */
79      @Parameter( property = "autoLinkItems", defaultValue = "true" )
80      private boolean autoLinkItems;
81  
82      /**
83       * If a property points to a version like <code>1.2.3</code> and your repository contains versions like
84       * <code>1.2.3</code> and <code>1.1.0</code> without settings this to <code>true</code> the property will never
85       * being changed back to <code>1.1.0</code> by using <code>-DnewVersion=[1.1.0]</code>.
86       * 
87       * @since 3.0.0
88       */
89      @Parameter( property = "allowDowngrade", defaultValue = "false" )
90      private boolean allowDowngrade;
91  
92      /**
93       * Whether to allow the major version number to be changed.
94       *
95       * @since 2.4
96       */
97      @Parameter( property = "allowMajorUpdates", defaultValue = "true" )
98      protected boolean allowMajorUpdates;
99  
100     /**
101      * Whether to allow the minor version number to be changed.
102      *
103      * @since 2.4
104      */
105     @Parameter( property = "allowMinorUpdates", defaultValue = "true" )
106     protected boolean allowMinorUpdates;
107 
108     /**
109      * Whether to allow the incremental version number to be changed.
110      *
111      * @since 2.4
112      */
113     @Parameter( property = "allowIncrementalUpdates", defaultValue = "true" )
114     protected boolean allowIncrementalUpdates;
115 
116     // -------------------------- STATIC METHODS --------------------------
117 
118     // -------------------------- OTHER METHODS --------------------------
119 
120     /**
121      * @param pom the pom to update.
122      * @throws MojoExecutionException when things go wrong
123      * @throws MojoFailureException when things go wrong in a very bad way
124      * @throws XMLStreamException when things go wrong with XML streaming
125      * @see AbstractVersionsUpdaterMojo#update(ModifiedPomXMLEventReader)
126      * @since 1.0-alpha-1
127      */
128     protected void update( ModifiedPomXMLEventReader pom )
129         throws MojoExecutionException, MojoFailureException, XMLStreamException
130     {
131         Property propertyConfig = new Property( property );
132         propertyConfig.setVersion( newVersion );
133         Map<Property, PropertyVersions> propertyVersions =
134             this.getHelper().getVersionPropertiesMap( getProject(), new Property[] { propertyConfig }, property, "",
135                                                       autoLinkItems );
136         for ( Map.Entry<Property, PropertyVersions> entry : propertyVersions.entrySet() )
137         {
138             Property property = entry.getKey();
139             PropertyVersions version = entry.getValue();
140 
141             final String currentVersion = getProject().getProperties().getProperty( property.getName() );
142             if ( currentVersion == null )
143             {
144                 continue;
145             }
146 
147             int segment = determineUnchangedSegment( allowMajorUpdates, allowMinorUpdates, allowIncrementalUpdates );
148             ArtifactVersion targetVersion = updatePropertyToNewestVersion( pom, property, version, currentVersion,
149                     allowDowngrade, segment );
150 
151             if (targetVersion != null)
152             {
153                 for ( final ArtifactAssociation association : version.getAssociations() )
154                 {
155                     this.getChangeRecorder().recordUpdate( "updateProperty", association.getGroupId(),
156                             association.getArtifactId(), currentVersion, targetVersion.toString() );
157                 }
158             }
159         }
160     }
