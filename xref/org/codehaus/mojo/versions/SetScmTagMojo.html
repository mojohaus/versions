21  /**
22   * Updates the current project's SCM tag.
23   *
24   * @author Anton Johansson
25   * @since 2.5
26   */
27  @Mojo( name = "set-scm-tag", requiresDirectInvocation = true, aggregator = true, threadSafe = true )
28  public class SetScmTagMojo extends AbstractVersionsUpdaterMojo
29  {
30  
31      /**
32       * The new SCM tag to set.
33       *
34       * @since 2.5
35       */
36      @Parameter( property = "newTag" )
37      private String newTag;
38  
39      /**
40       * The new SCM connection property
41       *
42       * @since 2.12.0
43       */
44      @Parameter( property = "connection" )
45      private String connection;
46  
47      /**
48       * The new SCM developerConnection property
49       *
50       * @since 2.12.0
51       */
52      @Parameter( property = "developerConnection" )
53      private String developerConnection;
54  
55      /**
56       * The new SCM url property
57       *
58       * @since 2.12.0
59       */
60      @Parameter( property = "url" )
61      private String url;
62  
63      /**
64       * Called when this mojo is executed.
65       *
66       * @throws org.apache.maven.plugin.MojoExecutionException when things go wrong.
67       * @throws org.apache.maven.plugin.MojoFailureException   when things go wrong.
68       */
69      @Override
70      public void execute() throws MojoExecutionException, MojoFailureException
71      {
72          if ( isAllBlank( newTag, connection, developerConnection, url ) )
73          {
74              throw new MojoFailureException(
75                      "One of: \"newTag\", \"connection\", \"developerConnection\", \"url\" should be provided." );
76          }
77  
78          super.execute();
79      }
80  
81      @Override
82      protected void update( ModifiedPomXMLEventReader pom )
83              throws MojoExecutionException, MojoFailureException, XMLStreamException, ArtifactMetadataRetrievalException
84      {
85          try
86          {
87              Scm scm = PomHelper.getRawModel( pom ).getScm();
88              if ( scm == null )
89              {
90                  throw new MojoFailureException( "No <scm> was present" );
91              }
92  
93              List<String> failures = new ArrayList<>();
94              if ( !isBlank( newTag ) )
95              {
96                  getLog().info( "Updating tag: " + scm.getTag() + " -> " + newTag );
97                  if ( !PomHelper.setProjectValue( pom, "/project/scm/tag", newTag ) )
98                  {
99                      failures.add( "tag: " + newTag );
100                 }
101             }
102             if ( !isBlank( connection ) )
103             {
104                 getLog().info( "Updating connection: " + scm.getConnection() + " -> " + connection );
105                 if ( !PomHelper.setProjectValue( pom, "/project/scm/connection", connection ) )
106                 {
107                     failures.add( "connection: " + connection );
108                 }
109             }
110             if ( !isBlank( developerConnection ) )
111             {
112                 getLog().info( "Updating developerConnection: " + scm.getDeveloperConnection() + " -> "
113                         + developerConnection );
114                 if ( !PomHelper.setProjectValue( pom, "/project/scm/developerConnection", developerConnection ) )
115                 {
116                     failures.add( "developerConnection: " + developerConnection );
117                 }
118             }
119             if ( !isBlank( url ) )
120             {
121                 getLog().info( "Updating url: " + scm.getUrl() + " -> " + url );
122                 if ( !PomHelper.setProjectValue( pom, "/project/scm/url", url ) )
123                 {
124                     failures.add( "url: " + url );
125                 }
126             }
127             if ( !failures.isEmpty() )
128             {
129                 throw new MojoFailureException( "Could not update one or more SCM elements: " + String.join( ", ",
130                         failures ) + ". Please make sure they are present in the original POM. " );
131             }
132         }
133         catch ( IOException e )
134         {
135             throw new MojoExecutionException( e.getMessage(), e );
136         }
137     }
138 }
