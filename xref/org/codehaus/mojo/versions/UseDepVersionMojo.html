26  import org.apache.maven.artifact.Artifact;
27  import org.apache.maven.artifact.metadata.ArtifactMetadataRetrievalException;
28  import org.apache.maven.model.Dependency;
29  import org.apache.maven.plugin.MojoExecutionException;
30  import org.apache.maven.plugin.MojoFailureException;
31  import org.apache.maven.plugins.annotations.Mojo;
32  import org.apache.maven.plugins.annotations.Parameter;
33  import org.codehaus.mojo.versions.api.ArtifactVersions;
34  import org.codehaus.mojo.versions.api.PomHelper;
35  import org.codehaus.mojo.versions.rewriting.ModifiedPomXMLEventReader;

37  /**
38   * Updates a dependency to a specific version.
39   * This can be useful if you have to manage versions for a very large (100+ module) projects where you canâ€™t always use
40   * the most up-to-date version of a particular third party component.
41   *
42   * @author Dan Arcari
43   * @since 2.3
44   */
45  @Mojo( name = "use-dep-version",
46         requiresProject = true,
47         threadSafe = true )
48  public class UseDepVersionMojo extends AbstractVersionsDependencyUpdaterMojo
49  {
50  
51      /**
52       * The exact version to be applied for the included dependencies
53       */
54      @Parameter( property = "depVersion",
55                  required = true )
56      protected String depVersion;
57  
58      /**
59       * If set to true, will use whatever version is supplied without attempting to validate that such a version is
60       * obtainable from the repository chain.
61       */
62      @Parameter( property = "forceVersion",
63                  defaultValue = "false" )
64      protected boolean forceVersion;
65  
66      @Override
67      protected void update( ModifiedPomXMLEventReader pom )
68          throws MojoExecutionException, MojoFailureException, XMLStreamException, ArtifactMetadataRetrievalException
69      {
70  
71          if ( depVersion == null || depVersion.equals( "" ) )
72          {
73              throw new IllegalArgumentException(
74                  "depVersion must be supplied with use-specific-version, and cannot be blank." );
75          }
76  
77          if ( !forceVersion && !hasIncludes() )
78          {
79              throw new IllegalArgumentException(
80                  "The use-specific-version goal is intended to be used with a single artifact. "
81                      + "Please specify a value for the 'includes' parameter, "
82                      + "or use -DforceVersion=true to override this check." );
83          }
84  
85          try
86          {
87              if ( getProject().getDependencyManagement() != null && isProcessingDependencyManagement() )
88              {
89                  useDepVersion( pom, getProject().getDependencyManagement().getDependencies() );
90              }
91  
92              if ( getProject().getDependencies() != null && isProcessingDependencies() )
93              {
94                  useDepVersion( pom, getProject().getDependencies() );
95              }
96          }
97          catch ( ArtifactMetadataRetrievalException e )
98          {
99              throw new MojoExecutionException( e.getMessage(), e );
100         }
101     }
102 
103     private void useDepVersion( ModifiedPomXMLEventReader pom, Collection<Dependency> dependencies )
104         throws MojoExecutionException, XMLStreamException, ArtifactMetadataRetrievalException
105     {
106         for ( Dependency dep : dependencies )
107         {
108             if ( isExcludeReactor() && isProducedByReactor( dep ) )
109             {
110                 getLog().info( "Ignoring reactor dependency: " + toString( dep ) );
111                 continue;
112             }
113 
114             if ( isHandledByProperty( dep ) )
115             {
116                 getLog().debug( "Ignoring dependency with property as version: " + toString( dep ) );
117                 continue;
118             }
119 
120             Artifact artifact = this.toArtifact( dep );
121 
122             if ( isIncluded( artifact ) )
123             {
124                 if ( !forceVersion )
125                 {
126                     ArtifactVersions versions = getHelper().lookupArtifactVersions( artifact, false );
127 
128                     if ( !versions.containsVersion( depVersion ) )
129                     {
130                         throw new MojoExecutionException(
131                             String.format( "Version %s is not available for artifact %s:%s",
132                                            depVersion, artifact.getGroupId(), artifact.getArtifactId() ) );
133                     }
134                 }
135 
136                 String version = dep.getVersion();
137 
138                 if ( PomHelper.setDependencyVersion( pom, dep.getGroupId(), dep.getArtifactId(), version, depVersion,
139                                                      getProject().getModel() ) )
140                 {
141                     getLog().info( "Updated " + toString( dep ) + " to version " + depVersion );
142 
143                     this.getChangeRecorder().recordUpdate( "useDependencyVersion", dep.getGroupId(),
144                                                            dep.getArtifactId(), version, depVersion );
145                 }
146             }
147         }
148     }
149 }
