22  import org.apache.maven.doxia.sink.Sink;
23  import org.apache.maven.plugin.MojoExecutionException;
24  import org.apache.maven.reporting.MavenReportException;
25  
26  import java.util.Locale;
27  import java.util.Map;
28  
29  /**
30   * Generates a report of available updates for properties of a project which are linked to the dependencies and/or
31   * plugins of a project.
32   *
33   * @author Stephen Connolly
34   * @goal property-updates-report
35   * @requiresDependencyResolution runtime
36   * @requiresProject true
37   * @since 1.0-beta-1
38   */
39  public class PropertyUpdatesReport
40      extends AbstractVersionsReport
41  {
42  
43  
44      /**
45       * Any restrictions that apply to specific properties.
46       *
47       * @parameter
48       * @since 1.0-beta-1
49       */
50      private Property[] properties;
51  
52      /**
53       * A comma separated list of properties to include in the report.
54       *
55       * @parameter property="includeProperties"
56       * @since 1.0-beta-1
57       */
58      private String includeProperties = null;
59  
60      /**
61       * A comma separated list of properties to not include in the report.
62       *
63       * @parameter property="excludeProperties"
64       * @since 1.0-beta-1
65       */
66      private String excludeProperties = null;
67  
68      /**
69       * Whether properties linking versions should be auto-detected or not.
70       *
71       * @parameter property="autoLinkItems" defaultValue="true"
72       * @since 1.0-beta-1
73       */
74      private Boolean autoLinkItems;
75  
76  
77      /**
78       * {@inheritDoc}
79       */
80      public boolean isExternalReport()
81      {
82          return false;
83      }
84  
85      /**
86       * {@inheritDoc}
87       */
88      public boolean canGenerateReport()
89      {
90          return haveBuildProperties();
91      }
92  
93      private boolean haveBuildProperties()
94      {
95          return getProject().getProperties() != null && !getProject().getProperties().isEmpty();
96      }
97  
98      protected void doGenerateReport( Locale locale, Sink sink )
99          throws MavenReportException
100     {
101         final Map updateSet;
102         try
103         {
104             updateSet =
105                 getHelper().getVersionPropertiesMap( getProject(), properties, includeProperties, excludeProperties,
106                                                      !Boolean.FALSE.equals( autoLinkItems ) );
107         }
108         catch ( MojoExecutionException e )
109         {
110             throw new MavenReportException( e.getMessage(), e );
111         }
112         PropertyUpdatesRenderer renderer =
113             new PropertyUpdatesRenderer( sink, getI18n(), getOutputName(), locale, updateSet );
114         renderer.render();
115     }
116 
117     /**
118      * {@inheritDoc}
119      */
120     public String getOutputName()
121     {
122         return "property-updates-report";
123     }
124 }
