71      public void testPerRuleVersionsIgnored() throws Exception
72      {
73          final ArtifactMetadataSource metadataSource = mock( ArtifactMetadataSource.class );
74          final Artifact artifact = mock( Artifact.class );
75          when( artifact.getGroupId() ).thenReturn( "com.mycompany.maven" );
76          when( artifact.getArtifactId() ).thenReturn( "artifact-one" );
77          
78          final List<ArtifactVersion> artifactVersions = new ArrayList<ArtifactVersion>();
79          
80          artifactVersions.add( new DefaultArtifactVersion( "one" ) );
81          artifactVersions.add( new DefaultArtifactVersion( "two" ) );
82          final ArtifactVersion three = new DefaultArtifactVersion( "three" );
83          artifactVersions.add( three );
84          final ArtifactVersion oneTwoHundred = new DefaultArtifactVersion( "1.200" );
85          artifactVersions.add( oneTwoHundred );
86          final ArtifactVersion illegal = new DefaultArtifactVersion( "illegalVersion" );
87          artifactVersions.add( illegal );
88  
89          when( metadataSource.retrieveAvailableVersions( same( artifact ), any( ArtifactRepository.class ), anyList() ) ).thenReturn( artifactVersions );
90          
91          VersionsHelper helper = createHelper( metadataSource );
92          
93          final ArtifactVersions versions = helper.lookupArtifactVersions( artifact, true );
94          
95          final List<ArtifactVersion> actual = asList( versions.getVersions( true ) );
96          
97          assertEquals( 3, actual.size() );
98          assertThat( actual, hasItems( three, oneTwoHundred, illegal ) );
99      }

133     public void testWildcardMatching()
134         throws Exception
135     {
136         assertTrue( DefaultVersionsHelper.exactMatch( "*", "com.foo.bar" ) );
137         assertFalse( DefaultVersionsHelper.exactMatch( "com.bar*", "com-bar" ) );
138         assertTrue( DefaultVersionsHelper.exactMatch( "com?foo.bar", "com.foo.bar" ) );
139         assertTrue( DefaultVersionsHelper.exactMatch( "co*.foo.b?r", "com.foo.bar" ) );
140         assertTrue( DefaultVersionsHelper.exactMatch( "c*oo*r", "com.foo.bar" ) );
141     }

143     public void testRuleSets()
144         throws Exception
145     {
146         VersionsHelper helper = createHelper();
147 
148         assertEquals( "no match gives default", VersionComparators.getVersionComparator( "maven" ),
149                       helper.getVersionComparator( "net.foo", "bar" ) );
150         assertEquals( "matches wildcard", VersionComparators.getVersionComparator( "mercury" ),
151                       helper.getVersionComparator( "org.apache.maven", "plugins" ) );
152         assertEquals( "exact match wins over initial match", VersionComparators.getVersionComparator( "mercury" ),
153                       helper.getVersionComparator( "com.mycompany.custom.maven", "plugins" ) );
154         assertEquals( "non-wildcard prefix wins over wildcard prefix match",
155                       VersionComparators.getVersionComparator( "maven" ),
156                       helper.getVersionComparator( "com.mycompany.maven.plugins", "plugins" ) );
157         assertEquals( VersionComparators.getVersionComparator( "maven" ),
158                       helper.getVersionComparator( "com.mycompany.maven", "new-maven-plugin" ) );
159         assertEquals( VersionComparators.getVersionComparator( "mercury" ),
160                       helper.getVersionComparator( "com.mycompany.maven", "old-maven-plugin" ) );
161     }

179     private VersionsHelper createHelper()
180         throws MojoExecutionException
181     {
182         return createHelper( new MavenMetadataSource() );
183     }
184     
185     private VersionsHelper createHelper( ArtifactMetadataSource metadataSource ) throws MojoExecutionException
186     {
187         final String resourcePath = "/" + getClass().getPackage().getName().replace( '.', '/' ) + "/rules.xml";
188         final String rulesUri = getClass().getResource( resourcePath ).toExternalForm();
189         VersionsHelper helper = createHelper( rulesUri, metadataSource );
190         return helper;
191     }
192 
193     private VersionsHelper createHelper( String rulesUri, ArtifactMetadataSource metadataSource )
194         throws MojoExecutionException
195     {
196         final DefaultWagonManager wagonManager = new DefaultWagonManager()
197         {
198             public Wagon getWagon( Repository repository )
199                 throws UnsupportedProtocolException, WagonConfigurationException
200             {
201                 return new FileWagon();
202             }
203         };
204 
205         VersionsHelper helper =
206             new DefaultVersionsHelper( new DefaultArtifactFactory(), new DefaultArtifactResolver(), metadataSource, new ArrayList(),
207                                        new ArrayList(),
208                                        new DefaultArtifactRepository( "", "", new DefaultRepositoryLayout() ),
209                                        wagonManager, new Settings(), "", rulesUri, mock( Log.class ), mock( MavenSession.class ),
210                                        new DefaultPathTranslator());
211         return helper;
212     }
