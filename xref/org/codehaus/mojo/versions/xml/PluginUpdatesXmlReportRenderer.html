55  /**
56   * XML renderer for DependencyUpdatesReport creates an xml file in target directory and writes report about available
57   * dependency/dependency management updates.
58   *
59   * @author Illia Dubinin
60   * @since 2.4
61   */
62  public class PluginUpdatesXmlReportRenderer implements ReportRenderer
63  {
64      private final PluginUpdatesModel model;
65      private final Path outputFile;
66      private final ArtifactVersionsCache newestUpdateCache
67              = new ArtifactVersionsCache( AbstractVersionDetails::getNewestUpdate );
68      /**
69       * Creates a new instance
70       * @param model object containing the updates model
71       * @param outputFile output file for the report
72       */
73      public PluginUpdatesXmlReportRenderer( PluginUpdatesModel model, Path outputFile )
74      {
75          this.model = model;
76          this.outputFile = outputFile;
77      }
78  
79      /**
80       * Creates an XML report
81       */
82      @Override
83      public void render()
84      {
85          try ( BufferedWriter writer = Files.newBufferedWriter( outputFile,
86                  StandardCharsets.UTF_8 ) )
87          {
88              new PluginUpdatesReportXpp3Writer().write( writer, new PluginUpdatesReport()
89              {{
90                  setSummary( new PluginReportSummary()
91                  {{
92                      PluginOverviewStats overviewStats = PluginOverviewStats.fromUpdates(
93                              model.getAllUpdates().values(), newestUpdateCache );
94                      setUsingLastVersion( String.valueOf( overviewStats.getUpToDate() ) );
95                      setNextVersionAvailable( String.valueOf( overviewStats.getAny() ) );
96                      setNextIncrementalAvailable( String.valueOf( overviewStats.getIncremental() ) );
97                      setNextMinorAvailable( String.valueOf( overviewStats.getMinor() ) );
98                      setNextMajorAvailable( String.valueOf( overviewStats.getMajor() ) );
99                      setDependencyUpdates( String.valueOf( overviewStats.getDependencies() ) );
100                 }} );
101                 setPluginManagements( createPluginInfo( model.getArtifactManagementUpdates() ) );
102                 setPlugins( createPluginInfo( model.getArtifactUpdates() ) );
103             }} );
104         }
105         catch ( IOException e )
106         {
107             throw new RuntimeException( e );
108         }
109     }
110 
111     private static void setSection( ArtifactVersions versions, Segment segment, Consumer<List<String>> setterFunction )
112     {
113         ofNullable( versions.getAllUpdates( of( segment ) ) )
114                 .map( v -> Arrays.stream( v )
115                         .map( ArtifactVersion::toString )
116                         .collect( Collectors.toList() ) )
117                 .ifPresent( setterFunction );
118     }
119 
120     private static List<PluginInfo> createPluginInfo( Map<Dependency, PluginUpdatesDetails> versions )
121     {
122         return versions.entrySet().stream().map( e ->
123                 new PluginInfo()
124         {{
125             setGroupId( e.getKey().getGroupId() );
126             setArtifactId( e.getKey().getArtifactId() );
127             setCurrentVersion( e.getKey().getVersion() );
128             setScope( e.getKey().getScope() );
129             setType( e.getKey().getType() );
130             setClassifier( e.getKey().getClassifier() );
131 
132             ofNullable( e.getValue().getNewestUpdate( empty() ) )
133                     .map( ArtifactVersion::toString ).ifPresent( this::setLastVersion );
134 
135             setSection( e.getValue(), INCREMENTAL, this::setIncrementals );
136             setSection( e.getValue(), MINOR, this::setMinors );
137             setSection( e.getValue(), MAJOR, this::setMajors );
138 
139             setStatus( getLastVersion() == null
140                     ? "no new available"
141                     : getIncrementals() != null
142                     ? "incremental available"
143                     : getMinors() != null
144                     ? "minor available"
145                     : "major available" );
146         }} ).collect( Collectors.toList() );
147     }
