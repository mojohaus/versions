23  import java.util.HashMap;
24  import java.util.Map;
25  import java.util.Optional;
26  import java.util.function.BiFunction;
27  
28  import org.apache.commons.lang3.tuple.Pair;
29  
30  /**
31   * Utility providing a cached {@link ArtifactVersions#getNewestUpdate(Optional)} API
32   */
33  public class ArtifactVersionsCache
34  {
35      private BiFunction<AbstractVersionDetails, Optional<Segment>, ?> cachedFunction;
36  
37      private Map<Pair<? extends AbstractVersionDetails, Optional<Segment>>, Object> updateCache = new HashMap<>();
38  
39      /**
40       * Constructs a new instance given the concrete function for obtaining the details
41       *
42       * @param cachedFunction reference to the function computing the required information
43       */
44      public ArtifactVersionsCache( BiFunction<AbstractVersionDetails, Optional<Segment>, ?>
45                                            cachedFunction )
46      {
47          this.cachedFunction = cachedFunction;
48      }
49  
50      /**
51       * Returns the required information for the given {@link ArtifactVersions} object and the given update scope.
52       * If the information is already present in cache, the cached version is returned. Otherwise,
53       * the {@code artifactVersions} object is queried and the response is cached.
54       *
55       * @param artifactVersions {@linkplain ArtifactVersions} object referring to the given dependency
56       * @param updateScope      update scope
57       * @return last retrieved update information
58       */
59      @SuppressWarnings( "unchecked" )
60      public <V extends AbstractVersionDetails, R> R get( V artifactVersions,
61                                                          Optional<Segment> updateScope )
62      {
63          return (R) updateCache.computeIfAbsent( Pair.of( artifactVersions, updateScope ),
64                  pair -> cachedFunction.apply( pair.getLeft(), pair.getRight() ) );
65      }
66  }
