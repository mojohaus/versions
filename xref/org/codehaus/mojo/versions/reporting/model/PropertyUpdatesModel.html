22  import java.util.Map;
23  import java.util.TreeMap;
24  
25  import org.codehaus.mojo.versions.Property;
26  import org.codehaus.mojo.versions.api.PropertyVersions;
27  import org.codehaus.mojo.versions.utils.PropertyComparator;
28  
29  /**
30   * Model class for using with the {@linkplain org.codehaus.mojo.versions.api.ReportRenderer} API
31   */
32  public class PropertyUpdatesModel
33  {
34      private final Map<Property, PropertyVersions> allUpdates;
35  
36      public PropertyUpdatesModel( Map<Property, PropertyVersions> propertyUpdates )
37      {
38          this.allUpdates = new TreeMap<>( PropertyComparator.INSTANCE );
39          this.allUpdates.putAll( propertyUpdates );
40      }
41  
42      public Map<Property, PropertyVersions> getAllUpdates()
43      {
44          return allUpdates;
45      }
46  }
