22  import javax.xml.stream.XMLStreamException;
23  import java.util.ArrayList;
24  import java.util.Arrays;
25  import java.util.List;
26  
27  /**
28   * Created by IntelliJ IDEA.
29   *
30   * @author Stephen Connolly
31   * @since 15-Sep-2010 16:01:35
32   */
33  public class CompositeVersionChanger
34      implements VersionChanger
35  {
36      private final List<VersionChanger> composites;
37  
38      public CompositeVersionChanger( VersionChanger[] composites )
39      {
40          this.composites = new ArrayList<VersionChanger>( Arrays.asList( composites ) );
41      }
42  
43      public CompositeVersionChanger( List<VersionChanger> composites )
44      {
45          this.composites = new ArrayList<VersionChanger>( composites );
46      }
47  
48      public void apply( VersionChange versionChange )
49          throws XMLStreamException
50      {
51          for ( VersionChanger delegate : composites )
52          {
53              delegate.apply( versionChange );
54          }
55      }
56  }
