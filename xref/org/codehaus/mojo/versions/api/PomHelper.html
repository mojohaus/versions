105         FileInputStream input = null;

112         catch ( XmlPullParserException e )
113         {
114             IOException ioe = new IOException( e.getMessage() );

116             throw ioe;
117         }
118         finally
119         {

122                 input.close();
123             }
124         }

159     /**
160      * Searches the pom re-defining the specified property to the specified version.
161      *
162      * @param pom The pom to modify.
163      * @param profileId The profile in which to modify the property.
164      * @param property The property to modify.
165      * @param value The new value of the property.
166      * @return <code>true</code> if a replacement was made.
167      * @throws XMLStreamException if somethinh went wrong.
168      */
169     public static boolean setPropertyVersion( final ModifiedPomXMLEventReader pom, final String profileId,
170                                               final String property, final String value )
171                                                   throws XMLStreamException
172     {
173         Stack<String> stack = new Stack<String>();
174         String path = "";
175         final Pattern propertyRegex;
176         final Pattern matchScopeRegex;
177         final Pattern projectProfileId;
178         boolean inMatchScope = false;
179         boolean madeReplacement = false;
180         if ( profileId == null )
181         {
182             propertyRegex = Pattern.compile( "/project/properties/" + RegexUtils.quote( property ) );
183             matchScopeRegex = Pattern.compile( "/project/properties" );
184             projectProfileId = null;
185         }
186         else
187         {
188             propertyRegex = Pattern.compile( "/project/profiles/profile/properties/" + RegexUtils.quote( property ) );
189             matchScopeRegex = Pattern.compile( "/project/profiles/profile" );
190             projectProfileId = Pattern.compile( "/project/profiles/profile/id" );
191         }
192 
193         pom.rewind();
194 
195         while ( pom.hasNext() )
196         {
197             XMLEvent event = pom.nextEvent();
198             if ( event.isStartElement() )
199             {
200                 stack.push( path );
201                 path = path + "/" + event.asStartElement().getName().getLocalPart();
202 
203                 if ( propertyRegex.matcher( path ).matches() )
204                 {
205                     pom.mark( 0 );
206                 }
207                 else if ( matchScopeRegex.matcher( path ).matches() )
208                 {
209                     // we're in a new match scope
210                     // reset any previous partial matches
211                     inMatchScope = profileId == null;
212                     pom.clearMark( 0 );
213                     pom.clearMark( 1 );
214                 }
215                 else if ( profileId != null && projectProfileId.matcher( path ).matches() )
216                 {
217                     String candidateId = pom.getElementText();
218                     path = stack.pop(); // since getElementText will be after the end element
219 
220                     inMatchScope = profileId.trim().equals( candidateId.trim() );
221                 }
222             }
223             if ( event.isEndElement() )
224             {
225                 if ( propertyRegex.matcher( path ).matches() )
226                 {
227                     pom.mark( 1 );
228                 }
229                 else if ( matchScopeRegex.matcher( path ).matches() )
230                 {
231                     if ( inMatchScope && pom.hasMark( 0 ) && pom.hasMark( 1 ) )
232                     {
233                         pom.replaceBetween( 0, 1, value );
234                         madeReplacement = true;
235                     }
236                     pom.clearMark( 0 );
237                     pom.clearMark( 1 );
238                     inMatchScope = false;
239                 }
240                 path = stack.pop();
241             }
242         }
243         return madeReplacement;
244     }

396     /**
397      * Gets the parent artifact from the pom.
398      *
399      * @param pom The pom.
400      * @param helper The helper (used to create the artifact).
401      * @return The parent artifact or <code>null</code> if no parent is specified.
402      * @throws XMLStreamException if something went wrong.
403      */
404     public static Artifact getProjectParent( final ModifiedPomXMLEventReader pom, VersionsHelper helper )
405         throws XMLStreamException
406     {
407         Stack<String> stack = new Stack<String>();
408         String path = "";
409         final Pattern matchScopeRegex = Pattern.compile( "/project/parent((/groupId)|(/artifactId)|(/version))" );
410         String groupId = null;
411         String artifactId = null;
412         String version = null;
413 
414         pom.rewind();
415 
416         while ( pom.hasNext() )
417         {
418             XMLEvent event = pom.nextEvent();
419             if ( event.isStartElement() )
420             {
421                 stack.push( path );
422                 final String elementName = event.asStartElement().getName().getLocalPart();
423                 path = path + "/" + elementName;
424 
425                 if ( matchScopeRegex.matcher( path ).matches() )
426                 {
427                     if ( "groupId".equals( elementName ) )
428                     {
429                         groupId = pom.getElementText().trim();
430                         path = stack.pop();
431                     }
432                     else if ( "artifactId".equals( elementName ) )
433                     {
434                         artifactId = pom.getElementText().trim();
435                         path = stack.pop();
436                     }
437                     else if ( "version".equals( elementName ) )
438                     {
439                         version = pom.getElementText().trim();
440                         path = stack.pop();
441                     }
442                 }
443             }
444             if ( event.isEndElement() )
445             {
446                 path = stack.pop();
447             }
448         }
449         if ( groupId == null || artifactId == null || version == null )
450         {
451             return null;
452         }
453         return helper.createDependencyArtifact( groupId, artifactId, VersionRange.createFromVersion( version ), "pom",
454                                                 null, null, false );
455     }

457     /**
458      * Searches the pom re-defining the specified dependency to the specified version.
459      *
460      * @param pom The pom to modify.
461      * @param groupId The groupId of the dependency.
462      * @param artifactId The artifactId of the dependency.
463      * @param oldVersion The old version of the dependency.
464      * @param newVersion The new version of the dependency.
465      * @return <code>true</code> if a replacement was made.
466      * @throws XMLStreamException if somethinh went wrong.
467      */
468     public static boolean setDependencyVersion( final ModifiedPomXMLEventReader pom, final String groupId,
469                                                 final String artifactId, final String oldVersion,
470                                                 final String newVersion )
471                                                     throws XMLStreamException
472     {
473         Stack<String> stack = new Stack<String>();
474         String path = "";
475 
476         Set<String> implicitPaths =
477             new HashSet<String>( Arrays.<String>asList( "/project/parent/groupId", "/project/parent/artifactId",
478                                                         "/project/parent/version", "/project/groupId",
479                                                         "/project/artifactId", "/project/version" ) );
480         Map<String, String> implicitProperties = new HashMap<String, String>();
481 
482         pom.rewind();
483 
484         while ( pom.hasNext() )
485         {
486             while ( pom.hasNext() )
487             {
488                 XMLEvent event = pom.nextEvent();
489                 if ( event.isStartElement() )
490                 {
491                     stack.push( path );
492                     final String elementName = event.asStartElement().getName().getLocalPart();
493                     path = path + "/" + elementName;
494 
495                     if ( implicitPaths.contains( path ) )
496                     {
497                         final String elementText = pom.getElementText().trim();
498                         implicitProperties.put( path.substring( 1 ).replace( '/', '.' ), elementText );
499                         path = stack.pop();
500                     }
501                 }
502                 if ( event.isEndElement() )
503                 {
504                     path = stack.pop();
505                 }
506             }
507         }
508 
509         boolean modified = true;
510         while ( modified )
511         {
512             modified = false;
513             for ( Map.Entry<String, String> entry : implicitProperties.entrySet() )
514             {
515                 if ( entry.getKey().contains( ".parent" ) )
516                 {
517                     String child = entry.getKey().replace( ".parent", "" );
518                     if ( !implicitProperties.containsKey( child ) )
519                     {
520                         implicitProperties.put( child, entry.getValue() );
521                         modified = true;
522                         break;
523                     }
524                 }
525             }
526         }
527 
528         stack = new Stack<String>();
529         path = "";
530         boolean inMatchScope = false;
531         boolean madeReplacement = false;
532         boolean haveGroupId = false;
533         boolean haveArtifactId = false;
534         boolean haveOldVersion = false;
535 
536         final Pattern matchScopeRegex = Pattern.compile( "/project" + "(/profiles/profile)?"
537             + "((/dependencyManagement)|(/build(/pluginManagement)?/plugins/plugin))?" + "/dependencies/dependency" );
538 
539         final Pattern matchTargetRegex = Pattern.compile( "/project" + "(/profiles/profile)?"
540             + "((/dependencyManagement)|(/build(/pluginManagement)?/plugins/plugin))?" + "/dependencies/dependency"
541             + "((/groupId)|(/artifactId)|(/version))" );
542 
543         pom.rewind();
544 
545         while ( pom.hasNext() )
546         {
547             XMLEvent event = pom.nextEvent();
548             if ( event.isStartElement() )
549             {
550                 stack.push( path );
551                 final String elementName = event.asStartElement().getName().getLocalPart();
552                 path = path + "/" + elementName;
553 
554                 if ( matchScopeRegex.matcher( path ).matches() )
555                 {
556                     // we're in a new match scope
557                     // reset any previous partial matches
558                     inMatchScope = true;
559                     pom.clearMark( 0 );
560                     pom.clearMark( 1 );
561 
562                     haveGroupId = false;
563                     haveArtifactId = false;
564                     haveOldVersion = false;
565                 }
566                 else if ( inMatchScope && matchTargetRegex.matcher( path ).matches() )
567                 {
568                     if ( "groupId".equals( elementName ) )
569                     {
570                         haveGroupId = groupId.equals( evaluate( pom.getElementText().trim(), implicitProperties ) );
571                         path = stack.pop();
572                     }
573                     else if ( "artifactId".equals( elementName ) )
574                     {
575                         haveArtifactId =
576                             artifactId.equals( evaluate( pom.getElementText().trim(), implicitProperties ) );
577                         path = stack.pop();
578                     }
579                     else if ( "version".equals( elementName ) )
580                     {
581                         pom.mark( 0 );
582                     }
583                 }
584             }
585             if ( event.isEndElement() )
586             {
587                 if ( matchTargetRegex.matcher( path ).matches()
588                     && "version".equals( event.asEndElement().getName().getLocalPart() ) )
589                 {
590                     pom.mark( 1 );
591                     String compressedPomVersion = StringUtils.deleteWhitespace( pom.getBetween( 0, 1 ).trim() );
592                     String compressedOldVersion = StringUtils.deleteWhitespace( oldVersion );
593 
594                     try
595                     {
596                         haveOldVersion = isVersionOverlap( compressedOldVersion, compressedPomVersion );
597                     }
598                     catch ( InvalidVersionSpecificationException e )
599                     {
600                         // fall back to string comparison
601                         haveOldVersion = compressedOldVersion.equals( compressedPomVersion );
602                     }
603                 }
604                 else if ( matchScopeRegex.matcher( path ).matches() )
605                 {
606                     if ( inMatchScope && pom.hasMark( 0 ) && pom.hasMark( 1 ) && haveGroupId && haveArtifactId
607                         && haveOldVersion )
608                     {
609                         pom.replaceBetween( 0, 1, newVersion );
610                         madeReplacement = true;
611                     }
612                     pom.clearMark( 0 );
613                     pom.clearMark( 1 );
614                     haveArtifactId = false;
615                     haveGroupId = false;
616                     haveOldVersion = false;
617                     inMatchScope = false;
618                 }
619                 path = stack.pop();
620             }
621         }
622         return madeReplacement;
623     }

625     /**
626      * A lightweight expression evaluation function.
627      *
628      * @param expr The expression to evaluate.
629      * @param properties The properties to substitute.
630      * @return The evaluated expression.
631      */
632     public static String evaluate( String expr, Map<String, String> properties )
633     {
634         if ( expr == null )
635         {
636             return null;
637         }
638 
639         String expression = stripTokens( expr );
640         if ( expression.equals( expr ) )
641         {
642             int index = expr.indexOf( "${" );
643             if ( index >= 0 )
644             {
645                 int lastIndex = expr.indexOf( "}", index );
646                 if ( lastIndex >= 0 )
647                 {
648                     String retVal = expr.substring( 0, index );
649 
650                     if ( index > 0 && expr.charAt( index - 1 ) == '$' )
651                     {
652                         retVal += expr.substring( index + 1, lastIndex + 1 );
653                     }
654                     else
655                     {
656                         retVal += evaluate( expr.substring( index, lastIndex + 1 ), properties );
657                     }
658 
659                     retVal += evaluate( expr.substring( lastIndex + 1 ), properties );
660                     return retVal;
661                 }
662             }
663 
664             // Was not an expression
665             if ( expression.contains( "$$" ) )
666             {
667                 return expression.replaceAll( "\\$\\$", "\\$" );
668             }
669             else
670             {
671                 return expression;
672             }
673         }
674 
675         String value = properties.get( expression );
676 
677         if ( value != null )
678         {
679             int exprStartDelimiter = value.indexOf( "${" );
680 
681             if ( exprStartDelimiter >= 0 )
682             {
683                 if ( exprStartDelimiter > 0 )
684                 {
685                     value = value.substring( 0, exprStartDelimiter )
686                         + evaluate( value.substring( exprStartDelimiter ), properties );
687                 }
688                 else
689                 {
690                     value = evaluate( value.substring( exprStartDelimiter ), properties );
691                 }
692             }
693         }
694         else
695         {
696             // TODO find a way to log that and not use this System.out!!
697             // this class could be a component with logger injected !!
698             System.out.println( "expression: " + expression + " no value " );
699         }
700         return value == null ? expr : value;
701     }

703     /**
704      * Strips the expression token markers from the start and end of the string.
705      *
706      * @param expr the string (perhaps with token markers)
707      * @return the string (definately without token markers)
708      */
709     private static String stripTokens( String expr )
710     {
711         if ( expr.startsWith( "${" ) && expr.indexOf( "}" ) == expr.length() - 1 )
712         {
713             expr = expr.substring( 2, expr.length() - 1 );
714         }
715         return expr;
716     }

718     /**
719      * Checks if two versions or ranges have an overlap.
720      *
721      * @param leftVersionOrRange the 1st version number or range to test
722      * @param rightVersionOrRange the 2nd version number or range to test
723      * @return true if both versions have an overlap
724      * @throws InvalidVersionSpecificationException if the versions can't be parsed to a range
725      */
726     public static boolean isVersionOverlap( String leftVersionOrRange, String rightVersionOrRange )
727         throws InvalidVersionSpecificationException
728     {
729         VersionRange pomVersionRange = createVersionRange( leftVersionOrRange );
730         if ( !pomVersionRange.hasRestrictions() )
731         {
732             return true;
733         }
734 
735         VersionRange oldVersionRange = createVersionRange( rightVersionOrRange );
736         if ( !oldVersionRange.hasRestrictions() )
737         {
738             return true;
739         }
740 
741         VersionRange result = oldVersionRange.restrict( pomVersionRange );
742         return result.hasRestrictions();
743     }
744 
745     private static VersionRange createVersionRange( String versionOrRange )
746         throws InvalidVersionSpecificationException
747     {
748         VersionRange versionRange = VersionRange.createFromVersionSpec( versionOrRange );
749         if ( versionRange.getRecommendedVersion() != null )
750         {
751             versionRange = VersionRange.createFromVersionSpec( "[" + versionOrRange + "]" );
752         }
753         return versionRange;

756     /**
757      * Searches the pom re-defining the specified plugin to the specified version.
758      *
759      * @param pom The pom to modify.
760      * @param groupId The groupId of the dependency.
761      * @param artifactId The artifactId of the dependency.
762      * @param oldVersion The old version of the dependency.
763      * @param newVersion The new version of the dependency.
764      * @return <code>true</code> if a replacement was made.
765      * @throws XMLStreamException if somethinh went wrong.
766      */
767     public static boolean setPluginVersion( final ModifiedPomXMLEventReader pom, final String groupId,
768                                             final String artifactId, final String oldVersion, final String newVersion )
769                                                 throws XMLStreamException
770     {
771         Stack<String> stack = new Stack<String>();
772         String path = "";
773         final Pattern matchScopeRegex;
774         final Pattern matchTargetRegex;
775         boolean inMatchScope = false;
776         boolean madeReplacement = false;
777         boolean haveGroupId = false;
778         boolean needGroupId = groupId != null && !APACHE_MAVEN_PLUGINS_GROUPID.equals( groupId );
779         boolean haveArtifactId = false;
780         boolean haveOldVersion = false;
781 
782         matchScopeRegex = Pattern.compile( "/project" + "(/profiles/profile)?"
783             + "((/build(/pluginManagement)?)|(/reporting))/plugins/plugin" );
784 
785         matchTargetRegex = Pattern.compile( "/project" + "(/profiles/profile)?"
786             + "((/build(/pluginManagement)?)|(/reporting))/plugins/plugin" + "((/groupId)|(/artifactId)|(/version))" );
787 
788         pom.rewind();
789 
790         while ( pom.hasNext() )
791         {
792             XMLEvent event = pom.nextEvent();
793             if ( event.isStartElement() )
794             {
795                 stack.push( path );
796                 final String elementName = event.asStartElement().getName().getLocalPart();
797                 path = path + "/" + elementName;
798 
799                 if ( matchScopeRegex.matcher( path ).matches() )
800                 {
801                     // we're in a new match scope
802                     // reset any previous partial matches
803                     inMatchScope = true;
804                     pom.clearMark( 0 );
805                     pom.clearMark( 1 );
806 
807                     haveGroupId = false;
808                     haveArtifactId = false;
809                     haveOldVersion = false;
810                 }
811                 else if ( inMatchScope && matchTargetRegex.matcher( path ).matches() )
812                 {
813                     if ( "groupId".equals( elementName ) )
814                     {
815                         haveGroupId = groupId.equals( pom.getElementText().trim() );
816                         path = stack.pop();
817                     }
818                     else if ( "artifactId".equals( elementName ) )
819                     {
820                         haveArtifactId = artifactId.equals( pom.getElementText().trim() );
821                         path = stack.pop();
822                     }
823                     else if ( "version".equals( elementName ) )
824                     {
825                         pom.mark( 0 );
826                     }
827                 }
828             }
829             if ( event.isEndElement() )
830             {
831                 if ( matchTargetRegex.matcher( path ).matches()
832                     && "version".equals( event.asEndElement().getName().getLocalPart() ) )
833                 {
834                     pom.mark( 1 );
835 
836                     try
837                     {
838                         haveOldVersion = isVersionOverlap( oldVersion, pom.getBetween( 0, 1 ).trim() );
839                     }
840                     catch ( InvalidVersionSpecificationException e )
841                     {
842                         // fall back to string comparison
843                         haveOldVersion = oldVersion.equals( pom.getBetween( 0, 1 ).trim() );
844                     }
845                 }
846                 else if ( matchScopeRegex.matcher( path ).matches() )
847                 {
848                     if ( inMatchScope && pom.hasMark( 0 ) && pom.hasMark( 1 ) && ( haveGroupId || !needGroupId )
849                         && haveArtifactId && haveOldVersion )
850                     {
851                         pom.replaceBetween( 0, 1, newVersion );
852                         madeReplacement = true;
853                         pom.clearMark( 0 );
854                         pom.clearMark( 1 );
855                         haveArtifactId = false;
856                         haveGroupId = false;
857                         haveOldVersion = false;
858                     }
859                     inMatchScope = false;
860                 }
861                 path = stack.pop();
862             }
863         }
864         return madeReplacement;
865     }

867     /**
868      * Examines the project to find any properties which are associated with versions of artifacts in the project.
869      *
870      * @param helper Our versions helper.
871      * @param project The project to examine.
872      * @return An array of properties that are associated within the project.
873      * @throws ExpressionEvaluationException if an expression cannot be evaluated.
874      * @throws IOException if the project's pom file cannot be parsed.
875      * @since 1.0-alpha-3
876      */
877     public static PropertyVersionsBuilder[] getPropertyVersionsBuilders( VersionsHelper helper, MavenProject project )
878         throws ExpressionEvaluationException, IOException
879     {
880         ExpressionEvaluator expressionEvaluator = helper.getExpressionEvaluator( project );
881         Model model = getRawModel( project );
882         Map<String, PropertyVersionsBuilder> result = new TreeMap<String, PropertyVersionsBuilder>();
883 
884         Set<String> activeProfiles = new TreeSet<String>();
885         for ( Profile profile : (List<Profile>) project.getActiveProfiles() )
886         {
887             activeProfiles.add( profile.getId() );
888         }
889 
890         // add any properties from profiles first (as they override properties from the project
891         for ( Profile profile : model.getProfiles() )
892         {
893             if ( !activeProfiles.contains( profile.getId() ) )
894             {
895                 continue;
896             }
897             addProperties( helper, result, profile.getId(), profile.getProperties() );
898             if ( profile.getDependencyManagement() != null )
899             {
900                 addDependencyAssocations( helper, expressionEvaluator, result,
901                                           profile.getDependencyManagement().getDependencies(), false );
902             }
903             addDependencyAssocations( helper, expressionEvaluator, result, profile.getDependencies(), false );
904             if ( profile.getBuild() != null )
905             {
906                 if ( profile.getBuild().getPluginManagement() != null )
907                 {
908                     addPluginAssociations( helper, expressionEvaluator, result,
909                                            profile.getBuild().getPluginManagement().getPlugins() );
910                 }
911                 addPluginAssociations( helper, expressionEvaluator, result, profile.getBuild().getPlugins() );
912             }
913             if ( profile.getReporting() != null )
914             {
915                 addReportPluginAssociations( helper, expressionEvaluator, result, profile.getReporting().getPlugins() );
916             }
917         }
918 
919         // second, we add all the properties in the pom
920         addProperties( helper, result, null, model.getProperties() );
921         if ( model.getDependencyManagement() != null )
922         {
923             addDependencyAssocations( helper, expressionEvaluator, result,
924                                       model.getDependencyManagement().getDependencies(), false );
925         }
926         addDependencyAssocations( helper, expressionEvaluator, result, model.getDependencies(), false );
927         if ( model.getBuild() != null )
928         {
929             if ( model.getBuild().getPluginManagement() != null )
930             {
931                 addPluginAssociations( helper, expressionEvaluator, result,
932                                        model.getBuild().getPluginManagement().getPlugins() );
933             }
934             addPluginAssociations( helper, expressionEvaluator, result, model.getBuild().getPlugins() );
935         }
936         if ( model.getReporting() != null )
937         {
938             addReportPluginAssociations( helper, expressionEvaluator, result, model.getReporting().getPlugins() );
939         }
940 
941         // third, we add any associations from the active profiles
942         for ( Profile profile : model.getProfiles() )
943         {
944             if ( !activeProfiles.contains( profile.getId() ) )
945             {
946                 continue;
947             }
948             if ( profile.getDependencyManagement() != null )
949             {
950                 addDependencyAssocations( helper, expressionEvaluator, result,
951                                           profile.getDependencyManagement().getDependencies(), false );
952             }
953             addDependencyAssocations( helper, expressionEvaluator, result, profile.getDependencies(), false );
954             if ( profile.getBuild() != null )
955             {
956                 if ( profile.getBuild().getPluginManagement() != null )
957                 {
958                     addPluginAssociations( helper, expressionEvaluator, result,
959                                            profile.getBuild().getPluginManagement().getPlugins() );
960                 }
961                 addPluginAssociations( helper, expressionEvaluator, result, profile.getBuild().getPlugins() );
962             }
963             if ( profile.getReporting() != null )
964             {
965                 addReportPluginAssociations( helper, expressionEvaluator, result, profile.getReporting().getPlugins() );
966             }
967         }
968 
969         // finally, remove any properties without associations
970         purgeProperties( result );
971 
972         return result.values().toArray( new PropertyVersionsBuilder[result.values().size()] );
973     }

975     /**
976      * Takes a list of {@link org.apache.maven.model.Plugin} instances and adds associations to properties used to
977      * define versions of the plugin artifact or any of the plugin dependencies specified in the pom.
978      *
979      * @param helper Our helper.
980      * @param expressionEvaluator Our expression evaluator.
981      * @param result The map of {@link org.codehaus.mojo.versions.api.PropertyVersionsBuilder} keyed by property name.
982      * @param plugins The list of {@link org.apache.maven.model.Plugin}.
983      * @throws ExpressionEvaluationException if an expression cannot be evaluated.
984      */
985     private static void addPluginAssociations( VersionsHelper helper, ExpressionEvaluator expressionEvaluator,
986                                                Map<String, PropertyVersionsBuilder> result, List<Plugin> plugins )
987                                                    throws ExpressionEvaluationException
988     {
989         if ( plugins == null )
990         {
991             return;
992         }
993         for ( Plugin plugin : plugins )
994         {
995             String version = plugin.getVersion();
996             if ( version != null && version.contains( "${" ) && version.indexOf( '}' ) != -1 )
997             {
998                 version = StringUtils.deleteWhitespace( version );
999                 for ( PropertyVersionsBuilder property : result.values() )
1000                 {
1001                     // any of these could be defined by a property
1002                     final String propertyRef = "${" + property.getName() + "}";
1003                     if ( version.contains( propertyRef ) )
1004                     {
1005                         String groupId = plugin.getGroupId();
1006                         if ( groupId == null || groupId.trim().length() == 0 )
1007                         {
1008                             // group Id has a special default
1009                             groupId = APACHE_MAVEN_PLUGINS_GROUPID;
1010                         }
1011                         else
1012                         {
1013                             groupId = (String) expressionEvaluator.evaluate( groupId );
1014                         }
1015                         String artifactId = plugin.getArtifactId();
1016                         if ( artifactId == null || artifactId.trim().length() == 0 )
1017                         {
1018                             // malformed pom
1019                             continue;
1020                         }
1021                         else
1022                         {
1023                             artifactId = (String) expressionEvaluator.evaluate( artifactId );
1024                         }
1025                         // might as well capture the current value
1026                         VersionRange versionRange =
1027                             VersionRange.createFromVersion( (String) expressionEvaluator.evaluate( plugin.getVersion() ) );
1028                         property.addAssociation( helper.createPluginArtifact( groupId, artifactId, versionRange ),
1029                                                  true );
1030                         if ( !propertyRef.equals( version ) )
1031                         {
1032                             addBounds( property, version, propertyRef, versionRange.toString() );
1033                         }
1034                     }
1035                 }
1036             }
1037             addDependencyAssocations( helper, expressionEvaluator, result, plugin.getDependencies(), true );
1038         }
1039     }

1157     private static void addBounds( PropertyVersionsBuilder builder, String rawVersionRange, String propertyRef,
1158                                    String evaluatedVersionRange )
1159     {
1160         Pattern lowerBound = Pattern.compile( "([(\\[])([^,]*)," + RegexUtils.quote( propertyRef ) + "([)\\]])" );
1161         Pattern upperBound = Pattern.compile( "([(\\[])" + RegexUtils.quote( propertyRef ) + ",([^,]*)([)\\]])" );
1162         Matcher m = lowerBound.matcher( rawVersionRange );
1163         if ( m.find() )
1164         {
1165             boolean includeLower = "[".equals( m.group( 1 ) );
1166             String lowerLimit = m.group( 2 );
1167             if ( StringUtils.isNotEmpty( lowerLimit ) )
1168             {
1169                 builder.addLowerBound( lowerLimit, includeLower );
1170             }
1171         }
1172         m = upperBound.matcher( rawVersionRange );
1173         if ( m.find() )
1174         {
1175             boolean includeUpper = "[".equals( m.group( 3 ) );
1176             String upperLimit = m.group( 2 );
1177             if ( StringUtils.isNotEmpty( upperLimit ) )
1178             {
1179                 builder.addUpperBound( upperLimit, includeUpper );
1180             }
1181         }
1182     }

1434      * @param logger The logger for logging.
1435      * @return A map of raw models keyed by path relative to the project's basedir.
1436      * @throws IOException if things go wrong.
1437      */
1438     public static Map<String, Model> getReactorModels( MavenProject project, Log logger )
1439         throws IOException
1440     {
1441         Map<String, Model> result = new LinkedHashMap<String, Model>();
1442         final Model model = getRawModel( project );
1443         final String path = "";

1445         result.putAll( getReactorModels( path, model, project, logger ) );

1449     /**
1450      * Builds a sub-map of raw models keyed by module path.
1451      *
1452      * @param path The relative path to base the sub-map on.
1453      * @param model The model at the relative path.
1454      * @param project The project to build from.
1455      * @param logger The logger for logging.
1456      * @return A map of raw models keyed by path relative to the project's basedir.
1457      * @throws IOException if things go wrong.
1458      */
1459     private static Map<String, Model> getReactorModels( String path, Model model, MavenProject project, Log logger )
1460         throws IOException
1461     {
1462         if ( path.length() > 0 && !path.endsWith( "/" ) )
1463         {
1464             path += '/';
1465         }
1466         Map<String, Model> result = new LinkedHashMap<String, Model>();
1467         Map<String, Model> childResults = new LinkedHashMap<String, Model>();
1468 
1469         File baseDir = path.length() > 0 ? new File( project.getBasedir(), path ) : project.getBasedir();
1470 
1471         Set<String> childModules = getAllChildModules( model, logger );
1472 
1473         removeMissingChildModules( logger, baseDir, childModules );
1474 
1475         for ( String moduleName : childModules )
1476         {
1477             String modulePath = path + moduleName;
1478 
1479             File moduleDir = new File( baseDir, moduleName );
1480 
1481             File moduleProjectFile;
1482 
1483             if ( moduleDir.isDirectory() )
1484             {
1485                 moduleProjectFile = new File( moduleDir, "pom.xml" );
1486             }
1487             else
1488             {
1489                 // i don't think this should ever happen... but just in case
1490                 // the module references the file-name
1491                 moduleProjectFile = moduleDir;
1492             }
1493 
1494             try
1495             {
1496                 // the aim of this goal is to fix problems when the project cannot be parsed by Maven
1497                 // so we have to work with the raw model and not the interpolated parsed model from maven
1498                 Model moduleModel = getRawModel( moduleProjectFile );
1499                 result.put( modulePath, moduleModel );
1500                 childResults.putAll( getReactorModels( modulePath, moduleModel, project, logger ) );
1501             }
1502             catch ( IOException e )
1503             {
1504                 logger.debug( "Could not parse " + moduleProjectFile.getPath(), e );
1505             }
1506         }
1507         result.putAll( childResults ); // more efficient update order if all children are added after siblings
1508         return result;
1509     }

1511     /**
1512      * Returns all the models that have a specified groupId and artifactId as parent.
1513      *
1514      * @param reactor The map of models keyed by path.
1515      * @param groupId The groupId of the parent.
1516      * @param artifactId The artifactId of the parent.
1517      * @return a map of models that have a specified groupId and artifactId as parent keyed by path.
1518      */
1519     public static Map<String, Model> getChildModels( Map<String, Model> reactor, String groupId, String artifactId )
1520     {
1521         final Map<String, Model> result = new LinkedHashMap<String, Model>();
1522         for ( Map.Entry<String, Model> entry : reactor.entrySet() )
1523         {
1524             final String path = entry.getKey();
1525             final Model model = entry.getValue();
1526             final Parent parent = model.getParent();
1527             if ( parent != null && groupId.equals( parent.getGroupId() )
1528                 && artifactId.equals( parent.getArtifactId() ) )
1529             {
1530                 result.put( path, model );
1531             }
1532         }
1533         return result;
1534     }

1537      * Returns the model that has the specified groupId and artifactId or <code>null</code> if no such model exists.
1538      *
1539      * @param reactor The map of models keyed by path.
1540      * @param groupId The groupId to match.
1541      * @param artifactId The artifactId to match.
1542      * @return The model or <code>null</code> if the model was not in the reactor.

1545     {

1547         return entry == null ? null : entry.getValue();

1550     /**
1551      * Returns the model that has the specified groupId and artifactId or <code>null</code> if no such model exists.
1552      *
1553      * @param reactor The map of models keyed by path.
1554      * @param groupId The groupId to match.
1555      * @param artifactId The artifactId to match.
1556      * @return The model entry or <code>null</code> if the model was not in the reactor.
1557      */
1558     public static Map.Entry<String, Model> getModelEntry( Map<String, Model> reactor, String groupId,
1559                                                           String artifactId )
1560     {
1561         for ( Map.Entry<String, Model> entry : reactor.entrySet() )
1562         {
1563             Model model = entry.getValue();
1564             if ( groupId.equals( getGroupId( model ) ) && artifactId.equals( getArtifactId( model ) ) )
1565             {
1566                 return entry;
1567             }
1568         }
1569         return null;
1570     }

1572     /**
1573      * Returns a count of how many parents a model has in the reactor.
1574      *
1575      * @param reactor The map of models keyed by path.
1576      * @param model The model.
1577      * @return The number of parents of this model in the reactor.
1578      */
1579     public static int getReactorParentCount( Map<String, Model> reactor, Model model )
1580     {
1581         if ( model.getParent() == null )
1582         {
1583             return 0;
1584         }
1585         else
1586         {
1587             Model parentModel = getModel( reactor, model.getParent().getGroupId(), model.getParent().getArtifactId() );
1588             if ( parentModel != null )
1589             {
1590                 return getReactorParentCount( reactor, parentModel ) + 1;
1591             }
1592             return 0;
1593         }
1594     }
