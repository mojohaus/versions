161     /**
162      * Returns the pluginManagement section of the super-pom.
163      *
164      * @return Returns the pluginManagement section of the super-pom.
165      * @throws MojoExecutionException when things go wrong.
166      */
167     private Map<String, String> getSuperPomPluginManagement()
168         throws MojoExecutionException
169     {
170         if ( new DefaultArtifactVersion( "3.0" ).compareTo( runtimeInformation.getApplicationVersion() ) <= 0 )
171         {
172             getLog().debug( "Using Maven 3.x strategy to determine superpom defined plugins" );
173             try
174             {
175                 Method getPluginsBoundByDefaultToAllLifecycles =
176                     LifecycleExecutor.class.getMethod( "getPluginsBoundByDefaultToAllLifecycles",
177                                                        new Class[] { String.class } );
178                 Set<Plugin> plugins =
179                     (Set<Plugin>) getPluginsBoundByDefaultToAllLifecycles.invoke( lifecycleExecutor, new Object[] {
180                         getProject().getPackaging() } );
181                 // we need to provide a copy with the version blanked out so that inferring from super-pom
182                 // works as for 2.x as 3.x fills in the version on us!
183                 Map<String, String> result = new LinkedHashMap<>( plugins.size() );
184                 for ( Plugin plugin : plugins )
185                 {
186                     result.put( plugin.getKey(), plugin.getVersion() );
187                 }
188                 URL superPom = getClass().getClassLoader().getResource( "org/apache/maven/model/pom-4.0.0.xml" );
189                 if ( superPom != null )
190                 {
191                     try
192                     {
193                         try( Reader reader = ReaderFactory.newXmlReader( superPom ) )
194                         {
195                             StringBuilder buf = new StringBuilder( IOUtil.toString( reader ) );
196                             ModifiedPomXMLEventReader pom = newModifiedPomXER( buf, superPom.toString() );
197 
198                             Pattern pathRegex = Pattern.compile( "/project(/profiles/profile)?"
199                                 + "((/build(/pluginManagement)?)|(/reporting))" + "/plugins/plugin" );
200                             Stack<StackState> pathStack = new Stack<>();
201                             StackState curState = null;
202                             while ( pom.hasNext() )
203                             {
204                                 XMLEvent event = pom.nextEvent();
205                                 if ( event.isStartDocument() )
206                                 {
207                                     curState = new StackState( "" );
208                                     pathStack.clear();
209                                 }
210                                 else if ( event.isStartElement() )
211                                 {
212                                     String elementName = event.asStartElement().getName().getLocalPart();
213                                     if ( curState != null && pathRegex.matcher( curState.path ).matches() )
214                                     {
215                                         if ( "groupId".equals( elementName ) )
216                                         {
217                                             curState.groupId = pom.getElementText().trim();
218                                             continue;
219                                         }
220                                         else if ( "artifactId".equals( elementName ) )
221                                         {
222                                             curState.artifactId = pom.getElementText().trim();
223                                             continue;
224 
225                                         }
226                                         else if ( "version".equals( elementName ) )
227                                         {
228                                             curState.version = pom.getElementText().trim();
229                                             continue;
230                                         }
231                                     }
232 
233                                     pathStack.push( curState );
234                                     curState = new StackState( curState.path + "/" + elementName );
235                                 }
236                                 else if ( event.isEndElement() )
237                                 {
238                                     if ( curState != null && pathRegex.matcher( curState.path ).matches() )
239                                     {
240                                         if ( curState.artifactId != null )
241                                         {
242                                             Plugin plugin = new Plugin();
243                                             plugin.setArtifactId( curState.artifactId );
244                                             plugin.setGroupId( curState.groupId == null
245                                                             ? PomHelper.APACHE_MAVEN_PLUGINS_GROUPID
246                                                             : curState.groupId );
247                                             plugin.setVersion( curState.version );
248                                             if ( !result.containsKey( plugin.getKey() ) )
249                                             {
250                                                 result.put( plugin.getKey(), plugin.getVersion() );
251                                             }
252                                         }
253                                     }
254                                     curState = pathStack.pop();
255                                 }
256                             }
257                         }
258                     }
259                     catch ( IOException | XMLStreamException e )
260                     {
261                         // ignore
262                     }
263                 }
264 
265                 return result;
266             }
267             catch ( NoSuchMethodException | InvocationTargetException | IllegalAccessException e1 )
268             {
269                 // no much we can do here
270             }
271         }
272         getLog().debug( "Using Maven 2.x strategy to determine superpom defined plugins" );
273         Map<String, String> superPomPluginManagement;
274         try
275         {
276             MavenProject superProject =
277                 projectBuilder.buildStandaloneSuperProject( new DefaultProjectBuilderConfiguration() );
278             superPomPluginManagement = new HashMap<>( getPluginManagement( superProject.getOriginalModel() ) );
279         }
280         catch ( ProjectBuildingException e )
281         {
282             throw new MojoExecutionException( "Could not determine the super pom.xml", e );
283         }
284         return superPomPluginManagement;
285     }

287     /**
288      * Gets the plugin management plugins of a specific project.
289      *
290      * @param model the model to get the plugin management plugins from.
291      * @return The map of effective plugin versions keyed by coordinates.
292      * @since 1.0-alpha-1
293      */
294     private Map<String, String> getPluginManagement( Model model )
295     {
296         // we want only those parts of pluginManagement that are defined in this project
297         Map<String, String> pluginManagement = new HashMap<>();
298         try
299         {
300             for ( Plugin plugin : model.getBuild().getPluginManagement().getPlugins() )
301             {
302                 String coord = plugin.getKey();
303                 String version = plugin.getVersion();
304                 if ( version != null )
305                 {
306                     pluginManagement.put( coord, version );
307                 }
308             }
309         }
310         catch ( NullPointerException e )
311         {
312             // guess there are no plugins here
313         }
314         try
315         {
316             for ( Profile profile : model.getProfiles() )
317             {
318                 try
319                 {
320                     for ( Plugin plugin : profile.getBuild().getPluginManagement().getPlugins() )
321                     {
322                         String coord = plugin.getKey();
323                         String version = plugin.getVersion();
324                         if ( version != null )
325                         {
326                             pluginManagement.put( coord, version );
327                         }
328                     }
329                 }
330                 catch ( NullPointerException e )
331                 {
332                     // guess there are no plugins here
333                 }
334             }
335         }
336         catch ( NullPointerException e )
337         {
338             // guess there are no profiles here
339         }
340 
341         return pluginManagement;
342     }

348     /**
349      * @throws MojoExecutionException when things go wrong
350      * @throws MojoFailureException when things go wrong in a very bad way
351      * @see AbstractVersionsUpdaterMojo#execute()
352      * @since 1.0-alpha-1
353      */
354     public void execute()
355         throws MojoExecutionException, MojoFailureException
356     {
357         logInit();
358         Set<String> pluginsWithVersionsSpecified;
359         try
360         {
361             pluginsWithVersionsSpecified = findPluginsWithVersionsSpecified( getProject() );
362         }
363         catch ( XMLStreamException | IOException e )
364         {
365             throw new MojoExecutionException( e.getMessage(), e );
366         }
367 
368         Map<String, String> superPomPluginManagement = getSuperPomPluginManagement();
369         getLog().debug( "superPom plugins = " + superPomPluginManagement );
370 
371         List<MavenProject> parents = getParentProjects( getProject() );
372         Map<String, String> parentPlugins = getParentsPlugins( parents );
373         // TODO remove, not used any more (found while extracting getParentsPlugins method and
374         //      renaming parentPluginManagement to parentPlugins)
375         // NOTICE: getProjectPlugins() takes profiles while getParentPlugins does not
376         //         there is probably a little inconsistency (if plugins configured in profiles of parents)
377         Map<String, String> parentBuildPlugins = new HashMap<>();
378         Map<String, String> parentReportPlugins = new HashMap<>();
379 
380         Set<Plugin> plugins = getProjectPlugins( superPomPluginManagement, parentPlugins, parentBuildPlugins,
381                                                  parentReportPlugins, pluginsWithVersionsSpecified );
382 
383         List<String> pluginUpdates = new ArrayList<>();
384         List<String> pluginLockdowns = new ArrayList<>();
385         ArtifactVersion curMavenVersion = runtimeInformation.getApplicationVersion();
386         ArtifactVersion specMavenVersion = MinimalMavenBuildVersionFinder.find( getProject(), "2.0", getLog() );
387         ArtifactVersion minMavenVersion = null;
388         boolean superPomDrivingMinVersion = false;
389         // if Maven prerequisite upgraded to a version, Map<plugin compact key, latest compatible plugin vesion>
390         Map<ArtifactVersion, Map<String, String>> mavenUpgrades = new TreeMap<>( new MavenVersionComparator() );
391 
392         for ( Plugin plugin : plugins )
393         {
394             String groupId = plugin.getGroupId();
395             String artifactId = plugin.getArtifactId();
396             String version = plugin.getVersion();
397             String coords = ArtifactUtils.versionlessKey( groupId, artifactId );
398 
399             if ( version == null )
400             {
401                 version = parentPlugins.get( coords );
402             }
403             getLog().debug( "Checking " + coords + " for updates newer than " + version );
404             String effectiveVersion = version;
405 
406             Artifact artifactRange;
407             try
408             {
409                 boolean unspecified = ( version == null );
410                 VersionRange versionRange = unspecified ? VersionRange.createFromVersionSpec( "[0,)" )
411                                 : VersionRange.createFromVersionSpec( version );
412                 artifactRange = artifactFactory.createPluginArtifact( groupId, artifactId, versionRange );
413             }
414             catch ( InvalidVersionSpecificationException e )
415             {
416                 throw new MojoExecutionException( "Invalid version range specification: " + version, e );
417             }
418 
419             ArtifactVersion artifactVersion = null;
420             try
421             {
422                 // now we want to find the newest versions and check their Maven version prerequisite
423                 ArtifactVersions artifactVersions = getHelper().lookupArtifactVersions( artifactRange, true );
424                 ArtifactVersion[] newerVersions = artifactVersions.getVersions( this.allowSnapshots );
425                 ArtifactVersion minRequires = null;
426                 for ( int j = newerVersions.length - 1; j >= 0; j-- )
427                 {
428                     Artifact probe =
429                         artifactFactory.createDependencyArtifact( groupId, artifactId,
430                                                                   VersionRange.createFromVersion( newerVersions[j].toString() ),
431                                                                   "pom", null, "runtime" );
432                     try
433                     {
434                         getHelper().resolveArtifact( probe, true );
435                         MavenProject pluginMavenProject =
436                             projectBuilder.buildFromRepository( probe, remotePluginRepositories, localRepository );
437                         ArtifactVersion pluginRequires = getPrerequisitesMavenVersion( pluginMavenProject );
438                         if ( artifactVersion == null && compare( specMavenVersion, pluginRequires ) >= 0 )
439                         {
440                             // ok, newer version compatible with current specMavenVersion
441                             artifactVersion = newerVersions[j];
442                         }
443                         if ( effectiveVersion == null && compare( curMavenVersion, pluginRequires ) >= 0 )
444                         {
445                             // version was unspecified, current version of maven thinks it should use this
446                             effectiveVersion = newerVersions[j].toString();
447                         }
448                         if ( artifactVersion != null && effectiveVersion != null )
449                         {
450                             // no need to look at any older versions: latest compatible found
451                             break;
452                         }
453                         // newer version not compatible with current specMavenVersion: track opportunity if Maven spec
454                         // upgrade
455                         if ( minRequires == null || compare( minRequires, pluginRequires ) > 0 )
456                         {
457                             Map<String, String> upgradePlugins =
458                                 mavenUpgrades.computeIfAbsent( pluginRequires, k -> new LinkedHashMap<>() );
459 
460                             String upgradePluginKey = compactKey( groupId, artifactId );
461                             if ( !upgradePlugins.containsKey( upgradePluginKey ) )
462                             {
463                                 String newer = newerVersions[j].toString();
464                                 if ( newer.equals( effectiveVersion ) )
465                                 {
466                                     // plugin version configured that require a Maven version higher than spec
467                                     upgradePlugins.put( upgradePluginKey,
468                                                         pad( upgradePluginKey, INFO_PAD_SIZE, newer ) );
469                                 }
470                                 else
471                                 {
472                                     // plugin that can be upgraded
473                                     upgradePlugins.put( upgradePluginKey, pad( upgradePluginKey, INFO_PAD_SIZE,
474                                                                                effectiveVersion, " -> ", newer ) );
475                                 }
476                             }
477                             minRequires = pluginRequires;
478                         }
479                     }
480                     catch ( ArtifactResolutionException | ArtifactNotFoundException | ProjectBuildingException e )
481                     {
482                         // ignore bad version
483                     }
484                 }
485                 if ( effectiveVersion != null )
486                 {
487                     VersionRange currentVersionRange = VersionRange.createFromVersion( effectiveVersion );
488                     Artifact probe = artifactFactory.createDependencyArtifact( groupId, artifactId, currentVersionRange,
489                                                                                "pom", null, "runtime" );
490                     try
491                     {
492                         getHelper().resolveArtifact( probe, true );
493                         MavenProject mavenProject =
494                             projectBuilder.buildFromRepository( probe, remotePluginRepositories, localRepository );
495                         ArtifactVersion requires = getPrerequisitesMavenVersion( mavenProject );
496                         if ( minMavenVersion == null || compare( minMavenVersion, requires ) < 0 )
497                         {
498                             minMavenVersion = requires;
499                         }
500                     }
501                     catch ( ArtifactResolutionException | ArtifactNotFoundException | ProjectBuildingException e )
502                     {
503                         // ignore bad version
504                     }
505                 }
506             }
507             catch ( ArtifactMetadataRetrievalException e )
508             {
509                 throw new MojoExecutionException( e.getMessage(), e );
510             }
511 
512             String newVersion;
513 
514             if ( version == null && pluginsWithVersionsSpecified.contains( coords ) )
515             {
516                 // Hack ALERT!
517                 //
518                 // All this should be re-written in a less "pom is xml" way... but it'll
519                 // work for now :-(
520                 //
521                 // we have removed the version information, as it was the same as from
522                 // the super-pom... but it actually was specified.
523                 version = artifactVersion != null ? artifactVersion.toString() : null;
524             }
525 
526             getLog().debug( "[" + coords + "].version=" + version );
527             getLog().debug( "[" + coords + "].artifactVersion=" + artifactVersion );
528             getLog().debug( "[" + coords + "].effectiveVersion=" + effectiveVersion );
529             getLog().debug( "[" + coords + "].specified=" + pluginsWithVersionsSpecified.contains( coords ) );
530             if ( version == null || !pluginsWithVersionsSpecified.contains( coords ) )
531             {
532                 version = superPomPluginManagement.get( coords );
533                 getLog().debug( "[" + coords + "].superPom.version=" + version );
534 
535                 newVersion = artifactVersion != null ? artifactVersion.toString()
536                                 : ( version != null ? version
537                                                 : ( effectiveVersion != null ? effectiveVersion : "(unknown)" ) );
538                 if ( version != null )
539                 {
540                     superPomDrivingMinVersion = true;
541                 }
542 
543                 pluginLockdowns.add( pad( compactKey( groupId, artifactId ), WARN_PAD_SIZE,
544                                     superPomDrivingMinVersion ? FROM_SUPER_POM : "", newVersion ) );
545             }
546             else if ( artifactVersion != null )
547             {
548                 newVersion = artifactVersion.toString();
549             }
550             else
551             {
552                 newVersion = null;
553             }
554             if ( version != null && artifactVersion != null && newVersion != null && effectiveVersion != null
555                 && new DefaultArtifactVersion( effectiveVersion ).compareTo( new DefaultArtifactVersion( newVersion ) ) < 0 )
556             {
557                 pluginUpdates.add( pad( compactKey( groupId, artifactId ), INFO_PAD_SIZE,
558                                   effectiveVersion, " -> ", newVersion ) );
559             }
560         }
561 
562         // info on each plugin gathered: now it's time to display the result!
563         //
564         logLine( false, "" );
565 
566         // updates keeping currently defined Maven version minimum
567         if ( pluginUpdates.isEmpty() )
568         {
569             logLine( false, "All plugins with a version specified are using the latest versions." );
570         }
571         else
572         {
573             logLine( false, "The following plugin updates are available:" );
574             for ( String update : new TreeSet<>(pluginUpdates) )
575             {
576                 logLine( false, update );
577             }
578         }
579         logLine( false, "" );
580 
581         // has every plugin a specified version?
582         if ( pluginLockdowns.isEmpty() )
583         {
584             logLine( false, "All plugins have a version specified." );
585         }
586         else
587         {
588             getLog().warn( "The following plugins do not have their version specified:" );
589             for ( String lockdown : new TreeSet<>(pluginLockdowns) )
590             {
591                 getLog().warn( lockdown );
592             }
593         }
594         logLine( false, "" );
595 
596         // information on minimum Maven version
597         boolean noMavenMinVersion = MinimalMavenBuildVersionFinder.find( getProject(), null, getLog() ) == null;
598         if ( noMavenMinVersion )
599         {
600             getLog().warn( "Project does not define minimum Maven version required for build, default is: 2.0" );
601         }
602         else
603         {
604             logLine( false, "Project requires minimum Maven version for build of: " + specMavenVersion );
605         }
606         logLine( false, "Plugins require minimum Maven version of: " + minMavenVersion );
607         if ( superPomDrivingMinVersion )
608         {
609             logLine( false, "Note: the super-pom from Maven " + curMavenVersion + " defines some of the plugin" );
610             logLine( false, "      versions and may be influencing the plugins required minimum Maven" );
611             logLine( false, "      version." );
612         }
613         logLine( false, "" );
614 
615         if ( isMavenPluginProject() )
616         {
617             if ( noMavenMinVersion )
618             {
619                 getLog().warn( "Project (which is a Maven plugin) does not define required minimum version of Maven." );
620                 getLog().warn( "Update the pom.xml to contain" );
621                 getLog().warn( "    <prerequisites>" );
622                 getLog().warn( "      <maven><!-- minimum version of Maven that the plugin works with --></maven>" );
623                 getLog().warn( "    </prerequisites>" );
624                 getLog().warn( "To build this plugin you need at least Maven " + minMavenVersion );
625                 getLog().warn( "A Maven Enforcer rule can be used to enforce this if you have not already set one up" );
626                 getLog().warn( "See https://maven.apache.org/enforcer/enforcer-rules/requireMavenVersion.html" );
627             }
628             else if ( minMavenVersion != null && compare( specMavenVersion, minMavenVersion ) < 0 )
629             {
630                 getLog().warn( "Project (which is a Maven plugin) targets Maven " + specMavenVersion + " or newer" );
631                 getLog().warn( "but requires Maven " + minMavenVersion + " or newer to build." );
632                 getLog().warn( "This may or may not be a problem. A Maven Enforcer rule can help " );
633                 getLog().warn( "enforce that the correct version of Maven is used to build this plugin." );
634                 getLog().warn( "See https://maven.apache.org/enforcer/enforcer-rules/requireMavenVersion.html" );
635             }
636             else
637             {
638                 logLine( false, "No plugins require a newer version of Maven than specified by the pom." );
639             }
640         }
641         else
642         {
643             if ( noMavenMinVersion )
644             {
645                 logLine( true, "Project does not define required minimum version of Maven." );
646                 logLine( true, "Update the pom.xml to contain maven-enforcer-plugin to" );
647                 logLine( true, "force the Maven version which is needed to build this project." );
648                 logLine( true, "See https://maven.apache.org/enforcer/enforcer-rules/requireMavenVersion.html" );
649                 logLine( true, "Using the minimum version of Maven: " + minMavenVersion );
650             }
651             else if ( minMavenVersion != null && compare( specMavenVersion, minMavenVersion ) < 0 )
652             {
653                 logLine( true, "Project requires an incorrect minimum version of Maven." );
654                 logLine( true, "Update the pom.xml to contain maven-enforcer-plugin to" );
655                 logLine( true, "force the Maven version which is needed to build this project." );
656                 logLine( true, "See https://maven.apache.org/enforcer/enforcer-rules/requireMavenVersion.html" );
657                 logLine( true, "Using the minimum version of Maven: " + specMavenVersion );
658             }
659             else
660             {
661                 logLine( false, "No plugins require a newer version of Maven than specified by the pom." );
662             }
663         }
664 
665         // updates if minimum Maven version is changed
666         for ( Map.Entry<ArtifactVersion, Map<String, String>> mavenUpgrade : mavenUpgrades.entrySet() )
667         {
668             ArtifactVersion mavenUpgradeVersion = mavenUpgrade.getKey();
669             Map<String, String> upgradePlugins = mavenUpgrade.getValue();
670             if ( upgradePlugins.isEmpty() || compare( specMavenVersion, mavenUpgradeVersion ) >= 0 )
671             {
672                 continue;
673             }
674             logLine( false, "" );
675             logLine( false, "Require Maven " + mavenUpgradeVersion + " to use the following plugin updates:" );
676             for ( Map.Entry<String, String> entry : upgradePlugins.entrySet() )
677             {
678                 logLine( false, entry.getValue() );
679             }
680         }
681         logLine( false, "" );
682     }

685     {
686         StringBuilder buf = new StringBuilder( len );

688         buf.append( start );
689         int padding = len;
690         for ( String end : ends )

692             padding -= end.length();
693         }

695         while ( buf.length() < padding )

701         {
702             buf.append( end );
703         }

711         for ( MavenProject parentProject : parents )
712         {
713             getLog().debug( "Processing parent: " + parentProject.getGroupId() + ":" + parentProject.getArtifactId()
714                 + ":" + parentProject.getVersion() + " -> " + parentProject.getFile() );
715 
716             StringWriter writer = new StringWriter();
717             boolean havePom = false;
718             Model interpolatedModel;
719             try
720             {
721                 Model originalModel = parentProject.getOriginalModel();

728                 }

730                 {
731                     new MavenXpp3Writer().write( writer, originalModel );
732                     writer.close();
733                     havePom = true;
734                 }
735                 catch ( IOException e )

737                     // ignore
738                 }
739                 interpolatedModel =
740                     modelInterpolator.interpolate( originalModel, null,
741                                                    new DefaultProjectBuilderConfiguration().setExecutionProperties( getProject().getProperties() ),
742                                                    false );

746                 throw new MojoExecutionException( e.getMessage(), e );
747             }

749             {

766                 }
767                 catch ( IOException | XMLStreamException e )
768                 {
769                     throw new MojoExecutionException( e.getMessage(), e );
770                 }
771             }
772             else
773             {
774                 parentPlugins.putAll( getPluginManagement( interpolatedModel ) );
775                 parentPlugins.putAll( getBuildPlugins( interpolatedModel, true ) );
776                 parentPlugins.putAll( getReportPlugins( interpolatedModel, true ) );
777             }

841     /**
842      * Returns a set of Strings which correspond to the plugin coordinates where there is a version specified.
843      *
844      * @param pomContents The project to get the plugins with versions specified.
845      * @param path Path that points to the source of the XML
846      * @return a set of Strings which correspond to the plugin coordinates where there is a version specified.
847      */
848     private Set<String> findPluginsWithVersionsSpecified( StringBuilder pomContents, String path )
849         throws IOException, XMLStreamException
850     {
851         Set<String> result = new HashSet<>();
852         ModifiedPomXMLEventReader pom = newModifiedPomXER( pomContents, path );
853 
854         Pattern pathRegex = Pattern.compile( "/project(/profiles/profile)?"
855             + "((/build(/pluginManagement)?)|(/reporting))" + "/plugins/plugin" );
856         Stack<StackState> pathStack = new Stack<>();
857         StackState curState = null;
858         while ( pom.hasNext() )
859         {
860             XMLEvent event = pom.nextEvent();
861             if ( event.isStartDocument() )
862             {
863                 curState = new StackState( "" );
864                 pathStack.clear();
865             }
866             else if ( event.isStartElement() )
867             {
868                 String elementName = event.asStartElement().getName().getLocalPart();
869                 if ( curState != null && pathRegex.matcher( curState.path ).matches() )
870                 {
871                     if ( "groupId".equals( elementName ) )
872                     {
873                         curState.groupId = pom.getElementText().trim();
874                         continue;
875                     }
876                     else if ( "artifactId".equals( elementName ) )
877                     {
878                         curState.artifactId = pom.getElementText().trim();
879                         continue;
880 
881                     }
882                     else if ( "version".equals( elementName ) )
883                     {
884                         curState.version = pom.getElementText().trim();
885                         continue;
886                     }
887                 }
888 
889                 pathStack.push( curState );
890                 curState = new StackState( curState.path + "/" + elementName );
891             }
892             else if ( event.isEndElement() )
893             {
894                 if ( curState != null && pathRegex.matcher( curState.path ).matches() )
895                 {
896                     if ( curState.artifactId != null && curState.version != null )
897                     {
898                         if ( curState.groupId == null )
899                         {
900                             curState.groupId = PomHelper.APACHE_MAVEN_PLUGINS_GROUPID;
901                         }
902                         result.add( curState.groupId + ":" + curState.artifactId );
903                     }
904                 }
905                 curState = pathStack.pop();
906             }
907         }
908 
909         return result;

1430     /**
1431      * Returns the set of all plugins used by the project.
1432      *
1433      * @param superPomPluginManagement the super pom's pluginManagement plugins.
1434      * @param parentPluginManagement the parent pom's pluginManagement plugins.
1435      * @param parentBuildPlugins the parent pom's build plugins.
1436      * @param parentReportPlugins the parent pom's report plugins.
1437      * @param pluginsWithVersionsSpecified the plugin coords that have a version defined in the project.
1438      * @return the set of plugins used by the project.
1439      * @throws org.apache.maven.plugin.MojoExecutionException if things go wrong.
1440      */
1441     private Set<Plugin> getProjectPlugins( Map<String, String> superPomPluginManagement,
1442                                            Map<String, String> parentPluginManagement,
1443                                            Map<String, String> parentBuildPlugins,
1444                                            Map<String, String> parentReportPlugins,
1445                                            Set<String> pluginsWithVersionsSpecified )
1446         throws MojoExecutionException
1447     {
1448         Map<String, Plugin> plugins = new HashMap<>();
1449 
1450         getLog().debug( "Building list of project plugins..." );
1451 
1452         if ( getLog().isDebugEnabled() )
1453         {
1454             StringWriter origModel = new StringWriter();
1455 
1456             try
1457             {
1458                 origModel.write( "Original model:\n" );
1459                 getProject().writeOriginalModel( origModel );
1460                 getLog().debug( origModel.toString() );
1461             }
1462             catch ( IOException e )
1463             {
1464                 // ignore
1465             }
1466         }
1467 
1468         debugVersionMap( "super-pom version map", superPomPluginManagement );
1469         debugVersionMap( "parent version map", parentPluginManagement );
1470 
1471         Map<String, String> excludePluginManagement = new HashMap<String, String>( superPomPluginManagement );
1472         excludePluginManagement.putAll( parentPluginManagement );
1473 
1474         debugVersionMap( "aggregate version map", excludePluginManagement );
1475 
1476         excludePluginManagement.keySet().removeAll( pluginsWithVersionsSpecified );
1477 
1478         debugVersionMap( "final aggregate version map", excludePluginManagement );
1479 
1480         Model originalModel;
1481         try
1482         {
1483             originalModel =
1484                 modelInterpolator.interpolate( getProject().getOriginalModel(), getProject().getBasedir(),
1485                                                new DefaultProjectBuilderConfiguration().setExecutionProperties( getProject().getProperties() ),
1486                                                true );
1487         }
1488         catch ( ModelInterpolationException e )
1489         {
1490             throw new MojoExecutionException( e.getMessage(), e );
1491         }
1492         try
1493         {
1494             addProjectPlugins( plugins, originalModel.getBuild().getPluginManagement().getPlugins(),
1495                                excludePluginManagement );
1496         }
1497         catch ( NullPointerException e )
1498         {
1499             // guess there are no plugins here
1500         }
1501         debugPluginMap( "after adding local pluginManagement", plugins );
1502 
1503         try
1504         {
1505             List<Plugin> lifecyclePlugins = new ArrayList<>( getLifecyclePlugins( getProject() ).values() );
1506             for ( Iterator<Plugin> i = lifecyclePlugins.iterator(); i.hasNext(); )
1507             {
1508                 Plugin lifecyclePlugin = i.next();
1509                 if ( lifecyclePlugin.getVersion() != null )
1510                 {
1511                     // version comes from lifecycle, therefore cannot modify
1512                     i.remove();
1513                 }
1514                 else
1515                 {
1516                     // lifecycle leaves version open
1517                     String parentVersion = parentPluginManagement.get( lifecyclePlugin.getKey() );
1518                     if ( parentVersion != null )
1519                     {
1520                         // parent controls version
1521                         i.remove();
1522                     }
1523                 }
1524             }
1525             addProjectPlugins( plugins, lifecyclePlugins, parentPluginManagement );
1526 
1527             debugPluginMap( "after adding lifecycle plugins", plugins );
1528         }
1529         catch ( NullPointerException e )
1530         {
1531             // using maven 3.x or newer
1532         }
1533 
1534         try
1535         {
1536             List<Plugin> buildPlugins = new ArrayList<>( originalModel.getBuild().getPlugins() );
1537             for ( Iterator<Plugin> i = buildPlugins.iterator(); i.hasNext(); )
1538             {
1539                 Plugin buildPlugin = i.next();
1540                 if ( buildPlugin.getVersion() == null )
1541                 {
1542                     String parentVersion = parentPluginManagement.get( buildPlugin.getKey() );
1543                     if ( parentVersion != null )
1544                     {
1545                         // parent controls version
1546                         i.remove();
1547                     }
1548                 }
1549             }
1550             addProjectPlugins( plugins, buildPlugins, parentBuildPlugins );
1551         }
1552         catch ( NullPointerException e )
1553         {
1554             // guess there are no plugins here
1555         }
1556         debugPluginMap( "after adding build plugins", plugins );
1557 
1558         try
1559         {
1560             List<ReportPlugin> reportPlugins = new ArrayList<>( originalModel.getReporting().getPlugins() );
1561             for ( Iterator<ReportPlugin> i = reportPlugins.iterator(); i.hasNext(); )
1562             {
1563                 ReportPlugin reportPlugin = i.next();
1564                 if ( reportPlugin.getVersion() == null )
1565                 {
1566                     String parentVersion = parentPluginManagement.get( reportPlugin.getKey() );
1567                     if ( parentVersion != null )
1568                     {
1569                         // parent controls version
1570                         i.remove();
1571                     }
1572                 }
1573             }
1574             addProjectPlugins( plugins, toPlugins( reportPlugins ), parentReportPlugins );
1575         }
1576         catch ( NullPointerException e )
1577         {
1578             // guess there are no plugins here
1579         }
1580         debugPluginMap( "after adding reporting plugins", plugins );
1581 
1582         for ( Profile profile : originalModel.getProfiles() )
1583         {
1584             try
1585             {
1586                 addProjectPlugins( plugins, profile.getBuild().getPluginManagement().getPlugins(),
1587                                    excludePluginManagement );
1588             }
1589             catch ( NullPointerException e )
1590             {
1591                 // guess there are no plugins here
1592             }
1593             debugPluginMap( "after adding build pluginManagement for profile " + profile.getId(), plugins );
1594 
1595             try
1596             {
1597                 addProjectPlugins( plugins, profile.getBuild().getPlugins(), parentBuildPlugins );
1598             }
1599             catch ( NullPointerException e )
1600             {
1601                 // guess there are no plugins here
1602             }
1603             debugPluginMap( "after adding build plugins for profile " + profile.getId(), plugins );
1604 
1605             try
1606             {
1607                 addProjectPlugins( plugins, toPlugins( profile.getReporting().getPlugins() ), parentReportPlugins );
1608             }
1609             catch ( NullPointerException e )
1610             {
1611                 // guess there are no plugins here
1612             }
1613             debugPluginMap( "after adding reporting plugins for profile " + profile.getId(), plugins );
1614         }
1615         Set<Plugin> result = new TreeSet<>( new PluginComparator() );
1616         result.addAll( plugins.values() );
1617         return result;
1618     }

1620     /**
1621      * Adds those project plugins which are not inherited from the parent definitions to the list of plugins.
1622      *
1623      * @param plugins The list of plugins.
1624      * @param projectPlugins The project's plugins.
1625      * @param parentDefinitions The parent plugin definitions.
1626      * @since 1.0-alpha-1
1627      */
1628     private void addProjectPlugins( Map<String, Plugin> plugins, Collection<Plugin> projectPlugins,
1629                                     Map<String, String> parentDefinitions )
1630     {
1631         for ( Plugin plugin : projectPlugins )
1632         {
1633             String coord = plugin.getKey();
1634             String version = plugin.getVersion();
1635             String parentVersion = parentDefinitions.get( coord );
1636             if ( version == null
1637                 && ( !plugins.containsKey( coord ) || plugins.get( coord ).getVersion() == null )
1638                 && parentVersion != null )
1639             {
1640                 Plugin parentPlugin = new Plugin();
1641                 parentPlugin.setGroupId( plugin.getGroupId() );
1642                 parentPlugin.setArtifactId( plugin.getArtifactId() );
1643                 parentPlugin.setVersion( parentVersion );
1644                 plugins.put( coord, parentPlugin );
1645             }
1646             else if ( parentVersion == null || !parentVersion.equals( version ) )
1647             {
1648                 if ( !plugins.containsKey( coord ) || plugins.get( coord ).getVersion() == null )
1649                 {
1650                     plugins.put( coord, plugin );
1651                 }
1652             }
1653             if ( !plugins.containsKey( coord ) )
1654             {
1655                 plugins.put( coord, plugin );
1656             }
1657         }
1658     }
