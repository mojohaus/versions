47  /**
48   * Set a property to a given version without any sanity checks. Please be careful this can lead to changes which might
49   * not build anymore. The sanity checks are done by other goals like <code>update-properties</code> or
50   * <code>update-property</code> etc. they are not done here. So use this goal with care.
51   *
52   * @author Karl Heinz Marbaise
53   * @since 2.5
54   */
55  @Mojo( name = "set-property", threadSafe = true )
56  public class SetPropertyMojo
57      extends AbstractVersionsUpdaterMojo
58  {
59  
60      // ------------------------------ FIELDS ------------------------------
61  
62      /**
63       * A property to update.
64       * You can also specify multiple property names separated by "," which are all set to the same new version.
65       */
66      @Parameter( property = "property" )
67      private String property = null;
68  
69      /**
70       * The new version to set the property.
71       */
72      @Parameter( property = "newVersion" )
73      private String newVersion = null;
74  
75      /**
76       * Whether properties linking versions should be auto-detected or not.
77       */
78      @Parameter( property = "autoLinkItems", defaultValue = "true" )
79      private boolean autoLinkItems;
80  
81      /**
82       * A property file name containing: property=value, to update several properties at the same time.
83       * If 'property' and 'newVersion' are also used, they will be ignored.
84       *
85       * @since 2.9
86       */
87  
88      @Parameter( property = "propertiesVersionsFile" )
89      private String propertiesVersionsFile;
90  
91      @Inject
92      public SetPropertyMojo( RepositorySystem repositorySystem,
93                                  MavenProjectBuilder projectBuilder,
94                                  ArtifactMetadataSource artifactMetadataSource,
95                                  WagonManager wagonManager,
96                                  ArtifactResolver artifactResolver )
97      {
98          super( repositorySystem, projectBuilder, artifactMetadataSource, wagonManager, artifactResolver );
99      }
100 
101     /**
102      * @param pom the pom to update.
103      * @throws MojoExecutionException when things go wrong
104      * @throws MojoFailureException   when things go wrong in a very bad way
105      * @throws XMLStreamException     when things go wrong with XML streaming
106      * @see AbstractVersionsUpdaterMojo#update(ModifiedPomXMLEventReader)
107      */
108     protected void update( ModifiedPomXMLEventReader pom )
109         throws MojoExecutionException, MojoFailureException, XMLStreamException
110     {
111         Property[] propertiesConfig;
112         String properties;
113         if ( !isEmpty( propertiesVersionsFile ) )
114         {
115             logWrongConfigWarning();
116             getLog().debug( "Reading properties and versions to update from file: " + propertiesVersionsFile );
117             PropertiesVersionsFileReader reader = new PropertiesVersionsFileReader( propertiesVersionsFile );
118             try
119             {
120                 reader.read();
121             }
122             catch ( IOException e )
123             {
124                 getLog().error( "Unable to read property file  " + propertiesVersionsFile
125                                     + ". re-run with -X option for more details." );
126                 getLog().debug( "Error while reading  property file " + propertiesVersionsFile, e );
127                 throw new MojoFailureException( "Unable to read property file " + propertiesVersionsFile );
128             }
129             propertiesConfig = reader.getPropertiesConfig();
130             properties = reader.getProperties();
131         }
132         else if ( !isEmpty( property ) )
133         {
134             getLog().debug( "Reading properties and versions to update from property and newVersion " );
135             propertiesConfig = Arrays.stream( split( property, "," ) ).map(
136                     prp ->
137                     {
138                         Property propertyConfig = new Property( prp );
139                         propertyConfig.setVersion( newVersion );
140                         return propertyConfig;
141                     } )
142                 .toArray( Property[]::new );
143             properties = property;
144         }
145         else
146         {
147             throw new MojoExecutionException(
148                 "Please provide either 'property' or 'propertiesVersionsFile' parameter." );
149         }
150         update( pom, propertiesConfig, properties );
151     }
152 
153     private void update( ModifiedPomXMLEventReader pom, Property[] propertiesConfig, String properties )
154         throws MojoExecutionException, XMLStreamException
155     {
156         Map<Property, PropertyVersions> propertyVersions =
157             this.getHelper().getVersionPropertiesMap( getProject(), propertiesConfig, properties, "",
158                                                       autoLinkItems );
159         for ( Map.Entry<Property, PropertyVersions> entry : propertyVersions.entrySet() )
160         {
161             Property currentProperty = entry.getKey();
162             PropertyVersions version = entry.getValue();
163             String newVersionGiven = currentProperty.getVersion();
164 
165             final String currentVersion = getProject().getProperties().getProperty( currentProperty.getName() );
166             if ( currentVersion == null )
167             {
168                 continue;
169             }
170             PomHelper.setPropertyVersion( pom, version.getProfileId(), currentProperty.getName(), 
171                     defaultString( newVersionGiven ) );
172         }
173     }
174 
175     private void logWrongConfigWarning()
176     {
177         if ( !isEmpty( property ) )
178         {
179             getLog().warn( "-Dproperty provided but will be ignored as -DpropertiesVersionsFile is used" );
180         }
181         if ( !isEmpty( newVersion ) )
182         {
183             getLog().warn( "-DnewVersion provided but will be ignored as -DpropertiesVersionsFile is used" );
184         }
185     }
