22  import org.apache.maven.artifact.versioning.ArtifactVersion;
23  import org.codehaus.mojo.versions.api.Segment;
24  
25  /**
26   * Represents an invalid segment being identified within a version.
27   */
28  public class InvalidSegmentException extends Exception
29  {
30      private final Segment segment;
31  
32      private final int segmentCount;
33  
34      private final ArtifactVersion version;
35  
36      /**
37       * Constructs the exception object
38       *
39       * @param segment the invalid segment index.
40       * @param segmentCount the number of segments.
41       * @param version the version object.
42       */
43      public InvalidSegmentException( Segment segment, int segmentCount, ArtifactVersion version )
44      {
45          super( String.format( "Invalid segment %s for the %d segment version: '%s'", segment.toString(), segmentCount,
46                  version.toString() ) );
47          this.segment = segment;
48          this.segmentCount = segmentCount;
49          this.version = version;
50      }
51  
52      /**
53       * @return segment
54       */
55      public Segment getSegment()
56      {
57          return segment;
58      }
59  
60      /**
61       * @return segment count
62       */
63      public int getSegmentCount()
64      {
65          return segmentCount;
66      }
67  
68      /**
69       * @return version object
70       */
71      public ArtifactVersion getVersion()
72      {
73          return version;
74      }
75  }
