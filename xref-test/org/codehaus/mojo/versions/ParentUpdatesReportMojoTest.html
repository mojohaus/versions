47  /**
48   * Basic tests for {@linkplain ParentUpdatesReportMojo}.
49   *
50   * @author Andrzej Jarmoniuk
51   */
52  public class ParentUpdatesReportMojoTest
53  {
54      private static final I18N MOCK_I18N = mockI18N();
55      @Test
56      public void testAllowSnapshots() throws IOException, MavenReportException
57      {
58          OutputStream os = new ByteArrayOutputStream();
59          SinkFactory sinkFactory = new Xhtml5SinkFactory();
60          new ParentUpdatesReportMojo( MOCK_I18N, mockRepositorySystem(), null,
61                  mockArtifactMetadataSource( new HashMap<String, String[]>()
62                  {{
63                      put( "default-artifact", new String[] {"1.0.0", "1.0.1", "1.1.0", "2.0.0", "2.0.1-SNAPSHOT"} );
64                  }} ), null, new ReportRendererFactoryImpl( MOCK_I18N ) )
65          {{
66              allowSnapshots = true;
67              project = new MavenProject( new Model()
68              {{
69                  setGroupId( "default-group" );
70                  setArtifactId( "child-artifact" );
71                  setVersion( "1.0.0" );
72              }} );
73              project.setParent( new MavenProject( new Model()
74              {{
75                  setGroupId( "default-group" );
76                  setArtifactId( "default-artifact" );
77                  setVersion( "1.0.0" );
78              }} ) );
79              reactorProjects = new ArrayList<>();
80              project.setParentArtifact( new DefaultArtifact( project.getParent().getGroupId(),
81                      project.getParent().getArtifactId(), project.getParent().getVersion(),
82                      Artifact.SCOPE_COMPILE, "pom", "default", null ) );
83          }}.generate( sinkFactory.createSink( os ), sinkFactory, Locale.getDefault() );
84  
85          String output = os.toString();
86          assertThat( output, allOf( containsString( "1.0.0" ),
87                  containsString( "1.0.1" ), containsString( "1.1.0" ),
88                  containsString( "2.0.0" ), containsString( "2.0.1-SNAPSHOT" ) ) );
89      }
