22  /**
23   * Utility methods to help with regex manipulation.
24   *
25   * @author Stephen Connolly
26   * @since 1.0-alpha-3
27   */
28  public final class RegexUtils
29  {
30      /**
31       * The end of a regex literal sequence.
32       *
33       * @since 1.0-alpha-1
34       */
35      public static final String REGEX_QUOTE_END = "\\E";
36  
37      /**
38       * The start of a regex literal sequence.
39       *
40       * @since 1.0-alpha-1
41       */
42      public static final String REGEX_QUOTE_START = "\\Q";
43  
44      /**
45       * Escape the escapes.
46       *
47       * @since 1.0-alpha-1
48       */
49      public static final String REGEX_QUOTE_END_ESCAPED = REGEX_QUOTE_END + '\\' + REGEX_QUOTE_END + REGEX_QUOTE_START;
50  
51      private RegexUtils()
52      {
53          throw new IllegalAccessError( "Utility classes should never be instantiated" );
54      }
55  
56      /**
57       * Takes a string and returns the regex that will match that string exactly.
58       *
59       * @param s The string to match.
60       * @return The regex that will match the string exactly.
61       * @since 1.0-alpha-1
62       */
63      public static String quote( String s )
64      {
65          int i = s.indexOf( REGEX_QUOTE_END );
66          if ( i == -1 )
67          {
68              // we're safe as nobody has a crazy \E in the string
69              return REGEX_QUOTE_START + s + REGEX_QUOTE_END;
70          }
71  
72          // damn there's at least one \E in the string
73          StringBuilder sb = new StringBuilder( s.length() + 32 );
74          // each escape-escape takes 10 chars...
75          // hope there's less than 4 of them
76  
77          sb.append( REGEX_QUOTE_START );
78          int pos = 0;
79          do
80          {
81              // we are safe from pos to i
82              sb.append( s.substring( pos, i ) );
83              // now escape-escape
84              sb.append( REGEX_QUOTE_END_ESCAPED );
85              // move the working start
86              pos = i + REGEX_QUOTE_END.length();
87              i = s.indexOf( REGEX_QUOTE_END, pos );
88          }
89          while ( i != -1 );
90  
91          sb.append( s.substring( pos, s.length() ) );
92          sb.append( REGEX_QUOTE_END );
93  
94          return sb.toString();
95      }

97      public static int getWildcardScore( String wildcardRule )
98      {
99          int score = 0;
100         for ( int i = 0; i < wildcardRule.length(); i++ )
101         {
102             char c = wildcardRule.charAt( i );
103             if ( c == '?' )
104             {
105                 score++;
106             }
107             else if ( c == '*' )
108             {
109                 score += 1000;
110             }
111         }
112         return score;
113     }

115     /**
116      * Converts a wildcard rule to a regex rule.
117      *
118      * @param wildcardRule the wildcard rule.
119      * @param exactMatch <code>true</code> results in an regex that will match the entire string, while
120      *            <code>false</code> will match the start of the string.
121      * @return The regex rule.
122      */
123     public static String convertWildcardsToRegex( String wildcardRule, boolean exactMatch )
124     {
125         StringBuilder regex = new StringBuilder();
126         int index = 0;
127         final int len = wildcardRule.length();
128         while ( index < len )
129         {
130             final int nextQ = wildcardRule.indexOf( '?', index );
131             final int nextS = wildcardRule.indexOf( '*', index );
132             if ( nextQ == -1 && nextS == -1 )
133             {
134                 regex.append( quote( wildcardRule.substring( index ) ) );
135                 break;
136             }
137             int nextIndex;
138             if ( nextQ == -1 )
139             {
140                 nextIndex = nextS;
141             }
142             else if ( nextS == -1 )
143             {
144                 nextIndex = nextQ;
145             }
146             else
147             {
148                 nextIndex = Math.min( nextQ, nextS );
149             }
150             if ( index < nextIndex )
151             {
152                 // we have some characters to match
153                 regex.append( quote( wildcardRule.substring( index, nextIndex ) ) );
154             }
155             char c = wildcardRule.charAt( nextIndex );
156             if ( c == '?' )
157             {
158                 regex.append( '.' );
159             }
160             else
161             {
162                 regex.append( ".*" );
163             }
164             index = nextIndex + 1;
165         }
166         if ( !exactMatch )
167         {
168             regex.append( ".*" );
169         }
170         return regex.toString();
171     }
