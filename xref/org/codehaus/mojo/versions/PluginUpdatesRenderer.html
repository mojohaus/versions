43  public class PluginUpdatesRenderer
44      extends AbstractVersionsReportRenderer
45  {
46      private final Map<Plugin, PluginUpdatesDetails> pluginUpdates;
47  
48      private final Map<Plugin, PluginUpdatesDetails> pluginManagementUpdates;
49  
50      public PluginUpdatesRenderer( Sink sink, I18N i18n, String bundleName, Locale locale,
51                                    Map<Plugin, PluginUpdatesDetails> pluginUpdates,
52                                    Map<Plugin, PluginUpdatesDetails> pluginManagementUpdates )
53      {
54          super( sink, bundleName, i18n, locale );
55          this.pluginUpdates = pluginUpdates;
56          this.pluginManagementUpdates = pluginManagementUpdates;
57      }
58  
59      protected void renderBody()
60      {
61          Map<Plugin, PluginUpdatesDetails> allUpdates = new TreeMap<>( new PluginComparator() );
62          allUpdates.putAll( pluginManagementUpdates );
63          allUpdates.putAll( pluginUpdates );
64  
65          sink.section1();
66          sink.sectionTitle1();
67          sink.text( getText( "report.overview.title" ) );
68          sink.sectionTitle1_();
69          sink.paragraph();
70          sink.text( getText( "report.overview.text" ) );
71          sink.paragraph_();
72  
73          renderSummaryTotalsTable( allUpdates );
74  
75          renderSummaryTable( "report.overview.pluginManagement", pluginManagementUpdates,
76                              "report.overview.noPluginManagement" );
77  
78          renderSummaryTable( "report.overview.plugin", pluginUpdates, "report.overview.noPlugin" );
79  
80          sink.section1_();
81  
82          sink.section1();
83          sink.sectionTitle1();
84          sink.text( getText( "report.detail.title" ) );
85          sink.sectionTitle1_();
86          sink.paragraph();
87          sink.text( getText( "report.detail.text" ) );
88          sink.paragraph_();
89  
90          for ( final Map.Entry<Plugin, PluginUpdatesDetails> entry : allUpdates.entrySet() )
91          {
92              renderPluginDetail( entry.getKey(), entry.getValue() );
93          }
94          sink.section1_();
95  
96      }
97  
98      private void renderSummaryTotalsTable( Map<Plugin, PluginUpdatesDetails> allUpdates )
99      {
100         int numInc = 0;
101         int numMin = 0;
102         int numMaj = 0;
103         int numAny = 0;
104         int numCur = 0;
105         int numDep = 0;
106         for ( PluginUpdatesDetails pluginDetails : allUpdates.values() )
107         {
108             ArtifactVersions details = pluginDetails.getArtifactVersions();
109             if ( details.getOldestUpdate( UpdateScope.SUBINCREMENTAL ) != null )
110             {
111                 numAny++;
112             }
113             else if ( details.getOldestUpdate( UpdateScope.INCREMENTAL ) != null )
114             {
115                 numInc++;
116             }
117             else if ( details.getOldestUpdate( UpdateScope.MINOR ) != null )
118             {
119                 numMin++;
120             }
121             else if ( details.getOldestUpdate( UpdateScope.MAJOR ) != null )
122             {
123                 numMaj++;
124             }
125             else
126             {
127                 numCur++;
128             }
129             if ( pluginDetails.isDependencyUpdateAvailable() )
130             {
131                 numDep++;
132             }
133         }
134         sink.table();
135         sink.tableRow();
136         sink.tableCell();
137         renderSuccessIcon();
138         sink.tableCell_();
139         sink.tableCell();
140         sink.text( getText( "report.overview.numUpToDate" ) );
141         sink.tableCell_();
142         sink.tableCell();
143         sink.text( Integer.toString( numCur ) );
144         sink.tableCell_();
145         sink.tableRow_();
146         sink.tableRow();
147         sink.tableCell();
148         renderWarningIcon();
149         sink.tableCell_();
150         sink.tableCell();
151         sink.text( getText( "report.overview.numNewerVersionAvailable" ) );
152         sink.tableCell_();
153         sink.tableCell();
154         sink.text( Integer.toString( numAny ) );
155         sink.tableCell_();
156         sink.tableRow_();
157         sink.tableRow();
158         sink.tableCell();
159         renderWarningIcon();
160         sink.tableCell_();
161         sink.tableCell();
162         sink.text( getText( "report.overview.numNewerIncrementalAvailable" ) );
163         sink.tableCell_();
164         sink.tableCell();
165         sink.text( Integer.toString( numInc ) );
166         sink.tableCell_();
167         sink.tableRow_();
168         sink.tableRow();
169         sink.tableCell();
170         renderWarningIcon();
171         sink.tableCell_();
172         sink.tableCell();
173         sink.text( getText( "report.overview.numNewerMinorAvailable" ) );
174         sink.tableCell_();
175         sink.tableCell();
176         sink.text( Integer.toString( numMin ) );
177         sink.tableCell_();
178         sink.tableRow_();
179         sink.tableRow();
180         sink.tableCell();
181         renderWarningIcon();
182         sink.tableCell_();
183         sink.tableCell();
184         sink.text( getText( "report.overview.numNewerMajorAvailable" ) );
185         sink.tableCell_();
186         sink.tableCell();
187         sink.text( Integer.toString( numMaj ) );
188         sink.tableCell_();
189         sink.tableRow_();
190         sink.tableRow();
191         sink.tableCell();
192         renderWarningIcon();
193         sink.tableCell_();
194         sink.tableCell();
195         sink.text( getText( "report.overview.numNewerDependenciesAvailable" ) );
196         sink.tableCell_();
197         sink.tableCell();
198         sink.text( Integer.toString( numDep ) );
199         sink.tableCell_();
200         sink.tableRow_();
201         sink.table_();
202     }

204     private void renderSummaryTable( String titleKey, Map<Plugin, PluginUpdatesDetails> contents, String emptyKey )
205     {
206         sink.section2();
207         sink.sectionTitle2();
208         sink.text( getText( titleKey ) );
209         sink.sectionTitle2_();
210 
211         if ( contents.isEmpty() )
212         {
213             sink.paragraph();
214             sink.text( getText( emptyKey ) );
215             sink.paragraph_();
216         }
217         else
218         {
219             sink.table();
220             renderSummaryTableHeader();
221             for ( final Map.Entry<Plugin, PluginUpdatesDetails> entry : contents.entrySet() )
222             {
223                 renderPluginSummary( entry.getKey(), entry.getValue() );
224             }
225             renderSummaryTableHeader();
226             sink.table_();
227         }
228         sink.section2_();
229     }

264     private void renderPluginSummary( Plugin plugin, PluginUpdatesDetails details )
265     {
266         sink.tableRow();
267         sink.tableCell();
268         if ( !details.isUpdateAvailable() )
269         {
270             renderSuccessIcon();
271         }
272         else
273         {
274             renderWarningIcon();
275         }
276         sink.tableCell_();
277         sink.tableCell();
278         sink.text( plugin.getGroupId() );
279         sink.tableCell_();
280         sink.tableCell();
281         sink.text( plugin.getArtifactId() );
282         sink.tableCell_();
283         sink.tableCell();
284         if ( !details.isArtifactUpdateAvailable() )
285         {
286             safeBold();
287         }
288         sink.text( plugin.getVersion() );
289         if ( !details.isArtifactUpdateAvailable() )
290         {
291             safeBold_();
292         }
293         sink.tableCell_();
294 
295         sink.tableCell();
296         if ( details.getArtifactVersions().getOldestUpdate( UpdateScope.SUBINCREMENTAL ) != null )
297         {
298             safeBold();
299             sink.text( details.getArtifactVersions().getOldestUpdate( UpdateScope.SUBINCREMENTAL ).toString() );
300             safeBold_();
301         }
302         sink.tableCell_();
303 
304         sink.tableCell();
305         if ( details.getArtifactVersions().getOldestUpdate( UpdateScope.INCREMENTAL ) != null )
306         {
307             safeBold();
308             sink.text( details.getArtifactVersions().getOldestUpdate( UpdateScope.INCREMENTAL ).toString() );
309             safeBold_();
310         }
311         sink.tableCell_();
312 
313         sink.tableCell();
314         if ( details.getArtifactVersions().getOldestUpdate( UpdateScope.MINOR ) != null )
315         {
316             safeBold();
317             sink.text( details.getArtifactVersions().getOldestUpdate( UpdateScope.MINOR ).toString() );
318             safeBold_();
319         }
320         sink.tableCell_();
321 
322         sink.tableCell();
323         if ( details.getArtifactVersions().getOldestUpdate( UpdateScope.MAJOR ) != null )
324         {
325             safeBold();
326             sink.text( details.getArtifactVersions().getOldestUpdate( UpdateScope.MAJOR ).toString() );
327             safeBold_();
328         }
329         sink.tableCell_();
330 
331         sink.tableCell();
332         if ( details.isDependencyUpdateAvailable() )
333         {
334             renderWarningIcon();
335         }
336         else
337         {
338             renderSuccessIcon();
339         }
340         sink.tableCell_();
341 
342         sink.tableRow_();
343     }

345     private void renderPluginDetail( Plugin plugin, PluginUpdatesDetails details )
346     {
347         final SinkEventAttributes headerAttributes = new SinkEventAttributeSet();
348         headerAttributes.addAttribute( SinkEventAttributes.WIDTH, "20%" );
349         final SinkEventAttributes cellAttributes = new SinkEventAttributeSet();
350         headerAttributes.addAttribute( SinkEventAttributes.WIDTH, "80%" );
351         sink.section2();
352         sink.sectionTitle2();
353         sink.text( MessageFormat.format( getText( "report.plugin" ), new Object[] {
354             ArtifactUtils.versionlessKey( plugin.getGroupId(), plugin.getArtifactId() ) } ) );
355         sink.sectionTitle2_();
356         sink.table();
357         sink.tableRows( new int[] { Sink.JUSTIFY_RIGHT, Sink.JUSTIFY_LEFT }, false );
358         sink.tableRow();
359         sink.tableHeaderCell( headerAttributes );
360         sink.text( getText( "report.status" ) );
361         sink.tableHeaderCell_();
362         sink.tableCell( cellAttributes );
363         ArtifactVersion[] versions = details.getArtifactVersions().getAllUpdates( UpdateScope.ANY );
364         if ( details.getArtifactVersions().getOldestUpdate( UpdateScope.SUBINCREMENTAL ) != null )
365         {
366             renderWarningIcon();
367             sink.nonBreakingSpace();
368             sink.text( getText( "report.otherUpdatesAvailable" ) );
369         }
370         else if ( details.getArtifactVersions().getOldestUpdate( UpdateScope.INCREMENTAL ) != null )
371         {
372             renderWarningIcon();
373             sink.nonBreakingSpace();
374             sink.text( getText( "report.incrementalUpdatesAvailable" ) );
375         }
376         else if ( details.getArtifactVersions().getOldestUpdate( UpdateScope.MINOR ) != null )
377         {
378             renderWarningIcon();
379             sink.nonBreakingSpace();
380             sink.text( getText( "report.minorUpdatesAvailable" ) );
381         }
382         else if ( details.getArtifactVersions().getOldestUpdate( UpdateScope.MAJOR ) != null )
383         {
384             renderWarningIcon();
385             sink.nonBreakingSpace();
386             sink.text( getText( "report.majorUpdatesAvailable" ) );
387         }
388         else
389         {
390             renderSuccessIcon();
391             sink.nonBreakingSpace();
392             sink.text( getText( "report.noUpdatesAvailable" ) );
393         }
394         sink.tableCell_();
395         sink.tableRow_();
396         sink.tableRow();
397         sink.tableHeaderCell( headerAttributes );
398         sink.text( getText( "report.groupId" ) );
399         sink.tableHeaderCell_();
400         sink.tableCell( cellAttributes );
401         sink.text( plugin.getGroupId() );
402         sink.tableCell_();
403         sink.tableRow_();
404         sink.tableRow();
405         sink.tableHeaderCell( headerAttributes );
406         sink.text( getText( "report.artifactId" ) );
407         sink.tableHeaderCell_();
408         sink.tableCell( cellAttributes );
409         sink.text( plugin.getArtifactId() );
410         sink.tableCell_();
411         sink.tableRow_();
412         sink.tableRow();
413         sink.tableHeaderCell( headerAttributes );
414         sink.text( getText( "report.currentVersion" ) );
415         sink.tableHeaderCell_();
416         sink.tableCell( cellAttributes );
417         sink.text( plugin.getVersion() );
418         sink.tableCell_();
419         sink.tableRow_();
420         if ( versions.length > 0 )
421         {
422             sink.tableRow();
423             sink.tableHeaderCell( headerAttributes );
424             sink.text( getText( "report.updateVersions" ) );
425             sink.tableHeaderCell_();
426             sink.tableCell( cellAttributes );
427             for ( int i = 0; i < versions.length; i++ )
428             {
429                 if ( i > 0 )
430                 {
431                     sink.lineBreak();
432                 }
433                 boolean bold = equals( versions[i],
434                                        details.getArtifactVersions().getOldestUpdate( UpdateScope.SUBINCREMENTAL ) )
435                     || equals( versions[i], details.getArtifactVersions().getOldestUpdate( UpdateScope.INCREMENTAL ) )
436                     || equals( versions[i], details.getArtifactVersions().getNewestUpdate( UpdateScope.INCREMENTAL ) )
437                     || equals( versions[i], details.getArtifactVersions().getOldestUpdate( UpdateScope.MINOR ) )
438                     || equals( versions[i], details.getArtifactVersions().getNewestUpdate( UpdateScope.MINOR ) )
439                     || equals( versions[i], details.getArtifactVersions().getOldestUpdate( UpdateScope.MAJOR ) )
440                     || equals( versions[i], details.getArtifactVersions().getNewestUpdate( UpdateScope.MAJOR ) );
441                 if ( bold )
442                 {
443                     safeBold();
444                 }
445                 sink.text( versions[i].toString() );
446                 if ( bold )
447                 {
448                     safeBold_();
449                     sink.nonBreakingSpace();
450                     safeItalic();
451                     if ( equals( versions[i],
452                                  details.getArtifactVersions().getOldestUpdate( UpdateScope.SUBINCREMENTAL ) ) )
453                     {
454                         sink.text( getText( "report.nextVersion" ) );
455                     }
456                     else if ( equals( versions[i],
457                                       details.getArtifactVersions().getOldestUpdate( UpdateScope.INCREMENTAL ) ) )
458                     {
459                         sink.text( getText( "report.nextIncremental" ) );
460                     }
461                     else if ( equals( versions[i],
462                                       details.getArtifactVersions().getNewestUpdate( UpdateScope.INCREMENTAL ) ) )
463                     {
464                         sink.text( getText( "report.latestIncremental" ) );
465                     }
466                     else if ( equals( versions[i],
467                                       details.getArtifactVersions().getOldestUpdate( UpdateScope.MINOR ) ) )
468                     {
469                         sink.text( getText( "report.nextMinor" ) );
470                     }
471                     else if ( equals( versions[i],
472                                       details.getArtifactVersions().getNewestUpdate( UpdateScope.MINOR ) ) )
473                     {
474                         sink.text( getText( "report.latestMinor" ) );
475                     }
476                     else if ( equals( versions[i],
477                                       details.getArtifactVersions().getOldestUpdate( UpdateScope.MAJOR ) ) )
478                     {
479                         sink.text( getText( "report.nextMajor" ) );
480                     }
481                     else if ( equals( versions[i],
482                                       details.getArtifactVersions().getNewestUpdate( UpdateScope.MAJOR ) ) )
483                     {
484                         sink.text( getText( "report.latestMajor" ) );
485                     }
486 
487                     safeItalic_();
488                 }
489             }
490             sink.tableCell_();
491             sink.tableRow_();
492         }
493         sink.tableRows_();
494         sink.table_();
495 
496         if ( !details.getDependencyVersions().isEmpty() )
497         {
498             sink.section3();
499             sink.sectionTitle3();
500             sink.text( MessageFormat.format( getText( "report.pluginDependencies" ), new Object[] {
501                 ArtifactUtils.versionlessKey( plugin.getGroupId(), plugin.getArtifactId() ) } ) );
502             sink.sectionTitle3_();
503 
504             renderDependencySummaryTable( details.getDependencyVersions(), false, true, true );
505 
506             sink.section3_();
507 
508             for ( Iterator i = details.getDependencyVersions().entrySet().iterator(); i.hasNext(); )
509             {
510                 Map.Entry entry = (Map.Entry) i.next();
511                 renderDependencyDetail( (Dependency) entry.getKey(), (ArtifactVersions) entry.getValue() );
512             }
513         }
514         sink.section2_();
515     }

517     private void renderDependencyDetail( Dependency dependency, ArtifactVersions details )
518     {
519         sink.section3();
520         sink.sectionTitle3();
521         sink.text( MessageFormat.format( getText( "report.pluginDependency" ), new Object[] {
522             ArtifactUtils.versionlessKey( dependency.getGroupId(), dependency.getArtifactId() ) } ) );
523         sink.sectionTitle3_();
524         renderDependencyDetailTable( dependency, details, false, true, true );
525         sink.section3_();
526     }
